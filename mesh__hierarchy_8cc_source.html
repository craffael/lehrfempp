72 void MeshHierarchy::RefineRegular(RefPat ref_pat) {
73  LF_VERIFY_MSG(
74  ref_pat == RefPat::rp_regular || ref_pat == RefPat::rp_barycentric,
75  "Only regular or barycentric uniform refinement possible");
76  // Retrieve the finest mesh in the hierarchy
77  const mesh::Mesh &finest_mesh(*meshes_.back());
78  // Arrays containing refinement information for finest mesh
79  std::vector<PointChildInfo> &finest_point_ci(point_child_infos_.back());
80  std::vector<EdgeChildInfo> &finest_edge_ci(edge_child_infos_.back());
81  std::vector<CellChildInfo> &finest_cell_ci(cell_child_infos_.back());
82 
83  LF_VERIFY_MSG(finest_point_ci.size() == finest_mesh.NumEntities(2),
84  "length mismatch PointChildInfo vector");
85  LF_VERIFY_MSG(finest_edge_ci.size() == finest_mesh.NumEntities(1),
86  "length mismatch EdgeChildInfo vector");
87  LF_VERIFY_MSG(finest_cell_ci.size() == finest_mesh.NumEntities(0),
88  "length mismatch CellChildInfo vector");
89 
90  // Flag all points as to be copied
91  for (const mesh::Entity *point : finest_mesh.Entities(2)) {
92  const lf::base::glb_idx_t point_index = finest_mesh.Index(*point);
93  PointChildInfo &pt_child_info(finest_point_ci[point_index]);
94  // Set the information about a point's children except the child pointer
95  pt_child_info.ref_pat = RefPat::rp_copy;
96  }
97  // Flag all edges as to be split
98  for (const mesh::Entity *edge : finest_mesh.Entities(1)) {
99  const lf::base::glb_idx_t edge_index = finest_mesh.Index(*edge);
100  EdgeChildInfo &ed_child_info(finest_edge_ci[edge_index]);
101  ed_child_info.ref_pat_ = RefPat::rp_split;
102  }
103  // All cells are to be refined regularly
104  for (const mesh::Entity *cell : finest_mesh.Entities(0)) {
105  const lf::base::glb_idx_t cell_index = finest_mesh.Index(*cell);
106  CellChildInfo &cell_child_info(finest_cell_ci[cell_index]);
107  cell_child_info.ref_pat_ = ref_pat;
108  }
109  // With all refinement patterns set, generate the new mesh
110  PerformRefinement();
111  // Finish initialization
112  initGeometryInParent();
113 }

115 void MeshHierarchy::RefineMarked() {
116  // Target the finest mesh
117  const lf::mesh::Mesh &finest_mesh(*meshes_.back());
118  // Arrays containing refinement information for finest mesh
119  std::vector<PointChildInfo> &finest_point_ci(point_child_infos_.back());
120  std::vector<EdgeChildInfo> &finest_edge_ci(edge_child_infos_.back());
121  std::vector<CellChildInfo> &finest_cell_ci(cell_child_infos_.back());
122 
123  LF_VERIFY_MSG(finest_point_ci.size() == finest_mesh.NumEntities(2),
124  "length mismatch PointChildInfo vector");
125  LF_VERIFY_MSG(finest_edge_ci.size() == finest_mesh.NumEntities(1),
126  "length mismatch EdgeChildInfo vector");
127  LF_VERIFY_MSG(finest_cell_ci.size() == finest_mesh.NumEntities(0),
128  "length mismatch CellChildInfo vector");
129 
130  // Flag all points as to be copied
131  for (const mesh::Entity *point : finest_mesh.Entities(2)) {
132  const glb_idx_t point_index = finest_mesh.Index(*point);
133  PointChildInfo &pt_child_info(finest_point_ci[point_index]);
134  // Set the information about a points children except the child pointer
135  pt_child_info.ref_pat = RefPat::rp_copy;
136  }
137 
138  // Set the split refinement patterrn for all marked edges
139  for (const lf::mesh::Entity *edge : finest_mesh.Entities(1)) {
140  LF_VERIFY_MSG(edge->RefEl() == lf::base::RefEl::kSegment(),
141  "Wrong type for an edge");
142  const glb_idx_t edge_index = finest_mesh.Index(*edge);
143  EdgeChildInfo &ed_ci(finest_edge_ci[edge_index]);
144  LF_VERIFY_MSG(ed_ci.ref_pat_ == RefPat::rp_nil,
145  "Edge " << edge_index << " already refined!");
146  if ((edge_marked_.back())[edge_index]) {
147  // Edge is to be refined
148  ed_ci.ref_pat_ = RefPat::rp_split;
149  } else {
150  // Just copy edge
151  ed_ci.ref_pat_ = RefPat::rp_copy;
152  }
153  }
154  // Now all edges are initially marked to be split or copied
155 
156  // To keep the mesh conforming refinement might have to propagate
157  // This is achieved in the following REPEAT ... UNTIL loop.
158  bool refinement_complete;
159  do {
160  refinement_complete = true;
161  // Visit all cells and update their refinement patterns
162  for (const lf::mesh::Entity *cell : finest_mesh.Entities(0)) {
163  // Obtain unique index of current cell
164  const glb_idx_t cell_index = finest_mesh.Index(*cell);
165  // Detailed description of refinement status
166  CellChildInfo &cell_child_info(finest_cell_ci[cell_index]);
167 
168  // Global indices of edges
169  std::array<glb_idx_t, 4> cell_edge_indices{};
170 
171  // Find edges which are marked as split
172  std::array<bool, 4> edge_split{{false, false, false, false}};
173  // Local indices of edges marked as split
174  std::array<sub_idx_t, 4> split_edge_idx{};
175  // Array of references to edge sub-entities of current cell
176  nonstd::span<const lf::mesh::Entity *const> sub_edges =
177  cell->SubEntities(1);
178  const size_type num_edges = cell->RefEl().NumSubEntities(1);
179  LF_VERIFY_MSG(num_edges <= 4, "Too many edges = " << num_edges);
180  // Obtain information about current splitting pattern of
181  // the edges of the cell. Count edges that will be split.
182  size_type split_edge_cnt = 0;
183  for (int k = 0; k < num_edges; k++) {
184  const glb_idx_t edge_index = finest_mesh.Index(*sub_edges[k]);
185  cell_edge_indices[k] = edge_index;
186  edge_split[k] =
187  (finest_edge_ci[edge_index].ref_pat_ == RefPat::rp_split);
188  if (edge_split[k]) {
189  split_edge_idx[split_edge_cnt] = k;
190  split_edge_cnt++;
191  }
192  }
193  switch (cell->RefEl()) {
194  case lf::base::RefEl::kTria(): {
195  // Case of a triangular cell: In this case bisection refinement
196  // is performed starting with the refinement edge.
197  // Local index of refinement edge for the current triangle, also
198  // called the "anchor edge" in the case of repeated bisection
199  const sub_idx_t anchor = refinement_edges_.back()[cell_index];
200  LF_VERIFY_MSG(anchor < 3, "Illegal anchor = " << anchor);
201 
202  // Refinement edge will always be the anchor edge
203  finest_cell_ci[cell_index].anchor_ = anchor;
204  const sub_idx_t mod_0 = anchor;
205  const sub_idx_t mod_1 = (anchor + 1) % 3;
206  const sub_idx_t mod_2 = (anchor + 2) % 3;
207  // Flag tuple indicating splitting status of an edge: true <-> split
208  std::tuple<bool, bool, bool> split_status(
209  {edge_split[mod_0], edge_split[mod_1], edge_split[mod_2]});
210 
211  // Determine updated refinement pattern for triangle depending on the
212  // splitting status of its edges. If the triangle is subdivided, the
213  // refinement must always be split in a first bisection step, even if
214  // it may not have been marked as split. In this case refinement may
215  // spread to neighboring cells and we have to iterative once more.
216  // Therefore the refinement_complete flag has to be set to 'false'.
217 
218  if (split_status ==
219  std::tuple<bool, bool, bool>({false, false, false})) {
220  // No edge to be split: just copy triangle
221  LF_VERIFY_MSG(split_edge_cnt == 0, "Wrong number of split edges");
222  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_copy;
223  } else if (split_status ==
224  std::tuple<bool, bool, bool>({true, false, false})) {
225  // Only refinement edge has to be split by a single bisection
226  // No additional edge will be split
227  LF_VERIFY_MSG(split_edge_cnt == 1, "Wrong number of split edges");
228  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_bisect;
229  } else if (split_status ==
230  std::tuple<bool, bool, bool>({true, true, false})) {
231  // Trisection refinement, no extra splitting of edges
232  LF_VERIFY_MSG(split_edge_cnt == 2, "Wrong number of split edges");
233  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_trisect;
234  } else if (split_status ==
235  std::tuple<bool, bool, bool>({false, true, false})) {
236  // Trisection refinement, triggering splitting of refinement edge
237  LF_VERIFY_MSG(split_edge_cnt == 1, "Wrong number of split edges");
238  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_trisect;
239  finest_edge_ci[cell_edge_indices[anchor]].ref_pat_ =
240  RefPat::rp_split;
241  refinement_complete = false;
242  } else if (split_status ==
243  std::tuple<bool, bool, bool>({true, false, true})) {
244  // Trisection refinement (other side), no extra splitting of edges
245  LF_VERIFY_MSG(split_edge_cnt == 2, "Wrong number of split edges");
246  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_trisect_left;
247  } else if (split_status ==
248  std::tuple<bool, bool, bool>({false, false, true})) {
249  // Trisection refinement (other side), triggering splitting of
250  // refinement edge
251  LF_VERIFY_MSG(split_edge_cnt == 1, "Wrong number of split edges");
252  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_trisect_left;
253  finest_edge_ci[cell_edge_indices[anchor]].ref_pat_ =
254  RefPat::rp_split;
255  refinement_complete = false;
256  } else if (split_status ==
257  std::tuple<bool, bool, bool>({true, true, true})) {
258  // Quadsection refinement, no extra splitting of edges
259  LF_VERIFY_MSG(split_edge_cnt == 3, "Wrong number of split edges");
260  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_quadsect;
261  } else if (split_status ==
262  std::tuple<bool, bool, bool>({false, true, true})) {
263  // Quadsection refinement requiring splitting of refinement edge
264  LF_VERIFY_MSG(split_edge_cnt == 2, "Wrong number of split edges");
265  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_quadsect;
266  finest_edge_ci[cell_edge_indices[anchor]].ref_pat_ =
267  RefPat::rp_split;
268  refinement_complete = false;
269  } else {
270  LF_VERIFY_MSG(false, "Impossible case");
271  }
272  break;
273  } // end case of a triangle
274  case lf::base::RefEl::kQuad(): {
275  // There is no refinement edge for quadrilaterals and so no extra edge
276  // splitting will be necessary. The refinement pattern for a
277  // quadrilateral will be determined from the number of edges split and
278  // their location to each other.
279  switch (split_edge_cnt) {
280  case 0: {
281  // No edge split: quadrilateral has to be copied
282  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_copy;
283  break;
284  }
285  case 1: {
286  // One edge split: trisection refinement of the quadrilateral
287  // Anchor edge is the split edge
288  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_trisect;
289  finest_cell_ci[cell_index].anchor_ = split_edge_idx[0];
290  break;
291  }
292  case 2: {
293  if ((split_edge_idx[1] - split_edge_idx[0]) == 2) {
294  // If the two split edges are opposite to each other, then
295  // bisection of the quadrilateral is the right refinement
296  // pattern.
297  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_bisect;
298  finest_cell_ci[cell_index].anchor_ = split_edge_idx[0];
299  } else {
300  // Tthe two split edges are adjacent, this case can be
301  // accommodated by quadsection refinement. Anchor is the split
302  // edge with the lower index (modulo 4).
303  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_quadsect;
304  if (((split_edge_idx[0] + 1) % 4) == split_edge_idx[1]) {
305  finest_cell_ci[cell_index].anchor_ = split_edge_idx[0];
306  } else if (((split_edge_idx[1] + 1) % 4) == split_edge_idx[0]) {
307  finest_cell_ci[cell_index].anchor_ = split_edge_idx[1];
308  } else {
309  LF_VERIFY_MSG(false,
310  "Quad: impossible situation for 2 split edges");
311  }
312  }
313  break;
314  }
315  case 3: {
316  // Three edges of the quadrilateral are split, which can be
317  // accommodated only by the rp_threeedge refinement pattern
318  // anchor is the edge with the middle index
319  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_threeedge;
320  if (!edge_split[0]) { // Split edges 1,2,3, middle edge 2
321  finest_cell_ci[cell_index].anchor_ = 2;
322  } else if (!edge_split[1]) { // Split edges 0,2,3, middle edge 3
323  finest_cell_ci[cell_index].anchor_ = 3;
324  } else if (!edge_split[2]) { // Split edges 0,1,3, middle edge 0
325  finest_cell_ci[cell_index].anchor_ = 0;
326  } else if (!edge_split[3]) { // Split edges 0,1,2, middle edge 1
327  finest_cell_ci[cell_index].anchor_ = 1;
328  } else {
329  LF_VERIFY_MSG(false, "Inconsistent split pattern");
330  }
331  break;
332  }
333  case 4: {
334  // All edges are split => regular refinement
335  finest_cell_ci[cell_index].ref_pat_ = RefPat::rp_regular;
336  break;
337  }
338  default: {
339  LF_VERIFY_MSG(false, "Illegal number " << split_edge_cnt
340  << " of split edges");
341  break;
342  }
343  } // end switch split_edge_cnt
344  break;
345  } // end case of a quadrilateral
346  default: {
347  LF_VERIFY_MSG(false, "Illegal cell type");
348  break;
349  }
350  } // end switch cell type
351  } // end loop over cells
352  } while (!refinement_complete);
353 
354  // Create finer mesh according to set refinment edges
355  PerformRefinement();
356  // Finish initialization
357  initGeometryInParent();
358 } // end RefineMarked

361 void MeshHierarchy::PerformRefinement() {
362  SPDLOG_LOGGER_DEBUG(Logger(),
363  "Entering MeshHierarchy::PerformRefinement: {} levels",
364  meshes_.size());
365 
366  // This function expects that the refinement patterns stored in the
367  // vectors point_child_infos_, edge_child_infos_ and cell_child_infos_
368  // have been initialized consistently for the finest mesh.
369 
370  // This functions will augement these vectors with information about
371  // the indices of child entities on a newly created finest mesh.
372 
373  // Retrieve the finest mesh in the hierarchy = parent mesh
374  const mesh::Mesh &parent_mesh(*meshes_.back());
375 
376  {
377  // Partly intialized vectors of child information
378  std::vector<PointChildInfo> &pt_child_info(point_child_infos_.back());
379  // ======================================================================
380  // First run through the vertices, create child vertices and register
381  // them with the mesh factory
382  // Store child indices in an auxiliary array
383  size_type new_node_cnt = 0;
384  const Hybrid2DRefinementPattern rp_copy_node(lf::base::RefEl::kPoint(),
385  RefPat::rp_copy);
386  for (const mesh::Entity *node : parent_mesh.Entities(2)) {
387  // Obtain index of node in coarse mesh
388  const lf::base::glb_idx_t node_index = parent_mesh.Index(*node);
389  LF_VERIFY_MSG(node_index < pt_child_info.size(),
390  "Node index " << node_index << " out of range");
391  // Find position of node in physical coordinates
392  const lf::geometry::Geometry &pt_geo(*node->Geometry());
393  if (pt_child_info[node_index].ref_pat != RefPat::rp_nil) {
394  // Generate a node for the fine mesh at the same position
395  std::vector<std::unique_ptr<geometry::Geometry>> pt_child_geo_ptrs(
396  pt_geo.ChildGeometry(rp_copy_node, 0));
397  LF_VERIFY_MSG(pt_child_geo_ptrs.size() == 1,
398  "A point can only have one chile");
399  // Store index of child node
400  pt_child_info[node_index].child_point_idx =
401  mesh_factory_->AddPoint(std::move(pt_child_geo_ptrs[0]));
402  new_node_cnt++;
403  }
404  } // end loop over nodes
405  SPDLOG_LOGGER_DEBUG(Logger(), "{} new nodes added", new_node_cnt);
406 
407  // ======================================================================
408 
409  // Partly intialized vectors of child information
410  std::vector<EdgeChildInfo> &ed_child_info(edge_child_infos_.back());
411  // ======================================================================
412  // Now traverse the edges. Depending on the refinement pattern,
413  // either copy them or split them.
414  // Supplement the refinement information for edges accordingly.
415  size_type new_edge_cnt = 0;
416  for (const mesh::Entity *edge : parent_mesh.Entities(1)) {
417  // Fetch global index of edge
418  lf::base::glb_idx_t edge_index = parent_mesh.Index(*edge);
419 
420  // Get indices of endpoints in parent mesh
421  auto ed_nodes = edge->SubEntities(1);
422  const lf::base::glb_idx_t ed_p0_coarse_idx =
423  parent_mesh.Index(*ed_nodes[0]);
424  const lf::base::glb_idx_t ed_p1_coarse_idx =
425  parent_mesh.Index(*ed_nodes[1]);
426 
427  // Obtain indices of the nodes at the same position in the fine mesh
428  const lf::base::glb_idx_t ed_p0_fine_idx =
429  pt_child_info[ed_p0_coarse_idx].child_point_idx;
430  const lf::base::glb_idx_t ed_p1_fine_idx =
431  pt_child_info[ed_p1_coarse_idx].child_point_idx;
432  // Prepare request of geometry after refinement
433  EdgeChildInfo &edge_ci(ed_child_info[edge_index]);
434  const RefPat edge_refpat(edge_ci.ref_pat_);
435  Hybrid2DRefinementPattern rp(edge->RefEl(), edge_refpat);
436 
437  // Distinguish between different local refinement patterns
438  switch (edge_refpat) {
439  case RefPat::rp_copy: {
440  // Edge has to be duplicated
441  std::vector<std::unique_ptr<lf::geometry::Geometry>> ed_copy(
442  edge->Geometry()->ChildGeometry(rp, 0));
443  LF_VERIFY_MSG(ed_copy.size() == 1,
444  "Copy may create only a single child!");
445  // Register the new edge
446  SPDLOG_LOGGER_TRACE(Logger(), "Copy edge {} new edge [{},{}]",
447  edge_index, ed_p0_fine_idx, ed_p1_fine_idx);
448 
449  // Store index of copy of edge on finer mesh
450  edge_ci.child_edge_idx.push_back(
451  mesh_factory_->AddEntity(edge->RefEl(),
452  std::array<lf::base::glb_idx_t, 2>{
453  {ed_p0_fine_idx, ed_p1_fine_idx}},
454  std::move(ed_copy[0])));
455  break;
456  } // end rp_copy
457  case rp_split: {
458  // Edge has to be split into two halves with a new node created at the
459  // midpoint position. First obtain geometry information about new node
460  // (sub-entity with relative co-dimension 1)
461  std::vector<std::unique_ptr<geometry::Geometry>> edge_nodes_geo_ptrs(
462  edge->Geometry()->ChildGeometry(rp, 1));
463  LF_VERIFY_MSG(edge_nodes_geo_ptrs.size() == 1,
464  "Split edge with " << edge_nodes_geo_ptrs.size()
465  << " child nodes!");
466  // Register midpoint as new node
467  const lf::base::glb_idx_t midpoint_fine_idx =
468  mesh_factory_->AddPoint(std::move(edge_nodes_geo_ptrs[0]));
469  edge_ci.child_point_idx.push_back(midpoint_fine_idx);
470  // Next get the geometry objects for the two child edges (co-dim == 0)
471  std::vector<std::unique_ptr<geometry::Geometry>> edge_child_geo_ptrs(
472  edge->Geometry()->ChildGeometry(rp, 0));
473  LF_VERIFY_MSG(
474  edge_child_geo_ptrs.size() == 2,
475  "Split edge with " << edge_child_geo_ptrs.size() << " parts!");
476  // Register the two new edges
477  // CAREFUL: Assignment of endpoints has to match implementation in
478  // refinement.cc
479  // First child edge adjacent to endpoint #0, second child edge
480  // adjacent to endpoint #1
481  SPDLOG_LOGGER_TRACE(Logger(),
482  "Split Edge {} new edges [{},{}], [{},{}]",
483  edge_index, ed_p0_fine_idx, midpoint_fine_idx,
484  midpoint_fine_idx, ed_p1_fine_idx);
485 
486  edge_ci.child_edge_idx.push_back(
487  mesh_factory_->AddEntity(edge->RefEl(),
488  std::array<lf::base::glb_idx_t, 2>{
489  {ed_p0_fine_idx, midpoint_fine_idx}},
490  std::move(edge_child_geo_ptrs[0])));
491  edge_ci.child_edge_idx.push_back(
492  mesh_factory_->AddEntity(edge->RefEl(),
493  std::array<lf::base::glb_idx_t, 2>{
494  {midpoint_fine_idx, ed_p1_fine_idx}},
495  std::move(edge_child_geo_ptrs[1])));
496  break;
497  } // end rp_split
498  default: {
499  LF_VERIFY_MSG(false, "Refinement pattern "
500  << static_cast<int>(edge_refpat)
501  << " illegal for edge");
502  break;
503  }
504  } // end switch refpat
505  new_edge_cnt++;
506  } // end edge loop
507 
508  SPDLOG_LOGGER_DEBUG(Logger(), "{} edges added", new_edge_cnt);
509  // ======================================================================
510 
511  // Partly intialized vectors of child information
512  std::vector<CellChildInfo> &cell_child_info(cell_child_infos_.back());
513  // ======================================================================
514  // Visit all cells, examine their refinement patterns, retrieve indices of
515  // their sub-entities, and those of the children.
516  LF_VERIFY_MSG(cell_child_info.size() == parent_mesh.NumEntities(0),
517  "Size mismatch for CellChildInfos");
518  for (const mesh::Entity *cell : parent_mesh.Entities(0)) {
519  // type of cell
520  const lf::base::RefEl ref_el(cell->RefEl());
521  const lf::base::size_type num_edges = ref_el.NumSubEntities(1);
522  const lf::base::size_type num_vertices = ref_el.NumSubEntities(2);
523  // fetch index of current cell
524  const lf::base::glb_idx_t cell_index(parent_mesh.Index(*cell));
525 
526  // Set up refinement object -> variable rp
527  CellChildInfo &cell_ci(cell_child_info[cell_index]);
528  const RefPat cell_refpat(cell_ci.ref_pat_);
529  const sub_idx_t anchor = cell_ci.anchor_; // anchor edge index
530 
531  SPDLOG_LOGGER_TRACE(Logger(), "Cell {} = {}, refpat = {}, anchor = {}",
532  cell_index, ref_el, static_cast<int>(cell_refpat),
533  anchor);
534 
535  Hybrid2DRefinementPattern rp(cell->RefEl(), cell_refpat, anchor);
536 
537  // Index offsets for refinement patterns requiring an ancchor edge
538  std::array<sub_idx_t, 4> mod{};
539  if (anchor != idx_nil) {
540  for (int k = 0; k < num_vertices; k++) {
541  mod[k] = (k + anchor) % num_vertices;
542  }
543  }
544 
545  // Obtain indices of subentities (co-dimension = outer array index)
546  std::array<std::vector<lf::base::glb_idx_t>, 3> cell_subent_idx;
547  cell_subent_idx[0].push_back(cell_index);
548  for (int codim = 1; codim <= 2; codim++) {
549  auto subentities = cell->SubEntities(codim);
550  for (const mesh::Entity *sub_ent : subentities) {
551  cell_subent_idx[codim].push_back(parent_mesh.Index(*sub_ent));
552  }
553  LF_VERIFY_MSG(
554  cell_subent_idx[codim].size() == ref_el.NumSubEntities(codim),
555  ref_el.ToString() << ": only " << cell_subent_idx[codim].size()
556  << " subents of codim = " << codim);
557 
558  SPDLOG_LOGGER_TRACE(Logger(), " Subent({}) = [{}], ", codim,
559  cell_subent_idx[codim]);
560  }
561  // Index information for sub-entities with respect to fine mesh
562  // Retrieve indices of vertices of cell on the fine mesh
563  std::array<lf::base::glb_idx_t, 4> vertex_child_idx{
564  {idx_nil, idx_nil, idx_nil, idx_nil}};
565  for (lf::base::sub_idx_t vt_lidx = 0; vt_lidx < num_vertices; vt_lidx++) {
566  LF_VERIFY_MSG(pt_child_info[cell_subent_idx[2][vt_lidx]].ref_pat ==
567  RefPat::rp_copy,
568  "Vertex must have been copied!");
569  vertex_child_idx[vt_lidx] =
570  pt_child_info[cell_subent_idx[2][vt_lidx]].child_point_idx;
571  }
572 
573  // Retrieve indices of midpoints of edges, if they exist
574  std::array<lf::base::glb_idx_t, 4> edge_midpoint_idx{
575  {idx_nil, idx_nil, idx_nil, idx_nil}};
576  for (lf::base::sub_idx_t ed_lidx = 0; ed_lidx < num_edges; ed_lidx++) {
577  const EdgeChildInfo &ed_ci(ed_child_info[cell_subent_idx[1][ed_lidx]]);
578  if (!ed_ci.child_point_idx.empty()) {
579  LF_VERIFY_MSG(ed_child_info[cell_subent_idx[1][ed_lidx]].ref_pat_ ==
580  RefPat::rp_split,
581  "Edge with a midpoint must have been split");
582  edge_midpoint_idx[ed_lidx] = ed_ci.child_point_idx[0];
583  }
584  } // end loop over local edges
585 
586  SPDLOG_LOGGER_TRACE(Logger(), "vt_child_idx = {}, ed_mp_idx = {}",
587  nonstd::span(vertex_child_idx.data(), num_vertices),
588  nonstd::span(edge_midpoint_idx.data(), num_edges));
589 
590  // Array of node indices (w.r.t. fine mesh) for sub-cells (triangles or
591  // quadrilaterals)
592  std::vector<std::vector<glb_idx_t>> child_cell_nodes;
593  std::vector<glb_idx_t> tria_ccn_tmp(3);
594  std::vector<glb_idx_t> quad_ccn_tmp(4);
595  // Array of node indices for interior child edges
596  std::vector<std::array<glb_idx_t, 2>> child_edge_nodes;
597  std::array<glb_idx_t, 2> cen_tmp{};
598 
599  // Local linking of child entities is very different depending on cell
600  // type and refinement patterns.
601  if (ref_el == lf::base::RefEl::kTria()) {
602  // Case of a triangle
603  switch (cell_refpat) {
604  case RefPat::rp_nil: {
605  LF_VERIFY_MSG(false, "Every triangle must be refined");
606  break;
607  }
608  case RefPat::rp_copy: {
609  // Just copy the parent triangle
610  child_cell_nodes.push_back({vertex_child_idx[0],
611  vertex_child_idx[1],
612  vertex_child_idx[2]});
613  break;
614  }
615  case RefPat::rp_bisect: {
616  LF_VERIFY_MSG(
617  anchor != idx_nil,
618  "Anchor must be set for bisection refinement of triangle");
619  // Splitting a triangle in two by bisecting anchor edge
620  tria_ccn_tmp[0] = vertex_child_idx[mod[0]];
621  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
622  tria_ccn_tmp[2] = vertex_child_idx[mod[2]];
623  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
624  "refpat = " << (int)cell_refpat << ": illegal index");
625  child_cell_nodes.push_back(tria_ccn_tmp);
626 
627  tria_ccn_tmp[0] = vertex_child_idx[mod[1]];
628  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
629  tria_ccn_tmp[2] = vertex_child_idx[mod[2]];
630  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
631  "refpat = " << (int)cell_refpat << ": illegal index");
632  child_cell_nodes.push_back(tria_ccn_tmp);
633 
634  // edges
635  cen_tmp[0] = edge_midpoint_idx[mod[0]];
636  cen_tmp[1] = vertex_child_idx[mod[2]];
637  child_edge_nodes.push_back(cen_tmp);
638  break;
639  }
640  case RefPat::rp_trisect: {
641  LF_VERIFY_MSG(
642  anchor != idx_nil,
643  "Anchor must be set for trisection refinement of triangle");
644  // Bisect through anchor edge first and then bisect through
645  // edge with the next larger index (mod 3); creates three
646  // child triangles.
647  tria_ccn_tmp[0] = vertex_child_idx[mod[0]];
648  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
649  tria_ccn_tmp[2] = vertex_child_idx[mod[2]];
650  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
651  "refpat = " << (int)cell_refpat << ": illegal index");
652  child_cell_nodes.push_back(tria_ccn_tmp);
653 
654  tria_ccn_tmp[0] = vertex_child_idx[mod[1]];
655  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
656  tria_ccn_tmp[2] = edge_midpoint_idx[mod[1]];
657  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
658  "refpat = " << (int)cell_refpat << ": illegal index");
659  child_cell_nodes.push_back(tria_ccn_tmp);
660 
661  tria_ccn_tmp[0] = vertex_child_idx[mod[2]];
662  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
663  tria_ccn_tmp[2] = edge_midpoint_idx[mod[1]];
664  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
665  "refpat = " << (int)cell_refpat << ": illegal index");
666  child_cell_nodes.push_back(tria_ccn_tmp);
667 
668  // edges
669  cen_tmp[0] = edge_midpoint_idx[mod[0]];
670  cen_tmp[1] = vertex_child_idx[mod[2]];
671  child_edge_nodes.push_back(cen_tmp);
672 
673  cen_tmp[0] = edge_midpoint_idx[mod[0]];
674  cen_tmp[1] = edge_midpoint_idx[mod[1]];
675  child_edge_nodes.push_back(cen_tmp);
676  break;
677  }
678  case RefPat::rp_trisect_left: {
679  LF_VERIFY_MSG(
680  anchor != idx_nil,
681  "Anchor must be set for trisection refinement of triangle");
682 
683  // Bisect through anchor edge first and then bisect through
684  // edge with the next smaller index (mod 3); creates three
685  // child triangles.
686  tria_ccn_tmp[0] = vertex_child_idx[mod[0]];
687  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
688  tria_ccn_tmp[2] = edge_midpoint_idx[mod[2]];
689  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
690  "refpat = " << (int)cell_refpat << ": illegal index");
691  child_cell_nodes.push_back(tria_ccn_tmp);
692 
693  tria_ccn_tmp[0] = vertex_child_idx[mod[1]];
694  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
695  tria_ccn_tmp[2] = vertex_child_idx[mod[2]];
696  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
697  "refpat = " << (int)cell_refpat << ": illegal index");
698  child_cell_nodes.push_back(tria_ccn_tmp);
699 
700  tria_ccn_tmp[0] = vertex_child_idx[mod[2]];
701  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
702  tria_ccn_tmp[2] = edge_midpoint_idx[mod[2]];
703  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
704  "refpat = " << (int)cell_refpat << ": illegal index");
705  child_cell_nodes.push_back(tria_ccn_tmp);
706 
707  // edges
708  cen_tmp[0] = edge_midpoint_idx[mod[0]];
709  cen_tmp[1] = vertex_child_idx[mod[2]];
710  child_edge_nodes.push_back(cen_tmp);
711 
712  cen_tmp[0] = edge_midpoint_idx[mod[0]];
713  cen_tmp[1] = edge_midpoint_idx[mod[2]];
714  child_edge_nodes.push_back(cen_tmp);
715  break;
716  }
717  case RefPat::rp_quadsect: {
718  LF_VERIFY_MSG(
719  anchor != idx_nil,
720  "Anchor must be set for quadsection refinement of triangle");
721  // Bisect through the anchor edge first and then
722  // through the two remaining edges; creates four child
723  // triangles; every edge is split.
724  tria_ccn_tmp[0] = vertex_child_idx[mod[0]];
725  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
726  tria_ccn_tmp[2] = edge_midpoint_idx[mod[2]];
727  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
728  "refpat = " << (int)cell_refpat << ": illegal index");
729  child_cell_nodes.push_back(tria_ccn_tmp);
730 
731  tria_ccn_tmp[0] = vertex_child_idx[mod[1]];
732  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
733  tria_ccn_tmp[2] = edge_midpoint_idx[mod[1]];
734  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
735  "refpat = " << (int)cell_refpat << ": illegal index");
736  child_cell_nodes.push_back(tria_ccn_tmp);
737 
738  tria_ccn_tmp[0] = vertex_child_idx[mod[2]];
739  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
740  tria_ccn_tmp[2] = edge_midpoint_idx[mod[1]];
741  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
742  "refpat = " << (int)cell_refpat << ": illegal index");
743  child_cell_nodes.push_back(tria_ccn_tmp);
744 
745  tria_ccn_tmp[0] = vertex_child_idx[mod[2]];
746  tria_ccn_tmp[1] = edge_midpoint_idx[mod[0]];
747  tria_ccn_tmp[2] = edge_midpoint_idx[mod[2]];
748  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
749  "refpat = " << (int)cell_refpat << ": illegal index");
750  child_cell_nodes.push_back(tria_ccn_tmp);
751 
752  // edges
753  cen_tmp[0] = edge_midpoint_idx[mod[0]];
754  cen_tmp[1] = vertex_child_idx[mod[2]];
755  child_edge_nodes.push_back(cen_tmp);
756 
757  cen_tmp[0] = edge_midpoint_idx[mod[0]];
758  cen_tmp[1] = edge_midpoint_idx[mod[2]];
759  child_edge_nodes.push_back(cen_tmp);
760 
761  cen_tmp[0] = edge_midpoint_idx[mod[0]];
762  cen_tmp[1] = edge_midpoint_idx[mod[1]];
763  child_edge_nodes.push_back(cen_tmp);
764  break;
765  }
766  case RefPat::rp_barycentric: {
767  // Split triangle into 6 smaller triangles by connecting
768  // the center of gravity with the vertices and the midpoints
769  // of the edges.
770  // Create a new interior vertex
771  std::vector<std::unique_ptr<geometry::Geometry>>
772  cell_center_geo_ptrs(cell->Geometry()->ChildGeometry(
773  rp, 2)); // point: co-dim == 2
774  LF_VERIFY_MSG(cell_center_geo_ptrs.size() == 1,
775  "Barycentrically refined triangle with "
776  << cell_center_geo_ptrs.size()
777  << " interior child nodes ??");
778  // Register midpoint as new node
779  const glb_idx_t center_fine_idx =
780  mesh_factory_->AddPoint(std::move(cell_center_geo_ptrs[0]));
781  cell_ci.child_point_idx.push_back(center_fine_idx);
782 
783  tria_ccn_tmp[0] = vertex_child_idx[0];
784  tria_ccn_tmp[1] = edge_midpoint_idx[0];
785  tria_ccn_tmp[2] = center_fine_idx;
786  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
787  "refpat = " << (int)cell_refpat << ": illegal index");
788  child_cell_nodes.push_back(tria_ccn_tmp);
789 
790  tria_ccn_tmp[0] = vertex_child_idx[1];
791  tria_ccn_tmp[1] = edge_midpoint_idx[0];
792  tria_ccn_tmp[2] = center_fine_idx;
793  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
794  "refpat = " << (int)cell_refpat << ": illegal index");
795  child_cell_nodes.push_back(tria_ccn_tmp);
796 
797  tria_ccn_tmp[0] = vertex_child_idx[1];
798  tria_ccn_tmp[1] = edge_midpoint_idx[1];
799  tria_ccn_tmp[2] = center_fine_idx;
800  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
801  "refpat = " << (int)cell_refpat << ": illegal index");
802  child_cell_nodes.push_back(tria_ccn_tmp);
803 
804  tria_ccn_tmp[0] = vertex_child_idx[2];
805  tria_ccn_tmp[1] = edge_midpoint_idx[1];
806  tria_ccn_tmp[2] = center_fine_idx;
807  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
808  "refpat = " << (int)cell_refpat << ": illegal index");
809  child_cell_nodes.push_back(tria_ccn_tmp);
810 
811  tria_ccn_tmp[0] = vertex_child_idx[2];
812  tria_ccn_tmp[1] = edge_midpoint_idx[2];
813  tria_ccn_tmp[2] = center_fine_idx;
814  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
815  "refpat = " << (int)cell_refpat << ": illegal index");
816  child_cell_nodes.push_back(tria_ccn_tmp);
817 
818  tria_ccn_tmp[0] = vertex_child_idx[0];
819  tria_ccn_tmp[1] = edge_midpoint_idx[2];
820  tria_ccn_tmp[2] = center_fine_idx;
821  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
822  "refpat = " << (int)cell_refpat << ": illegal index");
823  child_cell_nodes.push_back(tria_ccn_tmp);
824 
825  // edges
826  cen_tmp[0] = vertex_child_idx[0];
827  cen_tmp[1] = center_fine_idx;
828  child_edge_nodes.push_back(cen_tmp);
829 
830  cen_tmp[0] = vertex_child_idx[1];
831  cen_tmp[1] = center_fine_idx;
832  child_edge_nodes.push_back(cen_tmp);
833 
834  cen_tmp[0] = vertex_child_idx[2];
835  cen_tmp[1] = center_fine_idx;
836  child_edge_nodes.push_back(cen_tmp);
837 
838  cen_tmp[0] = edge_midpoint_idx[0];
839  cen_tmp[1] = center_fine_idx;
840  child_edge_nodes.push_back(cen_tmp);
841 
842  cen_tmp[0] = edge_midpoint_idx[1];
843  cen_tmp[1] = center_fine_idx;
844  child_edge_nodes.push_back(cen_tmp);
845 
846  cen_tmp[0] = edge_midpoint_idx[2];
847  cen_tmp[1] = center_fine_idx;
848  child_edge_nodes.push_back(cen_tmp);
849  break;
850  }
851  case RefPat::rp_regular: {
852  // regular refinement into four congruent triangles
853  // Child cells
854  tria_ccn_tmp[0] = vertex_child_idx[0];
855  tria_ccn_tmp[1] = edge_midpoint_idx[0];
856  tria_ccn_tmp[2] = edge_midpoint_idx[2];
857  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
858  "refpat = " << (int)cell_refpat << ": illegal index");
859  child_cell_nodes.push_back(tria_ccn_tmp);
860 
861  tria_ccn_tmp[0] = vertex_child_idx[1];
862  tria_ccn_tmp[1] = edge_midpoint_idx[0];
863  tria_ccn_tmp[2] = edge_midpoint_idx[1];
864  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
865  "refpat = " << (int)cell_refpat << ": illegal index");
866  child_cell_nodes.push_back(tria_ccn_tmp);
867 
868  tria_ccn_tmp[0] = vertex_child_idx[2];
869  tria_ccn_tmp[1] = edge_midpoint_idx[2];
870  tria_ccn_tmp[2] = edge_midpoint_idx[1];
871  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
872  "refpat = " << (int)cell_refpat << ": illegal index");
873  child_cell_nodes.push_back(tria_ccn_tmp);
874 
875  tria_ccn_tmp[0] = edge_midpoint_idx[0];
876  tria_ccn_tmp[1] = edge_midpoint_idx[1];
877  tria_ccn_tmp[2] = edge_midpoint_idx[2];
878  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
879  "refpat = " << (int)cell_refpat << ": illegal index");
880  child_cell_nodes.push_back(tria_ccn_tmp);
881 
882  // Child edges (interior edges)
883  cen_tmp[0] = edge_midpoint_idx[0];
884  cen_tmp[1] = edge_midpoint_idx[2];
885  child_edge_nodes.push_back(cen_tmp);
886 
887  cen_tmp[0] = edge_midpoint_idx[0];
888  cen_tmp[1] = edge_midpoint_idx[1];
889  child_edge_nodes.push_back(cen_tmp);
890 
891  cen_tmp[0] = edge_midpoint_idx[2];
892  cen_tmp[1] = edge_midpoint_idx[1];
893  child_edge_nodes.push_back(cen_tmp);
894  // No new interior node
895  break;
896  }
897  default: {
898  LF_VERIFY_MSG(
899  false, "Refinement pattern not (yet) implemented for triangle");
900  break;
901  }
902  } // end switch refpat
903  } else if (ref_el == lf::base::RefEl::kQuad()) {
904  // Case of a quadrilateral
905  switch (cell_refpat) {
906  case RefPat::rp_nil: {
907  LF_VERIFY_MSG(false, "Every quadrilateral must be refined");
908  break;
909  }
910  case RefPat::rp_copy: {
911  // Just copy the parent triangle
912  child_cell_nodes.push_back(
913  {vertex_child_idx[0], vertex_child_idx[1], vertex_child_idx[2],
914  vertex_child_idx[3]});
915  break;
916  }
917  case RefPat::rp_trisect: {
918  LF_VERIFY_MSG(
919  anchor != idx_nil,
920  "Anchor must be set for trisection refinement of quad");
921 
922  // Partition a quad into three triangle, the anchor edge
923  // being split in the process
924  tria_ccn_tmp[0] = edge_midpoint_idx[mod[0]];
925  tria_ccn_tmp[1] = vertex_child_idx[mod[2]];
926  tria_ccn_tmp[2] = vertex_child_idx[mod[3]];
927  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
928  "refpat = " << (int)cell_refpat << ": illegal index");
929  child_cell_nodes.push_back(tria_ccn_tmp);
930 
931  tria_ccn_tmp[0] = edge_midpoint_idx[mod[0]];
932  tria_ccn_tmp[1] = vertex_child_idx[mod[0]];
933  tria_ccn_tmp[2] = vertex_child_idx[mod[3]];
934  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
935  "refpat = " << (int)cell_refpat << ": illegal index");
936  child_cell_nodes.push_back(tria_ccn_tmp);
937 
938  tria_ccn_tmp[0] = edge_midpoint_idx[mod[0]];
939  tria_ccn_tmp[1] = vertex_child_idx[mod[1]];
940  tria_ccn_tmp[2] = vertex_child_idx[mod[2]];
941  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
942  "refpat = " << (int)cell_refpat << ": illegal index");
943  child_cell_nodes.push_back(tria_ccn_tmp);
944 
945  // edges
946  cen_tmp[0] = edge_midpoint_idx[mod[0]];
947  cen_tmp[1] = vertex_child_idx[mod[2]];
948  child_edge_nodes.push_back(cen_tmp);
949 
950  cen_tmp[0] = edge_midpoint_idx[mod[0]];
951  cen_tmp[1] = vertex_child_idx[mod[3]];
952  child_edge_nodes.push_back(cen_tmp);
953  break;
954  }
955  case RefPat::rp_quadsect: {
956  LF_VERIFY_MSG(
957  anchor != idx_nil,
958  "Anchor must be set for quadsection refinement of triangle");
959  // Partition a quad into four triangle, thus
960  // splitting two edges. The one with the smaller sub index is the
961  // anchor edge
962  tria_ccn_tmp[0] = vertex_child_idx[mod[0]];
963  tria_ccn_tmp[1] = vertex_child_idx[mod[3]];
964  tria_ccn_tmp[2] = edge_midpoint_idx[mod[0]];
965  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
966  "refpat = " << (int)cell_refpat << ": illegal index");
967  child_cell_nodes.push_back(tria_ccn_tmp);
968 
969  tria_ccn_tmp[0] = vertex_child_idx[mod[1]];
970  tria_ccn_tmp[1] = edge_midpoint_idx[mod[1]];
971  tria_ccn_tmp[2] = edge_midpoint_idx[mod[0]];
972  LF_ASSERT_MSG(checkValidIndex(tria_ccn_tmp),
973  "refpat = " << (int)cell_refpat << ": illegal index");
974  child_cell_nodes.push_back(tria_ccn_tmp);
