410  [[nodiscard]] constexpr sub_idx_t SubSubEntity2SubEntity(
411  dim_t sub_codim, sub_idx_t sub_index, dim_t sub_rel_codim,
412  sub_idx_t sub_rel_index) const {
413  LF_ASSERT_MSG_CONSTEXPR(sub_codim >= 0, "sub_codim negative");

415  LF_ASSERT_MSG_CONSTEXPR(sub_index >= 0, "sub_index negative");
416  LF_ASSERT_MSG_CONSTEXPR(sub_index <= NumSubEntities(sub_codim),
417  "sub_index >= NumSubEntities");
418  LF_ASSERT_MSG_CONSTEXPR(sub_rel_codim >= 0, "sub_rel_codim negative.");
419  LF_ASSERT_MSG_CONSTEXPR(sub_rel_codim <= Dimension() - sub_codim,
420  "subSubCodim out of bounds.");
421  LF_ASSERT_MSG_CONSTEXPR(sub_rel_index >= 0, "sub_rel_index negative.");
422  LF_ASSERT_MSG_CONSTEXPR(
423  sub_rel_index <
424  SubType(sub_codim, sub_index).NumSubEntities(sub_rel_codim),
425  "sub_sub_index out of bounds.");
426 
427  if (type_ == RefElType::kPoint) {
428  return 0;
429  }
430  if (sub_codim == 0) {
431  return sub_rel_index;
432  }
433  if (sub_codim == Dimension()) {
434  return sub_index;
435  }
436 
437  // from here on, it must be a segment
438  switch (type_) {
439  case RefElType::kTria:
440  return sub_sub_entity_index_tria_[sub_index][sub_rel_index];
441  case RefElType::kQuad:
442  return sub_sub_entity_index_quad_[sub_index][sub_rel_index];
443  default:
444  LF_ASSERT_MSG_CONSTEXPR(false, "This code should never be reached.");
445  }
446 
447  return 0; // Prevent warnings from compiler...
448  }
