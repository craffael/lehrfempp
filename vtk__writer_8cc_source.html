421 void ValidateVtkFile(const VtkFile& vtk_file) {
422  // check that header doesn't contain a new line character:
423  if (vtk_file.header.find('\n') != std::string::npos) {
424  throw base::LfException(
425  "Header of vtk file contains a new line character, this is not "
426  "allowed");
427  }
428  if (vtk_file.header.size() > 256) {
429  throw base::LfException("Header of vtk file is longer than 256 characters");
430  }
431  if (vtk_file.unstructured_grid.cell_types.size() !=
432  vtk_file.unstructured_grid.cells.size()) {
433  throw base::LfException(
434  "Mismatch of size of cell_types and cells in VtkFile.");
435  }
436  for (auto& d : vtk_file.point_data) {
437  boost::apply_visitor(
438  [&](auto e) {
439  if (e.data.size() != vtk_file.unstructured_grid.points.size()) {
440  throw base::LfException("Length of dataset " + e.name +
441  " does not match the number of points.");
442  }
443  },
444  d);
445  }
446  for (auto& d : vtk_file.cell_data) {
447  boost::apply_visitor(
448  [&](auto e) {
449  if (e.data.size() != vtk_file.unstructured_grid.cells.size()) {
450  throw base::LfException("Length of dataset " + e.name +
451  " does not match the number of cells.");
452  }
453  },
454  d);
455  }
456 }

459 unsigned int NumAuxNodes(base::RefEl ref_el, unsigned char order) {
460  switch (ref_el) {
461  case base::RefEl::kPoint():
462  return 1;
463  case base::RefEl::kSegment():
464  return order - 1;
465  case base::RefEl::kTria():
466  return (2 - 3 * order + order * order) / 2;
467  case base::RefEl::kQuad():
468  return (order - 1) * (order - 1);
469  default:
470  LF_VERIFY_MSG(false, "RefElType " << ref_el << " not supported.");
471  }
472 }

474 // Compute aux reference coordinates of lagrange points on segment
475 Eigen::MatrixXd AuxNodesSegment(unsigned char order) {
476  LF_ASSERT_MSG(order > 1, "order <= 1");
477  return Eigen::VectorXd::LinSpaced(order - 1, 1. / (order),
478  (order - 1.) / order)
479  .transpose();
480 }
481 
482 // compute aux reference coordinates of lagrange points on Tria:
483 Eigen::MatrixXd AuxNodesTria(unsigned char order) {
484  if (order < 3) {
485  return Eigen::MatrixXd();
486  }
487  Eigen::MatrixXd result(2, NumAuxNodes(base::RefEl::kTria(), order));
488  if (order == 3) {
489  result << 1. / 3., 1. / 3.;
490  } else if (order == 4) {
491  result << 0.25, 0.5, 0.25, 0.25, 0.25, 0.5;
492  } else {
493  // assign nodes:
494  result(0, 0) = 1. / order;
495  result(1, 0) = 1. / order;
496  result(0, 1) = (order - 2.) / order;
497  result(1, 1) = 1. / order;
498  result(0, 2) = 1. / order;
499  result(1, 2) = (order - 2.) / order;
500 
501  if (order > 4) {
502  auto segment_points = AuxNodesSegment(order - 3).eval();
503  // assign edges:
504  result.block(0, 3, 2, order - 4) =
505  Eigen::Vector2d::UnitX() * segment_points * (order - 3.) / order +
506  Eigen::Vector2d(1. / order, 1. / order) *
507  Eigen::MatrixXd::Ones(1, order - 4);
508  result.block(0, order - 1, 2, order - 4) =
509  Eigen::Vector2d(-1, 1) * (order - 3.) / order * segment_points +
510  Eigen::Vector2d((order - 2.) / order, 1. / order) *
511  Eigen::MatrixXd::Ones(1, order - 4);
512  result.block(0, 2 * order - 5, 2, order - 4) =
513  -Eigen::Vector2d::UnitY() * (order - 3.) / order * segment_points +
514  Eigen::Vector2d(1. / order, (order - 2.) / order) *
515  Eigen::MatrixXd::Ones(1, order - 4);
516  }
517  if (order > 5) {
518  // assign interior points recursively:
519  auto points = AuxNodesTria(order - 3);
520  result.block(0, 3 * order - 9, 2, points.cols()) =
521  AuxNodesTria(order - 3) * (order - 5.) / order +
522  Eigen::Vector2d(2. / order, 2. / order) *
523  Eigen::MatrixXd::Ones(1, points.cols());
524  }
525  }
526  return result;
527 }
528 
529 // compute aux reference coordinates of lagrange points on quad:
530 Eigen::MatrixXd AuxNodesQuad(unsigned char order) {
531  Eigen::MatrixXd result(2, NumAuxNodes(base::RefEl::kQuad(), order));
532  result.row(0) =
533  Eigen::VectorXd::LinSpaced(order - 1, 1. / order, (order - 1.) / order)
534  .transpose()
535  .replicate(1, order - 1);
536  result.row(1) =
537  Eigen::kroneckerProduct(Eigen::VectorXd::LinSpaced(order - 1, 1. / order,
538  (order - 1.) / order),
539  Eigen::VectorXd::Constant(order - 1, 1.))
540  .transpose();
541  return result;
542 }

546 void WriteToFile(const VtkFile& vtk_file, const std::string& filename) {
547  std::ofstream file(filename, std::ios_base::out | std::ios_base::binary |
548  std::ios_base::trunc);
549  ValidateVtkFile(vtk_file);
550 
551  if (!file.is_open()) {
552  throw base::LfException("Could not open file " + filename +
553  " for writing.");
554  }
555  karma::ostream_iterator<char> outit(file);
556 
557  bool result = false;
558  if (vtk_file.format == VtkFile::Format::BINARY) {
559  VtkGrammar<decltype(outit), true> grammar{};

572 VtkWriter::VtkWriter(std::shared_ptr<const mesh::Mesh> mesh,
573  std::string filename, dim_t codim, unsigned char order)
574  : mesh_(std::move(mesh)),
575  filename_(std::move(filename)),
576  codim_(codim),
577  order_(order),
578  aux_node_offset_{{{mesh_, 0}, {mesh_, 1}}} {
579  auto dim_mesh = mesh_->DimMesh();
580  auto dim_world = mesh_->DimWorld();
581  LF_ASSERT_MSG(dim_world > 0 && dim_world <= 4,
582  "VtkWriter supports only dim_world = 1,2 or 3");
583  LF_ASSERT_MSG(codim >= 0 && codim < dim_mesh, "codim out of bounds.");
584  LF_ASSERT_MSG(order > 0 && order <= 10, "order must lie in [1,10]");
585 
586  // initialize reference coordinates of auxilliary nodes (if order_ > 0)
587  if (order_ > 1) {
588  aux_nodes_[base::RefEl::kSegment().Id()] = AuxNodesSegment(order);
589  aux_nodes_[base::RefEl::kTria().Id()] = AuxNodesTria(order);
590  aux_nodes_[base::RefEl::kQuad().Id()] = AuxNodesQuad(order);
591  }
592 
593  // calculate total number of nodes (main + auxilliary nodes)
594  unsigned int numNodes = mesh_->NumEntities(dim_mesh);
595  for (auto ref_el :
596  {base::RefEl::kSegment(), base::RefEl::kTria(), base::RefEl::kQuad()}) {
597  if (ref_el.Dimension() > dim_mesh - codim_) {
598  continue;
599  }
600  numNodes += NumAuxNodes(ref_el, order_) * mesh_->NumEntities(ref_el);
601  }
602 
603  // insert main nodes:
604  vtk_file_.unstructured_grid.points.resize(numNodes);
605  Eigen::Matrix<double, 0, 1> zero;
606  for (auto p : mesh_->Entities(dim_mesh)) {
607  auto index = mesh_->Index(*p);
608  Eigen::Vector3f coord;
609  if (dim_world == 1) {
610  coord(0) = p->Geometry()->Global(zero)(0);
611  coord(1) = 0.F;
612  coord(2) = 0.F;
613  } else if (dim_world == 2) {
614  coord.topRows<2>() = p->Geometry()->Global(zero).cast<float>();
615  coord(2) = 0.F;
616  } else {
617  coord = p->Geometry()->Global(zero).cast<float>();
618  }
619 
620  vtk_file_.unstructured_grid.points[index] = std::move(coord);
621  }
622 
623  // insert auxilliary nodes:
624  if (order > 1) {
625  auto index_offset = mesh_->NumEntities(dim_mesh);
626  for (char cd = static_cast<char>(dim_mesh - 1);
627  cd >= static_cast<char>(codim); --cd) {
628  for (auto e : mesh_->Entities(cd)) {
629  auto ref_el = e->RefEl();
630  if (ref_el == base::RefEl::kTria() && order < 3) {
631  continue;
632  }
633 
634  Eigen::MatrixXf coords(3, NumAuxNodes(ref_el, order));
635 
636  if (dim_world == 1) {
637  coords.row(0) =
638  e->Geometry()->Global(aux_nodes_[ref_el.Id()]).cast<float>();
639  coords.row(1).setZero();
640  coords.row(2).setZero();
641  } else if (dim_world == 2) {
642  coords.topRows(2) =
643  e->Geometry()->Global(aux_nodes_[ref_el.Id()]).cast<float>();
644  coords.row(2).setZero();
645  } else {
646  coords = e->Geometry()->Global(aux_nodes_[ref_el.Id()]).cast<float>();
647  }
648  for (Eigen::Index i = 0; i < coords.cols(); ++i) {
649  vtk_file_.unstructured_grid.points[index_offset + i] = coords.col(i);
650  }
651  aux_node_offset_[cd](*e) = index_offset;
652  index_offset += coords.cols();
653  }
654  }
655  }
656 
657  // compute number of main/aux nodes for each element:
658  std::array<unsigned int, 5> num_nodes{};
659  num_nodes[base::RefEl::kSegment().Id()] =
660  2 + NumAuxNodes(base::RefEl::kSegment(), order);
661  num_nodes[base::RefEl::kTria().Id()] =
662  3 + 3 * NumAuxNodes(base::RefEl::kSegment(), order) +
663  NumAuxNodes(base::RefEl::kTria(), order);
664  num_nodes[base::RefEl::kQuad().Id()] =
665  4 + 4 * NumAuxNodes(base::RefEl::kSegment(), order) +
666  NumAuxNodes(base::RefEl::kQuad(), order);
667 
668  // insert elements:
669  vtk_file_.unstructured_grid.cells.resize(mesh_->NumEntities(codim));
670  vtk_file_.unstructured_grid.cell_types.resize(mesh_->NumEntities(codim));
671  auto points_per_segment = NumAuxNodes(base::RefEl::kSegment(), order);
672  for (auto e : mesh_->Entities(codim)) {
673  auto index = mesh_->Index(*e);
674  auto ref_el = e->RefEl();
675  auto& node_indices = vtk_file_.unstructured_grid.cells[index];
676  node_indices.reserve(num_nodes[ref_el.Id()]);
677 
678  // node indices that make up this cell:
679  for (auto p : e->SubEntities(dim_mesh - codim)) {
680  node_indices.push_back(mesh_->Index(*p));
681  }
682 
683  // node indices of segments of this cell:
684  auto addSegmentNodes = [&](const mesh::Entity& s, bool invert) {
685  auto start_index = aux_node_offset_[dim_mesh - 1](s);
686  if (!invert) {
687  for (unsigned int i = 0; i < points_per_segment; ++i) {
688  node_indices.push_back(start_index + i);
689  }
690  } else {
691  for (int i = static_cast<int>(points_per_segment - 1); i >= 0; --i) {
692  node_indices.push_back(start_index + i);
693  }
694  }
695  };
696  switch (ref_el) {
697  case base::RefEl::kSegment():
698  addSegmentNodes(*e, false);
699  break;
700  case base::RefEl::kTria(): {
701  auto iterator = e->SubEntities(1).begin();
702  auto o_iterator = e->RelativeOrientations().begin();
703  addSegmentNodes(**iterator,
704  (*o_iterator) == mesh::Orientation::negative);
705  ++iterator;
706  ++o_iterator;
707  addSegmentNodes(**iterator,
708  (*o_iterator) == mesh::Orientation::negative);
709  ++iterator;
710  ++o_iterator;
711  addSegmentNodes(**iterator,
712  (*o_iterator) == mesh::Orientation::negative);
713  break;
714  }
715  case base::RefEl::kQuad(): {
716  auto iterator = e->SubEntities(1).begin();
717  auto o_iterator = e->RelativeOrientations().begin();
718  addSegmentNodes(**iterator,
719  (*o_iterator) == mesh::Orientation::negative);
720  ++iterator;
721  ++o_iterator;
722  addSegmentNodes(**iterator,
723  (*o_iterator) == mesh::Orientation::negative);
724  ++iterator;
725  ++o_iterator;
726  addSegmentNodes(**iterator,
727  (*o_iterator) == mesh::Orientation::positive);
728  ++iterator;
729  ++o_iterator;
730  addSegmentNodes(**iterator,
731  (*o_iterator) == mesh::Orientation::positive);
732  break;
733  }
734  default:
735  LF_VERIFY_MSG(false, "something is wrong");
736  }
737 
738  // indices in the interior of the cell:
739  if (dim_mesh - codim > 1) {
740  auto offset = aux_node_offset_[0](*e);
741  for (unsigned i = 0; i < NumAuxNodes(e->RefEl(), order); ++i) {
742  node_indices.push_back(offset + i);
743  }
744  }
745 
746  if (order_ == 1) {
747  if (ref_el == base::RefEl::kSegment()) {
748  vtk_file_.unstructured_grid.cell_types[index] =
749  VtkFile::CellType::VTK_LINE;
750  } else if (ref_el == base::RefEl::kTria()) {
751  vtk_file_.unstructured_grid.cell_types[index] =
752  VtkFile::CellType::VTK_TRIANGLE;
753  } else if (ref_el == base::RefEl::kQuad()) {
754  vtk_file_.unstructured_grid.cell_types[index] =
755  VtkFile::CellType::VTK_QUAD;
756  }
757  } else {
758  switch (ref_el) {
759  case base::RefEl::kSegment():
760  vtk_file_.unstructured_grid.cell_types[index] =
761  VtkFile::CellType::VTK_LAGRANGE_CURVE;
762  break;
763  case base::RefEl::kTria():
764  vtk_file_.unstructured_grid.cell_types[index] =
765  VtkFile::CellType::VTK_LAGRANGE_TRIANGLE;
766  break;
767  case base::RefEl::kQuad():
768  vtk_file_.unstructured_grid.cell_types[index] =
769  VtkFile::CellType::VTK_LAGRANGE_QUADRILATERAL;
770  break;
771  default:
772  LF_VERIFY_MSG(false, "Unsupported RefElType " << ref_el);
773  }
774  }
775  }
776 }
