128  const std::vector<Eigen::Matrix<S2, R2, C2, O2, MR2, MC2>>& v,
129  int /*unused*/) const {
130  using scalar_t = decltype(S1(0) + S2(0));
131  if constexpr (R1 != Eigen::Dynamic && C1 != Eigen::Dynamic &&
132  R2 != Eigen::Dynamic && C2 != Eigen::Dynamic) { // NOLINT
133  // add two static size eigen matrices to each other
134  static_assert(R1 == R2, "cannot add matrices with different #rows.");
135  static_assert(C1 == C2, "cannot add matrices with different #cols.");

138  &u[0](0, 0), 1, u.size() * R1 * C1);
139  Eigen::Map<const Eigen::Matrix<S2, 1, Eigen::Dynamic>> vm(
140  &v[0](0, 0), 1, v.size() * R1 * C1);
141 
142  std::vector<Eigen::Matrix<scalar_t, R1, C1>> result(u.size());
143  Eigen::Map<Eigen::Matrix<scalar_t, 1, Eigen::Dynamic>> rm(
144  &result[0](0, 0), 1, u.size() * R1 * C1);
145  rm = um + vm;
146  return result;
147  }

149  (R2 != Eigen::Dynamic && C2 != Eigen::Dynamic)) { // NOLINT

152  constexpr int fixedCols = std::max(C1, C2);
153 
154  std::vector<Eigen::Matrix<scalar_t, fixedRows, fixedCols>> result(
155  u.size());

157  LF_ASSERT_MSG(u[i].rows() == v[i].rows(),
158  "mismatch in #rows of matrices added to each other.");
159  LF_ASSERT_MSG(u[i].cols() == v[i].cols(),
160  "mismatch in #cols of matrices added to each other.");
161  result[i] = u[i] + v[i];
162  }

164  } else { // NOLINT

166  std::vector<Eigen::Matrix<scalar_t, Eigen::Dynamic, Eigen::Dynamic>>
