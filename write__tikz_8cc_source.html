24 bool writeTikZ(const lf::mesh::Mesh &mesh, const std::string &filename,
25  std::function<bool(const lf::mesh::Entity &)> &&selector,
26  TikzOutputCtrl output_ctrl) {
27  // Open output file for writing and quit in case of failure
28  std::ofstream outfile(filename);
29  if (!outfile) {
30  return false;
31  }
32  // ----------------------------------------------------------------
33  // For the enum flags: TikzOutputCtrl
34  bool EdgeNumOn =
35  static_cast<bool>(output_ctrl & TikzOutputCtrl::EdgeNumbering);
36  bool NodeNumOn =
37  static_cast<bool>(output_ctrl & TikzOutputCtrl::NodeNumbering);
38  bool CellNumOn =
39  static_cast<bool>(output_ctrl & TikzOutputCtrl::CellNumbering);
40  bool VerticeNumOn =
41  static_cast<bool>(output_ctrl & TikzOutputCtrl::VerticeNumbering);
42  bool RenderCellsOn =
43  static_cast<bool>(output_ctrl & TikzOutputCtrl::RenderCells);
44  bool ArrowsOn = static_cast<bool>(output_ctrl & TikzOutputCtrl::ArrowTips);
45  bool WithPreamble =
46  static_cast<bool>(output_ctrl & TikzOutputCtrl::WithPreamble);
47 
48  using size_type = std::size_t; // lf::base::size_type;
49  using dim_t = lf::base::RefEl::dim_t; // lf::base::dim_t;
50  const Eigen::MatrixXd zero(Eigen::MatrixXd::Zero(0, 1));
51 
52  // Obtain topological dimension of the mesh
53  const dim_t dim_mesh = mesh.DimMesh();
54  LF_VERIFY_MSG(dim_mesh == 2, "writeTikZ() only available for 2D meshes");
55 
56  // Run through nodes
57  const dim_t node_codim(dim_mesh); // Codimension number for nodes in the mesh
58  const size_type no_of_nodes =
59  mesh.NumEntities(node_codim); // No. of nodes in codimension for nodes
60  size_type node_count = 0;
61 
62  // START writing to file
63  if (WithPreamble) {
64  outfile << "\\documentclass[12pt,a4paper]{article}\n\\usepackage{tikz}"
65  "\n\\begin{document}\n"
66  << std::endl;
67  }
68  outfile << "% TikZ document graphics \n";
69 
70  // Scale font size for large meshes
71  if (no_of_nodes > 50) {
72  outfile << "\\begin{tikzpicture}[scale=6, >= stealth, inner sep=0pt, "
73  "minimum size=0.2cm]\n";
74  outfile << "\\tikzstyle{every node}=[font=\\tiny]\n";
75  } else {
76  outfile << "\\begin{tikzpicture}[scale=4, >= stealth, inner sep=0pt, "
77  "minimum size=0.35cm]\n";
78  }
79 
80  // ---------------------------------------------------------------------------
81 
82  // Loop through codimensions
83  for (int co_dim = 0; co_dim <= dim_mesh; co_dim++) {
84  // Loop through all types of entities
85  for (const mesh::Entity *obj : mesh.Entities(co_dim)) {
86  if (selector(*obj)) { // IF SELECTOR -------
87  size_type obj_idx = mesh.Index(*obj);
88  lf::base::RefEl obj_refel = obj->RefEl();
89  int num_nodes_obj = base::narrow<int>(obj_refel.NumNodes());
90  const geometry::Geometry *obj_geo_ptr = obj->Geometry();
91  const Eigen::MatrixXd &obj_corners(obj_refel.NodeCoords());
92  const Eigen::MatrixXd vertices = obj_geo_ptr->Global(obj_corners);
93  const Eigen::MatrixXd center =
94  vertices.rowwise().sum() / vertices.cols();
95  Eigen::MatrixXd center_mat(center.rows(), num_nodes_obj);
96 
97  switch (obj_refel) {
98  case lf::base::RefEl::kPoint(): {
99  if (NodeNumOn) {
100  outfile << "\\draw[red, fill = white] (" << vertices(0, 0) << ","
101  << vertices(1, 0) << ") "
102  << "node[draw, circle, fill = white] {" << obj_idx
103  << "};\n";
104  } else {
105  outfile << "\\draw[red] (" << vertices(0, 0) << ","
106  << vertices(1, 0) << ") "
107  << "node[] {*};\n";
108  } // if NodeNumOn
109  break;
110 
111  } // case kPoint
112 
113  case lf::base::RefEl::kSegment(): {
114  center_mat << center, center;
115  const Eigen::MatrixXd scaled_vertices =
116  vertices * 0.80 + center_mat * 0.2;
117  const Eigen::MatrixXd semi_scaled_vertices =
118  vertices * 0.95 + center_mat * 0.05;
119 
120  if (ArrowsOn) {
121  if (EdgeNumOn && NodeNumOn) {
122  outfile << "\\draw[->] (" << scaled_vertices(0, 0) << ","
123  << scaled_vertices(1, 0) << ") -- node[black] {"
124  << obj_idx << "} "
125  << "(" << scaled_vertices(0, 1) << ","
126  << scaled_vertices(1, 1) << ");\n";
127  } else if (NodeNumOn && !EdgeNumOn) {
128  outfile << "\\draw[->] (" << scaled_vertices(0, 0) << ","
129  << scaled_vertices(1, 0) << ") -- "
130  << "(" << scaled_vertices(0, 1) << ","
131  << scaled_vertices(1, 1) << ");\n";
132  } else if (!NodeNumOn && EdgeNumOn) {
133  outfile << "\\draw[->] (" << semi_scaled_vertices(0, 0) << ","
134  << semi_scaled_vertices(1, 0) << ") -- node[black] {"
135  << obj_idx << "} "
136  << "(" << semi_scaled_vertices(0, 1) << ","
137  << semi_scaled_vertices(1, 1) << ");\n";
138  } else if (!NodeNumOn && !EdgeNumOn) {
139  outfile << "\\draw[->] (" << semi_scaled_vertices(0, 0) << ","
140  << semi_scaled_vertices(1, 0) << ") -- "
141  << "(" << semi_scaled_vertices(0, 1) << ","
142  << semi_scaled_vertices(1, 1) << ");\n";
143  } else {
144  std::cout << "Check EdgeNumOn and NodeNumOn for kSegment "
145  << obj_idx << std::endl;
146  }
147 
148  } else {
149  if (EdgeNumOn) {
150  outfile << "\\draw[] (" << vertices(0, 0) << ","
151  << vertices(1, 0) << ") -- node[black] {" << obj_idx
152  << "} "
153  << "(" << vertices(0, 1) << "," << vertices(1, 1)
154  << ");\n";
155  } else if (!EdgeNumOn) {
156  outfile << "\\draw[] (" << vertices(0, 0) << ","
157  << vertices(1, 0) << ") -- "
158  << "(" << vertices(0, 1) << "," << vertices(1, 1)
159  << ");\n";
160  } else {
161  std::cout << "Check EdgeNumOn and NodeNumOn for kSegment "
162  << obj_idx << std::endl;
163  }
164  } // arrows on
165 
166  break;
167  } // case kSegment
168 
169  case lf::base::RefEl::kTria(): {
170  center_mat << center, center, center;
171  const Eigen::MatrixXd scaled_vertices =
172  vertices * 0.70 + center_mat * 0.30;
173 
174  if (RenderCellsOn) {
175  if (VerticeNumOn) {
176  outfile << "\\draw[green] (" << scaled_vertices(0, 0) << ","
177  << scaled_vertices(1, 0) << ") node[] {0} -- ("
178  << scaled_vertices(0, 1) << "," << scaled_vertices(1, 1)
179  << ") node[] {1} -- (" << scaled_vertices(0, 2) << ","
180  << scaled_vertices(1, 2) << ") node[] {2} -- cycle;\n";
181  } else {
182  outfile << "\\draw[green] (" << scaled_vertices(0, 0) << ","
183  << scaled_vertices(1, 0) << ") -- ("
184  << scaled_vertices(0, 1) << "," << scaled_vertices(1, 1)
185  << ") -- (" << scaled_vertices(0, 2) << ","
186  << scaled_vertices(1, 2) << ") -- cycle;\n";
187  } // if EdgeNumOn
188 
189  if (CellNumOn) {
190  outfile << "\\draw[green] (" << center(0, 0) << ","
191  << center(1, 0) << ") node[] {" << obj_idx << "};\n";
192  }
193 
194  } // RenderCellsOn
195  break;
196  } // case kTria
197 
198  case lf::base::RefEl::kQuad(): {
199  center_mat << center, center, center, center;
200  const Eigen::MatrixXd scaled_vertices =
201  vertices * 0.70 + center_mat * 0.3;
202 
203  if (RenderCellsOn) {
204  if (VerticeNumOn) {
205  outfile << "\\draw[magenta] (" << scaled_vertices(0, 0) << ","
206  << scaled_vertices(1, 0) << ") node[] {0} -- ("
207  << scaled_vertices(0, 1) << "," << scaled_vertices(1, 1)
208  << ") node[] {1} -- (" << scaled_vertices(0, 2) << ","
209  << scaled_vertices(1, 2) << ") node[] {2} -- ("
210  << scaled_vertices(0, 3) << "," << scaled_vertices(1, 3)
211  << ") node[] {3} -- cycle;\n";
212  } else {
213  outfile << "\\draw[magenta] (" << scaled_vertices(0, 0) << ","
214  << scaled_vertices(1, 0) << ") -- ("
215  << scaled_vertices(0, 1) << "," << scaled_vertices(1, 1)
216  << ") -- (" << scaled_vertices(0, 2) << ","
217  << scaled_vertices(1, 2) << ") -- ("
218  << scaled_vertices(0, 3) << "," << scaled_vertices(1, 3)
219  << ") -- cycle;\n";
220  }
221 
222  if (CellNumOn) {
223  outfile << "\\draw[magenta] (" << center(0, 0) << ","
224  << center(1, 0) << ") node[] {" << obj_idx << "};\n";
225  }
226 
227  } // RenderCellsOn
228 
229  break;
230  } // case kQuad
231 
232  default: {
233  std::cout << "Error for object " << obj_idx << " in co-dim "
234  << co_dim << std::endl;
235  std::cout << "Object type: " << obj_refel << std::endl;
236  break;
237  } // default
238  } // switch
239  } // IF SELECTOR --------------------
240  } // for entities
241 
242  node_count++;
243  // LF_VERIFY_MSG(node_count == no_of_nodes, "Node count mismatch");
244  } // for codim
245 
246  outfile << "\\end{tikzpicture}\n";
247  if (WithPreamble) {
248  outfile << "\n\\end{document}" << std::endl;
249  }
250  return true;
251 } // writetikz
