27 double legendre(unsigned n, double x, double t) {
28  if (n == 0) {
29  return 1;
30  }
31  if (n == 1) {
32  return 2 * x - t;
33  }
34  double Pim1 = 1;
35  double Pi = 2 * x - t;
36  for (unsigned i = 2; i <= n; ++i) {
37  const double Pip1 =
38  (2 * i - 1) * (2 * x - t) / i * Pi - (i - 1) * t * t / i * Pim1;
39  Pim1 = Pi;
40  Pi = Pip1;
41  }
42  return Pi;
43 }

146 double jacobi(unsigned n, double alpha, double beta, double x) {
147  // The recurrence relation is for the non-shifted Jacobi Polynomials
148  // We thus map [0, 1] onto [-1, 1]
149  x = 2 * x - 1;
150  if (n == 0) {
151  return 1;
152  }
153  if (n == 1) {
154  return 0.5 * (alpha - beta + x * (alpha + beta + 2));
155  }
156  double p0 = 1;
157  double p1 = 0.5 * (alpha - beta + x * (alpha + beta + 2));
158  double p2 = 0;
159  for (unsigned j = 1; j < n; ++j) {
160  const double alpha1 =
161  2 * (j + 1) * (j + alpha + beta + 1) * (2 * j + alpha + beta);
162  const double alpha2 =
163  (2 * j + alpha + beta + 1) * (alpha * alpha - beta * beta);
164  const double alpha3 = (2 * j + alpha + beta) * (2 * j + alpha + beta + 1) *
165  (2 * j + alpha + beta + 2);
166  const double alpha4 =
167  2 * (j + alpha) * (j + beta) * (2 * j + alpha + beta + 2);
168  p2 = 1. / alpha1 * ((alpha2 + alpha3 * x) * p1 - alpha4 * p0);
169  p0 = p1;
170  p1 = p2;
171  }
172  return p2;
173 }

207 double ijacobi(unsigned n, double alpha, double x) {
208  if (n == 0) {
209  return -1;
210  }
211  if (n == 1) {
212  return x;
213  }
214  // Compute the n-th, (n-1)-th and (n-2)-th Jacobi Polynomial
215  // This uses the same recurrence relation as the `eval` method
216  double ajp1P = 2 * 2 * (2 + alpha) * (2 * 2 + alpha - 2);
217  double bjp1P = 2 * 2 + alpha - 1;
218  double cjp1P = (2 * 2 + alpha) * (2 * 2 + alpha - 2);
219  double djp1P = 2 * (2 + alpha - 1) * (2 - 1) * (2 * 2 + alpha);
220  double Pjm2 = 1;
221  double Pjm1 = (2 + alpha) * x - 1;
222  double Pj =
223  (bjp1P * (cjp1P * (2 * x - 1) + alpha * alpha) * Pjm1 - djp1P * Pjm2) /
224  ajp1P;
225  for (unsigned j = 2; j < n; ++j) {
226  ajp1P = 2 * (j + 1) * ((j + 1) + alpha) * (2 * (j + 1) + alpha - 2);
227  bjp1P = 2 * (j + 1) + alpha - 1;
228  cjp1P = (2 * (j + 1) + alpha) * (2 * (j + 1) + alpha - 2);
229  djp1P = 2 * ((j + 1) + alpha - 1) * ((j + 1) - 1) * (2 * (j + 1) + alpha);
230  double Pjp1 =
231  (bjp1P * (cjp1P * (2 * x - 1) + alpha * alpha) * Pj - djp1P * Pjm1) /
232  ajp1P;
233  Pjm2 = Pjm1;
234  Pjm1 = Pj;
235  Pj = Pjp1;
236  }
237  // Compute the integral by the formula in the docstring
238  const double anL = (n + alpha) / ((2 * n + alpha - 1) * (2 * n + alpha));
239  const double bnL = alpha / ((2 * n + alpha - 2) * (2 * n + alpha));
240  const double cnL = (n - 1) / ((2 * n + alpha - 2) * (2 * n + alpha - 1));
241  return anL * Pj + bnL * Pjm1 - cnL * Pjm2;
242 }
