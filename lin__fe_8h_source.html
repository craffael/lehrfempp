162 typename LinearFELocalLoadVector<SCALAR, FUNCTOR>::ElemVec
163 LinearFELocalLoadVector<SCALAR, FUNCTOR>::Eval(
164  const lf::mesh::Entity &cell) const {
165  // Topological type of the cell
166  const lf::base::RefEl ref_el{cell.RefEl()};
167  const lf::base::size_type num_nodes{ref_el.NumNodes()};
168 
169  // Obtain the vertex coordinates of the cell, which completely
170  // describe its shape.
171  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
172  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
173  LF_ASSERT_MSG((geo_ptr->DimGlobal() == 2) && (geo_ptr->DimLocal() == 2),
174  "Only 2D implementation available!");
175  const Eigen::MatrixXd &ref_el_corners(ref_el.NodeCoords());
176  // World coordinates of vertices
177  // const Eigen::MatrixXd vertices{geo_ptr->Global(ref_el_corners)};
178  // Debugging output
179  SPDLOG_LOGGER_TRACE(LinearFeLocalLoadVectorLogger(), "{}, shape = \n{}",
180  ref_el, geo_ptr->Global(ref_el_corners));
181 
182  // Midpoints of edges in the reference cell
183  Eigen::MatrixXd ref_mp(2, 4);
184  switch (ref_el) {
185  case lf::base::RefEl::kTria(): {
186  // clang-format off
187  ref_mp << 0.5,0.5,0.0,-1.0,
188  0.0,0.5,0.5,-1.0;
189  // clang-format on
190  break;
191  }
192  case lf::base::RefEl::kQuad(): {
193  // clang-format off
194  ref_mp << 0.5,1,0.5,0.0,
195  0.0,0.5,1.0,0.5;
196  // clang-format on
197  break;
198  }
199  default: {
200  LF_ASSERT_MSG(false, "Illegal entity type!");
201  break;
202  }
203  } // end switch
204 
205  const double area = lf::geometry::Volume(*geo_ptr);
206 
207  // Vector for returning element vector
208  ElemVec elem_vec = ElemVec::Zero();
209  // Run over the midpoints of edges and fetch values of the source function
210  // there
211  auto fvals = f_(cell, ref_mp);
212 
213  for (int k = 0; k < num_nodes; k++) {
214  elem_vec[k] += 0.5 * fvals[k];
215  elem_vec[(k + 1) % num_nodes] += 0.5 * fvals[k];
216  }
217  SPDLOG_LOGGER_TRACE(LinearFeLocalLoadVectorLogger(), "element vector = {}",
218  elem_vec.head(num_nodes).transpose());
219 
220  return (area / num_nodes) * elem_vec;
221 }
