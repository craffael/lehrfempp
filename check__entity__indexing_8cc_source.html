7 void checkEntityIndexing(const Mesh& mesh) {
8  using size_type = lf::base::size_type; // type for indices
9  const auto dim_mesh = mesh.DimMesh(); // dimension of meshed manifold
10  // Now run over all co-dimensions to check indexing
11  for (size_type co_dim = 0; co_dim <= dim_mesh; ++co_dim) {
12  // Number of entities of current co-dimension
13  const auto no_of_entities = mesh.NumEntities(co_dim);
14  // counting array for occurrences of an index
15  std::vector<int> idx_use(no_of_entities, 0);
16  // Traverse all entities of a given co-dimension
17  for (const Entity* e : mesh.Entities(co_dim)) {
18  // Fetch index of current entity
19  const lf::base::glb_idx_t current_idx = mesh.Index(*e);
20  // Check whether index is in range and, if so, count it
21  EXPECT_LT(current_idx, no_of_entities)
22  << "Index " << current_idx << " out of range";
23  idx_use[current_idx]++;
24 
25  // Check consistency of indexing
26  const lf::mesh::Entity* e_ptr = mesh.EntityByIndex(co_dim, current_idx);
27  EXPECT_TRUE(e_ptr) << "Invalid pointer to entity "
28  << static_cast<int>(current_idx);
29  if (e_ptr != nullptr) {
30  EXPECT_EQ(current_idx, mesh.Index(*e_ptr))
31  << "Index mismatch for codim " << co_dim << ", index "
32  << current_idx;
33  }
34  } // end loop over entities
35 
36  // Indices should occur only once. This means that the counting array
37  // should contain 1 in every component.
38  for (size_type idx_cnt = 0; idx_cnt < no_of_entities; ++idx_cnt) {
39  // NOLINTNEXTLINE
40  EXPECT_EQ(idx_use[idx_cnt], 1)
41  << "Index " << idx_cnt << " occurs " << idx_use[idx_cnt] << " times!";
42  }
43  } // end for
44 } // end checkEntityIndexing()
