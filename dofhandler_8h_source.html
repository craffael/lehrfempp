557 template <typename LOCALDOFINFO>
558 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
559 DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
560 LOCALDOFINFO &&locdof)
561 : mesh_p_(std::move(mesh_p)) {
562 LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
563
564 // Index counter for global shape functions = global dof
565 gdof_idx_t dof_idx = 0;
566
567 // Step I: Set indices for shape functions on nodes
568 // Run through node indices (entities of co-dimension 2)
569 const size_type no_nodes = mesh_p_->NumEntities(2);
570 num_int_dofs_[2].resize(no_nodes, 0);
571 offsets_[2].resize(no_nodes + 1, 0);
572 // Traverse nodes (co-dimension-2 entities) based on indices
573 for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
574 // Obtain pointer to node entity
575 const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
576 LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
577 // Offset for indices of node in index vector
578 const glb_idx_t node_dof_offset = dof_idx;
579 offsets_[2][node_idx] = node_dof_offset;
580 // Request number of local shape functions associated with the node
581 const size_type no_int_dof_node = locdof(*node_p);
582 num_int_dofs_[2][node_idx] = no_int_dof_node;
583
584 // Store dof indices in array
585 for (unsigned j = 0; j < no_int_dof_node; j++) {
586 dofs_[2].push_back(dof_idx);
587 dof_entities_.push_back(node_p); // Store entity for current dof
588 dof_idx++; // Move on to next index
589 }
590 }
591 // Set sentinel
592 offsets_[2][no_nodes] = dof_idx;
593
594 // Step II: Set indices for shape functions on edges (co-dimension = 1)
595 const size_type no_edges = mesh_p_->NumEntities(1);
596 // Set length of edge-related index vectors
597 num_int_dofs_[1].resize(no_edges, 0);
598 offsets_[1].resize(no_edges + 1, 0);
599 // points to the position of the current dof index
600 size_type edge_dof_offset = 0;
601 for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
602 // Obtain pointer to edge entity
603 const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
604 LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
605 // Offset for indices of edge dof in index vector
606 offsets_[1][edge_idx] = edge_dof_offset;
607 const size_type no_int_dof_edge = locdof(*edge);
608 num_int_dofs_[1][edge_idx] = no_int_dof_edge;
609
610 // Obtain indices for basis functions sitting at endpoints
611 // Endpoints are mesh entities with co-dimension = 2
612 for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
613 const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
614 const glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
615 const size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
616 // Copy indices of shape functions from nodes to edge
617 for (unsigned j = 0; j < no_int_dofs_ep; j++) {
618 dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
619 edge_dof_offset++;
620 }
621 }
622 // Set indices for interior edge degrees of freedom
623 for (unsigned j = 0; j < no_int_dof_edge; j++) {
624 dofs_[1].push_back(dof_idx);
625 edge_dof_offset++;
626 dof_entities_.push_back(edge);
627 dof_idx++;
628 }
629 }
630 // Set sentinel
631 offsets_[1][no_edges] = edge_dof_offset;
632
633 // Step III: Set indices for shape functions on cells (co-dimension = 0)
634 const size_type no_cells = mesh_p_->NumEntities(0);
635 // Set length of cell-related index vectors
636 num_int_dofs_[0].resize(no_cells, 0);
637 offsets_[0].resize(no_cells + 1, 0);
638 // points to the position of the current dof index
639 size_type cell_dof_offset = 0;
640 for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
641 // Obtain pointer to current ell
642 const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
643 // Offset for indices of cell dof in index vector
644 offsets_[0][cell_idx] = cell_dof_offset;
645 const size_type no_int_dof_cell = locdof(*cell);
646 num_int_dofs_[0][cell_idx] = no_int_dof_cell;
647
648 // Obtain indices for basis functions in vertices
649 for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
650 const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
651 const glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
652 const size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
653 // Copy indices of shape functions from nodes to cell
654 for (unsigned j = 0; j < no_int_dofs_vt; j++) {
655 dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
656 cell_dof_offset++;
657 }
658 }
659
660 // Collect indices of interior shape functions of edges
661 // Internal ordering will depend on the orientation of the edge
662 auto edge_orientations = cell->RelativeOrientations();
663 auto edges = cell->SubEntities(1);
664 // Loop over edges
665 for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
666 ed_sub_idx++) {
667 const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
668 const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
669 const glb_idx_t edge_int_dof_offset =
670 offsets_[1][edge_idx + 1] - no_int_dof_edge;
671
672 // Copy indices of shape functions from edges to cell
673 // The order, in which they are copied depends on the relative
674 // orientation of the edge w.r.t. the cell
675 switch (edge_orientations[ed_sub_idx]) {
676 case lf::mesh::Orientation::positive: {
677 for (int j = 0; j < no_int_dof_edge; j++) {
678 dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
679 cell_dof_offset++;
680 }
681 break;
682 }
683 case lf::mesh::Orientation::negative: {
684 for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
685 dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
686 cell_dof_offset++;
687 }
688 break;
689 }
690 } // end switch
691 } // end loop over edges
692
693 // Set indices for interior shape functions of the cell
694 for (unsigned j = 0; j < no_int_dof_cell; j++) {
695 dofs_[0].push_back(dof_idx);
696 cell_dof_offset++;
697 dof_entities_.push_back(cell);
698 dof_idx++;
699 } // end loop over interior dof of cell
700 } // end loop over cells
701 // Set sentinel
702 offsets_[0][no_cells] = cell_dof_offset;
703
704 // Finally set number of global shape functions
705 num_dof_ = dof_idx;
706 } // end constructor
