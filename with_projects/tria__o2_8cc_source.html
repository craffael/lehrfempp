14 TriaO2::TriaO2(Eigen::Matrix<double, Eigen::Dynamic, 6> coords)
15  : coords_(std::move(coords)),
16  alpha_(coords_.rows()),
17  beta_(coords_.rows(), 2),
18  gamma_(coords_.rows(), 2),
19  delta_(coords_.rows()),
20  gamma_x_2_(coords_.rows(), 2) {
21  /*
22  * 2 C
23  * | \ | \
24  * 5 4 -> F E
25  * | \ | \
26  * 0 - 3 - 1 A - D - B
27  */
28  const Eigen::VectorXd& A = coords_.col(0);
29  const Eigen::VectorXd& B = coords_.col(1);
30  const Eigen::VectorXd& C = coords_.col(2);
31  const Eigen::VectorXd& D = coords_.col(3);
32  const Eigen::VectorXd& E = coords_.col(4);
33  const Eigen::VectorXd& F = coords_.col(5);
34 
35  // Compute monomial coeffcients of componentwise quadratic
36  // polynomial
37  alpha_ << A;
38  beta_ << 4. * D - 3. * A - B, 4. * F - 3. * A - C;
39  gamma_ << 2. * (A + B) - 4. * D, 2. * (A + C) - 4. * F;
40  delta_ << 4. * (A + E - D - F);
41  // coefficient for Jacobian()
42  gamma_x_2_ << 2. * gamma_;
43 }

46 Eigen::MatrixXd TriaO2::Global(const Eigen::MatrixXd& local) const {
47  LF_VERIFY_MSG((0. <= local.array()).all() && (local.array() <= 1.).all(),
48  "local coordinates out of bounds for reference element");
49  // Direct vectorized evaluation of componentwise quadratic mapping
50  // given through its monomial coefficients.
51  return ((beta_ * local) + (gamma_ * local.array().square().matrix()) +
52  (delta_ * local.row(0).cwiseProduct(local.row(1))))
53  .colwise() +
54  alpha_;
55 }

60 Eigen::MatrixXd TriaO2::Jacobian(const Eigen::MatrixXd& local) const {
61  LF_VERIFY_MSG((0. <= local.array()).all() && (local.array() <= 1.).all(),
62  "local coordinates out of bounds for reference element");
63 
64  Eigen::MatrixXd tmp(gamma_.rows(), 2 * local.cols());
65 
66  for (long i = 0; i < local.cols(); ++i) {
67  tmp.block(0, 2 * i, tmp.rows(), 2) =
68  gamma_x_2_.array().rowwise() * local.col(i).transpose().array();
69  }
70 
71  Eigen::MatrixXd local_reversed = local.colwise().reverse();
72  local_reversed.resize(1, local.size());
73 
74  return beta_.replicate(1, local.cols()) + tmp +
75  (local_reversed.replicate(delta_.rows(), 1).array().colwise() *
76  delta_.array())
77  .matrix();
78 }

83 Eigen::MatrixXd TriaO2::JacobianInverseGramian(
84  const Eigen::MatrixXd& local) const {
85  LF_VERIFY_MSG((0. <= local.array()).all() && (local.array() <= 1.).all(),
86  "local coordinates out of bounds for reference element");
87 
88  Eigen::MatrixXd jac = Jacobian(local);
89  Eigen::MatrixXd jacInvGram(jac.rows(), jac.cols());
90 
91  if (DimGlobal() == 2) {
92  for (long i = 0; i < local.cols(); ++i) {
93  auto jacobian = jac.block(0, 2 * i, 2, 2);
94  jacInvGram.block(0, 2 * i, 2, 2) << jacobian(1, 1), -jacobian(1, 0),
95  -jacobian(0, 1), jacobian(0, 0);
96  jacInvGram.block(0, 2 * i, 2, 2) /= jacobian.determinant();
97  }
98  } else {
99  for (long i = 0; i < local.cols(); ++i) {
100  auto jacobian = jac.block(0, 2 * i, jac.rows(), 2);
101 
102  auto A = jacobian.col(0);
103  auto B = jacobian.col(1);
104  auto AB = A.dot(B);
105 
106  Eigen::MatrixXd tmp(2, 2);
107  tmp << B.dot(B), -AB, -AB, A.dot(A);
108 
109  jacInvGram.block(0, 2 * i, jac.rows(), 2) =
110  jacobian * tmp / tmp.determinant();
111  }
112  }
113 
114  return jacInvGram;
115 }

119 Eigen::VectorXd TriaO2::IntegrationElement(const Eigen::MatrixXd& local) const {
120  LF_VERIFY_MSG((0. <= local.array()).all() && (local.array() <= 1.).all(),
121  "local coordinates out of bounds for reference element");
122 
123  Eigen::MatrixXd jac = Jacobian(local);
124  Eigen::VectorXd intElem(local.cols());
125 
126  if (DimGlobal() == 2) {
127  for (long i = 0; i < local.cols(); ++i) {
128  intElem(i) = std::abs(jac.block(0, 2 * i, 2, 2).determinant());
129  }
130  } else {
131  for (long i = 0; i < local.cols(); ++i) {
132  auto jacobian = jac.block(0, 2 * i, jac.rows(), 2);
133 
134  auto A = jacobian.col(0);
135  auto B = jacobian.col(1);
136  auto AB = A.dot(B);
137 
138  intElem(i) = std::sqrt(std::abs(A.dot(A) * B.dot(B) - AB * AB));
139  }
140  }
141  return intElem;
142 }

146  switch (codim) {
147  case 0: {

158  }

169 std::vector<std::unique_ptr<Geometry>> TriaO2::ChildGeometry(
170  const RefinementPattern& ref_pat, lf::base::dim_t codim) const {
171  LF_VERIFY_MSG(ref_pat.RefEl() == lf::base::RefEl::kTria(),
172  "Refinement pattern for " << ref_pat.RefEl().ToString());
173  LF_VERIFY_MSG(codim < 3, "Illegal codim " << codim);
174 
175  const double hLattice = 1. / static_cast<double>(ref_pat.LatticeConst());
176  std::vector<std::unique_ptr<Geometry>> childGeoPtrs = {};
177 
178  // get coordinates of childGeometries
179  std::vector<Eigen::Matrix<int, Eigen::Dynamic, Eigen::Dynamic>> childPolygons(
180  ref_pat.ChildPolygons(codim));
181 
182  const int noChildren = childPolygons.size();
183  LF_VERIFY_MSG(
184  noChildren == ref_pat.NumChildren(codim),
185  "NumChildren " << noChildren << " <-> " << ref_pat.NumChildren(codim));
186 
187  // create a geometry object for each child
188  for (size_t child = 0; child < noChildren; ++child) {
189  // codim == 0: a child triangle is described by a lattice polygon with six
190  // vertices
191  // codim == 1: a child segment is described by a polygon with three vertices
192  // codim == 2: a child point by a single point ("polygon with one corner")
193  LF_VERIFY_MSG(
194  childPolygons[child].rows() == 2,
195  "childPolygons[child].rows() = " << childPolygons[child].rows());
196  LF_VERIFY_MSG(
197  childPolygons[child].cols() == (3 - codim),
198  "childPolygons[child].cols() = " << childPolygons[child].cols());
199 
200  const Eigen::MatrixXd locChildPolygonCoords(
201  hLattice * childPolygons[child].cast<double>());
202 
203  switch (codim) {
204  case 0: {
205  Eigen::MatrixXd locChildCoords(locChildPolygonCoords.rows(), 6);
206  locChildCoords << locChildPolygonCoords,
207  (locChildPolygonCoords.col(0) + locChildPolygonCoords.col(1)) / 2.,
208  (locChildPolygonCoords.col(1) + locChildPolygonCoords.col(2)) / 2.,
209  (locChildPolygonCoords.col(2) + locChildPolygonCoords.col(0)) / 2.;
210 
211  childGeoPtrs.push_back(
212  std::make_unique<TriaO2>(Global(locChildCoords)));
213 
214  break;
215  }
216  case 1: {
217  Eigen::MatrixXd locChildCoords(locChildPolygonCoords.rows(), 3);
218  locChildCoords << locChildPolygonCoords,
219  (locChildPolygonCoords.col(0) + locChildPolygonCoords.col(1)) / 2.;
220 
221  childGeoPtrs.push_back(
222  std::make_unique<SegmentO2>(Global(locChildCoords)));
223 
224  break;
225  }
226  case 2: {
227  childGeoPtrs.push_back(
228  std::make_unique<Point>(Global(locChildPolygonCoords)));
229 
230  break;
231  }
232  default: {
233  LF_VERIFY_MSG(false, "Illegal co-dimension");
234  }
235  }
236  }
237 
238  return childGeoPtrs;
239 }
