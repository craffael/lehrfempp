26 void PrintInfo(std::ostream &stream, const DofHandler &dof_handler,
27  unsigned ctrl) {
28  // Obtain pointer to the underlying mesh
29  auto mesh = dof_handler.Mesh();
30  // Number of degrees of freedom managed by the DofHandler object
31  const lf::assemble::size_type N_dofs(dof_handler.NumDofs());
32 
33  stream << "DofHandler(" << dof_handler.NumDofs() << " dofs)";
34  if (ctrl > 0) {
35  // More detailed output
36  stream << std::endl;
37  if (ctrl % 2 == 0) {
38  for (lf::base::dim_t codim = 0; codim <= mesh->DimMesh(); codim++) {
39  // Visit all entities of a specific codimension
40  for (const lf::mesh::Entity *e : mesh->Entities(codim)) {
41  // Fetch unique index of current entity supplied by mesh object
42  const lf::base::glb_idx_t e_idx = mesh->Index(*e);
43  // Number of shape functions covering current entity
44  const lf::assemble::size_type no_dofs(dof_handler.NumLocalDofs(*e));
45  // Obtain global indices of those shape functions ...
46  nonstd::span<const gdof_idx_t> doflist(
47  dof_handler.GlobalDofIndices(*e));
48  // and print them
49  stream << *e << ' ' << e_idx << ": " << no_dofs << " dofs = [";
50  for (const lf::assemble::gdof_idx_t &dof : doflist) {
51  stream << dof << ' ';
52  }
53  stream << ']';
54  if (ctrl % 5 == 0) {
55  // Also output indices of interior shape functions
56  nonstd::span<const gdof_idx_t> intdoflist(
57  dof_handler.InteriorGlobalDofIndices(*e));
58  stream << " int = [";
59  for (lf::assemble::gdof_idx_t int_dof : intdoflist) {
60  stream << int_dof << ' ';
61  }
62  stream << ']';
63  }
64  stream << std::endl;
65  }
66  }
67  }
68  if (ctrl % 3 == 0) {
69  // List entities associated wit the dofs managed by the current
70  // DofHandler object
71  for (lf::assemble::gdof_idx_t dof_idx = 0; dof_idx < N_dofs; dof_idx++) {
72  const lf::mesh::Entity &e(dof_handler.Entity(dof_idx));
73  stream << "dof " << dof_idx << " -> " << e << " " << mesh->Index(e)
74  << std::endl;
75  }
76  }
77  }
78 }

84 UniformFEDofHandler::UniformFEDofHandler(
85  std::shared_ptr<const lf::mesh::Mesh> mesh, dof_map_t dofmap)
86  : mesh_(std::move(mesh)), num_dofs_() {
87  LF_ASSERT_MSG((mesh_->DimMesh() == 2), "Can handle 2D meshes only");
88 
89  // For checking whether a key was found
90  auto map_end = dofmap.end();
91 
92  // Get no of interior dof specified for nodes
93  auto map_el_pt = dofmap.find(lf::base::RefEl::kPoint());
94  if (map_el_pt != map_end) {
95  num_loc_dof_point_ = map_el_pt->second;
96  }
97 
98  // Get no of interior dof specified for edges
99  auto map_el_ed = dofmap.find(lf::base::RefEl::kSegment());
100  if (map_el_ed != map_end) {
101  num_loc_dof_segment_ = map_el_ed->second;
102  }
103 
104  // Get no of interior dof specified for triangles
105  auto map_el_tr = dofmap.find(lf::base::RefEl::kTria());
106  if (map_el_tr != map_end) {
107  num_loc_dof_tria_ = map_el_tr->second;
108  }
109 
110  // Get no of interior dof specified for quads
111  auto map_el_qd = dofmap.find(lf::base::RefEl::kQuad());
112  if (map_el_qd != map_end) {
113  num_loc_dof_quad_ = map_el_qd->second;
114  }
115 
116  // If an entity type is not represented in the map, we assume that
117  // no shape functions are attached to those entities
118 
119  // Initialize total number of shape functions covering an entity.
120  InitTotalNumDofs();
121 
122  // Initializatin of dof index arrays
123  initIndexArrays();
124 }

126 void UniformFEDofHandler::InitTotalNumDofs() {
127  num_dofs_[kNodeOrd] = num_loc_dof_point_;
128  num_dofs_[kEdgeOrd] = 2 * num_loc_dof_point_ + num_loc_dof_segment_;
129  num_dofs_tria_ =
130  3 * num_loc_dof_point_ + 3 * num_loc_dof_segment_ + num_loc_dof_tria_;
131  num_dofs_quad_ =
132  4 * num_loc_dof_point_ + 4 * num_loc_dof_segment_ + num_loc_dof_quad_;
133  num_dofs_[kCellOrd] = std::max(num_dofs_tria_, num_dofs_quad_);
134 }

136 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
137 void UniformFEDofHandler::initIndexArrays() {
138  // This method assumes a proper initialization
139  // of the data in no_loc_dof_* and num_dofs_, num_dof_tria, num_dofs_quad_
140  gdof_idx_t dof_idx = 0;
141 
142  // Step I: Set indices for shape functions on nodes
143  // Total number of degrees of freedom on nodes (entities of co-dim = 2)
144  const size_type no_nodes = mesh_->NumEntities(2);
145  const size_type num_dofs_nodes = no_nodes * num_dofs_[kNodeOrd];
146  dofs_[kNodeOrd].resize(num_dofs_nodes);
147  // Run through nodes in the order given by their numbering
148  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
149  const mesh::Entity *node_p{mesh_->EntityByIndex(2, node_idx)};
150  LF_ASSERT_MSG(mesh_->Index(*node_p) == node_idx, "Node index mismatch");
151  // Beginning of section for concrete node in the dof index vector
152  // for entities of co-dimension 2
153  glb_idx_t node_dof_offset = node_idx * num_dofs_[kNodeOrd];
154  for (unsigned j = 0; j < num_loc_dof_point_; j++) {
155  dofs_[kNodeOrd][node_dof_offset++] = dof_idx;
156  dof_entities_.push_back(node_p); // Store entity for current dof
157  dof_idx++; // Move on to next index
158  }
159  }
160 
161  // Step II: Set indices for shape functions on edges
162  // Total number of degrees of freedom belonging to edges (entities of co-dim =
163  // 1)
164  const size_type no_edges = mesh_->NumEntities(1);
165  const size_type num_dofs_edges = no_edges * num_dofs_[kEdgeOrd];
166  dofs_[kEdgeOrd].resize(num_dofs_edges);
167  // Visit all edges
168  // Old implementation, see remarks above
169  // for (const lf::mesh::Entity &edge : mesh_->Entities(1)) {
170  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
171  // Obtain pointer to edge entity
172  const mesh::Entity *edge_p{mesh_->EntityByIndex(1, edge_idx)};
173  LF_ASSERT_MSG(mesh_->Index(*edge_p) == edge_idx, "Edge index mismatch");
174  // Beginning of section for concrete edge in the dof index vector
175  // for entities of co-dimension 1
176  glb_idx_t edge_dof_offset = edge_idx * num_dofs_[kEdgeOrd];
177 
178  // Obtain indices for basis functions sitting at endpoints
179  for (const lf::mesh::Entity *endpoint : edge_p->SubEntities(1)) {
180  const glb_idx_t ep_idx(mesh_->Index(*endpoint));
181  glb_idx_t ep_dof_offset = ep_idx * num_dofs_[kNodeOrd];
182  // Copy indices of shape functions from nodes to edge
183  for (unsigned j = 0; j < num_dofs_[kNodeOrd]; j++) {
184  dofs_[kEdgeOrd][edge_dof_offset++] = dofs_[kNodeOrd][ep_dof_offset++];
185  }
186  }
187  // Set indices for interior edge degrees of freedom
188  for (unsigned j = 0; j < num_loc_dof_segment_; j++) {
189  dofs_[kEdgeOrd][edge_dof_offset++] = dof_idx;
190  dof_entities_.push_back(edge_p);
191  dof_idx++;
192  }
193  }
194 
195  // Step III: Set indices for shape functions on cells
196  const size_type no_cells = mesh_->NumEntities(0);
197  const size_type max_num_dof_cells = no_cells * num_dofs_[kCellOrd];
198  dofs_[kCellOrd].resize(max_num_dof_cells);
199 
200  // Number of (non-)interior shape functins for edges
201  const size_type no_int_dof_edge = num_loc_dof_segment_;
202  const size_type num_ext_dof_edge = num_dofs_[kEdgeOrd] - no_int_dof_edge;
203 
204  // Visit all cells
205  // Old implementation without strong link between cell
206  // indices and ordering of global shape functions
207  // for (const lf::mesh::Entity &cell : mesh_->Entities(0)) {
208  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
209  // Obtain pointer to current ell
210  const mesh::Entity *cell_p{mesh_->EntityByIndex(0, cell_idx)};
211  LF_ASSERT_MSG(cell_idx == mesh_->Index(*cell_p), "cell index mismatch");
212  // Offset for cell dof indices in large dof index vector
213  glb_idx_t cell_dof_offset = cell_idx * num_dofs_[kCellOrd];
214 
215  // Obtain indices for basis functions in vertices
216  for (const lf::mesh::Entity *vertex : cell_p->SubEntities(2)) {
217  const glb_idx_t vt_idx(mesh_->Index(*vertex));
218  glb_idx_t vt_dof_offset = vt_idx * num_dofs_[kNodeOrd];
219  // Copy indices of shape functions from nodes to cell
220  for (unsigned j = 0; j < num_dofs_[kNodeOrd]; j++) {
221  dofs_[kCellOrd][cell_dof_offset++] = dofs_[kNodeOrd][vt_dof_offset++];
222  }
223  }
224 
225  // Collect indices of interior shape functions of edges
226  // Internal ordering will depend on the orientation of the edge
227  auto edge_orientations = cell_p->RelativeOrientations();
228  auto edges = cell_p->SubEntities(1);
229  // Loop over edges
230  const size_type no_edges_cell = cell_p->RefEl().NumSubEntities(1);
231  for (int ed_sub_idx = 0; ed_sub_idx < no_edges_cell; ed_sub_idx++) {
232  const glb_idx_t edge_idx = mesh_->Index(*edges[ed_sub_idx]);
233  glb_idx_t edge_int_dof_offset =
234  edge_idx * num_dofs_[kEdgeOrd] + num_ext_dof_edge;
235  // Copy indices of shape functions from edges to cell
236  // The order, in which they are copied depends on the relative orientation
237  // of the edge w.r.t. the cell
238  switch (edge_orientations[ed_sub_idx]) {
239  case lf::mesh::Orientation::positive: {
240  for (int j = 0; j < no_int_dof_edge; j++) {
241  dofs_[kCellOrd][cell_dof_offset++] =
242  dofs_[kEdgeOrd][edge_int_dof_offset + j];
243  }
244  break;
245  }
246  case lf::mesh::Orientation::negative: {
247  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
248  dofs_[kCellOrd][cell_dof_offset++] =
249  dofs_[kEdgeOrd][edge_int_dof_offset + j];
250  }
251  break;
252  }
253  } // end switch
254  }
255 
256  // Set indices for interior cell degrees of freedom depending on the type of
257  // cell. Here we add new degrees of freedom
258  size_type num_int_dofs_cell;
259  if (cell_p->RefEl() == lf::base::RefEl::kTria()) {
260  num_int_dofs_cell = num_loc_dof_tria_;
261  } else if (cell_p->RefEl() == lf::base::RefEl::kQuad()) {
262  num_int_dofs_cell = num_loc_dof_quad_;
263  } else {
264  LF_ASSERT_MSG(
265  false, "Illegal cell type; only triangles and quads are supported");
266  }
267 
268  // enlist new interior cell-associated dofs
269  for (unsigned j = 0; j < num_int_dofs_cell; j++) {
270  dofs_[kCellOrd][cell_dof_offset++] = dof_idx;
271  dof_entities_.push_back(cell_p);
272  dof_idx++;
273  }
274  }
275  // Finally store total number of shape functions on the mesh.
276  num_dof_ = dof_idx;
277 } // end constructor

279 nonstd::span<const gdof_idx_t> UniformFEDofHandler::GlobalDofIndices(
280  lf::base::RefEl ref_el_type, glb_idx_t entity_index) const {
281  // Co-dimension of entity in a 2D mesh
282  const dim_t codim = 2 - ref_el_type.Dimension();
283  const size_type no_covered_dofs = NumCoveredDofs(ref_el_type);
284 
285  LF_ASSERT_MSG((mesh_->NumEntities(codim) > entity_index),
286  "Index " << entity_index << " out of range");
287  // Pointers to range of dof indices
288  const gdof_idx_t *begin =
289  dofs_[codim].data() +
290  (static_cast<std::size_t>(num_dofs_[codim]) * entity_index);
291  const gdof_idx_t *end = begin + no_covered_dofs;
292  return {begin, end};
293 }
