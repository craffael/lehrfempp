266// Main method for the computation of the element matrix
267template <base::Scalar SCALAR, mesh::utils::MeshFunction DIFF_COEFF,
268 mesh::utils::MeshFunction REACTION_COEFF>
269typename lf::uscalfe::ReactionDiffusionElementMatrixProvider<
270 SCALAR, DIFF_COEFF, REACTION_COEFF>::ElemMat
271ReactionDiffusionElementMatrixProvider<
272 SCALAR, DIFF_COEFF, REACTION_COEFF>::Eval(const lf::mesh::Entity &cell) {
273 // Topological type of the cell
274 const lf::base::RefEl ref_el{cell.RefEl()};
275 // Obtain precomputed information about values of local shape functions
276 // and their gradients at quadrature points.
277 const PrecomputedScalarReferenceFiniteElement<SCALAR> &pfe =
278 fe_precomp_[ref_el.Id()];
279 if (!pfe.isInitialized()) {
280 // Accident: cell is of a type not covered by finite element
281 // specifications or there is no quadrature rule available for this
282 // reference element type
283 std::stringstream temp;
284 temp << "No local shape function information or no quadrature rule for "
285 "reference element type "
286 << ref_el;
287 throw base::LfException(temp.str());
288 }
289
290 // Query the shape of the cell
291 const lf::geometry::Geometry *geo_ptr = cell.Geometry();
292 LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
293 LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
294 "Only 2D implementation available!");
295 SPDLOG_LOGGER_TRACE(ReactionDiffusionElementMatrixProviderLogger(),
296 "{}, shape = \n{}", ref_el,
297 geo_ptr->Global(ref_el.NodeCoords()));
298
299 // Physical dimension of the cell
300 const dim_t world_dim = geo_ptr->DimGlobal();
301 // Gram determinant at quadrature points
302 const Eigen::VectorXd determinants(
303 geo_ptr->IntegrationElement(pfe.Qr().Points()));
304 LF_ASSERT_MSG(
305 determinants.size() == pfe.Qr().NumPoints(),
306 "Mismatch " << determinants.size() << " <-> " << pfe.Qr().NumPoints());
307 // Fetch the transformation matrices for the gradients
308 const Eigen::MatrixXd JinvT(
309 geo_ptr->JacobianInverseGramian(pfe.Qr().Points()));
310 LF_ASSERT_MSG(
311 JinvT.cols() == 2 * pfe.Qr().NumPoints(),
312 "Mismatch " << JinvT.cols() << " <-> " << 2 * pfe.Qr().NumPoints());
313 LF_ASSERT_MSG(JinvT.rows() == world_dim,
314 "Mismatch " << JinvT.rows() << " <-> " << world_dim);
315
316 // compute values of coefficients alpha, gamma at quadrature points:
317 auto alphaval = alpha_(cell, pfe.Qr().Points());
318 auto gammaval = gamma_(cell, pfe.Qr().Points());
319
320 // Element matrix
321 ElemMat mat(pfe.NumRefShapeFunctions(), pfe.NumRefShapeFunctions());
322 mat.setZero();
323
324 // Loop over quadrature points
325 for (base::size_type k = 0; k < pfe.Qr().NumPoints(); ++k) {
326 const double w = pfe.Qr().Weights()[k] * determinants[k];
327 // Transformed gradients
328 const auto trf_grad(
329 JinvT.block(0, 2 * static_cast<Eigen::Index>(k), world_dim, 2) *
330 pfe.PrecompGradientsReferenceShapeFunctions()
331 .block(0, 2 * k, mat.rows(), 2)
332 .transpose());
333 // Transformed gradients multiplied with coefficient
334 const auto alpha_trf_grad(alphaval[k] * trf_grad);
335 mat += w * (trf_grad.adjoint() * alpha_trf_grad +
336 (gammaval[k] * pfe.PrecompReferenceShapeFunctions().col(k)) *
337 (pfe.PrecompReferenceShapeFunctions().col(k).adjoint()));
338 }
339 return mat;
340}

494typename MassEdgeMatrixProvider<SCALAR, COEFF, EDGESELECTOR>::ElemMat
495MassEdgeMatrixProvider<SCALAR, COEFF, EDGESELECTOR>::Eval(
496 const lf::mesh::Entity &edge) {
497 // Topological type of the cell
498 const lf::base::RefEl ref_el{edge.RefEl()};
499 LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
500 "Edge must be of segment type");
501 // Query the shape of the edge
502 const lf::geometry::Geometry *geo_ptr = edge.Geometry();
503 LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
504
505 // Obtain the metric factors for the quadrature points
506 const Eigen::VectorXd determinants(
507 geo_ptr->IntegrationElement(fe_precomp_.Qr().Points()));
508 LF_ASSERT_MSG(determinants.size() == fe_precomp_.Qr().NumPoints(),
509 "Mismatch " << determinants.size() << " <-> "
510 << fe_precomp_.Qr().NumPoints());
511
512 // Element matrix
513 ElemMat mat(fe_precomp_.NumRefShapeFunctions(),
514 fe_precomp_.NumRefShapeFunctions());
515 mat.setZero();
516
517 auto gammaval = gamma_(edge, fe_precomp_.Qr().Points());
518
519 // Loop over quadrature points
520 for (long k = 0; k < determinants.size(); ++k) {
521 // Build local matrix by summing rank-1 contributions
522 // from quadrature points.
523 const auto w =
524 (fe_precomp_.Qr().Weights()[k] * determinants[k]) * gammaval[k];
525 mat += ((fe_precomp_.PrecompReferenceShapeFunctions().col(k)) *
526 (fe_precomp_.PrecompReferenceShapeFunctions().col(k).adjoint())) *
527 w;
528 }

725 determinants.transpose());
726

728 ElemVec vec(pfe.NumRefShapeFunctions());
729 vec.setZero();
730

732

740 pfe.PrecompReferenceShapeFunctions().col(k).conjugate();

893 const lf::geometry::Geometry *geo_ptr = edge.Geometry();
894 LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
895
896 // Quadrature points on physical edge
897 const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(pfe_.Qr().Points()));
898 LF_ASSERT_MSG(
899 mapped_qpts.cols() == pfe_.Qr().NumPoints(),
900 "Mismatch " << mapped_qpts.cols() << " <-> " << pfe_.Qr().NumPoints());
901
902 // Obtain the metric factors for the quadrature points
903 const Eigen::VectorXd determinants(
904 geo_ptr->IntegrationElement(pfe_.Qr().Points()));
905 LF_ASSERT_MSG(
906 determinants.size() == pfe_.Qr().NumPoints(),
907 "Mismatch " << determinants.size() << " <-> " << pfe_.Qr().NumPoints());
908
909 // Element vector
910 ElemVec vec(pfe_.NumRefShapeFunctions());
911 vec.setZero();
912
913 auto g_vals = g_(edge, pfe_.Qr().Points());
914
915 // Loop over quadrature points
916 for (base::size_type k = 0; k < pfe_.Qr().NumPoints(); ++k) {
917 // Add contribution of quadrature point to local vector
918 const auto w = (pfe_.Qr().Weights()[k] * determinants[k]) * g_vals[k];
919 vec += pfe_.PrecompReferenceShapeFunctions().col(k).conjugate() * w;
920 }
921 return vec;
922}
