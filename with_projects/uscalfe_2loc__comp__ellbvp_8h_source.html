266 // Main method for the computation of the element matrix
267 template <typename SCALAR, typename DIFF_COEFF, typename REACTION_COEFF>
268 typename lf::uscalfe::ReactionDiffusionElementMatrixProvider<
269  SCALAR, DIFF_COEFF, REACTION_COEFF>::ElemMat
270 ReactionDiffusionElementMatrixProvider<
271  SCALAR, DIFF_COEFF, REACTION_COEFF>::Eval(const lf::mesh::Entity &cell) {
272  // Topological type of the cell
273  const lf::base::RefEl ref_el{cell.RefEl()};
274  // Obtain precomputed information about values of local shape functions
275  // and their gradients at quadrature points.
276  PrecomputedScalarReferenceFiniteElement<SCALAR> &pfe =
277  fe_precomp_[ref_el.Id()];
278  if (!pfe.isInitialized()) {
279  // Accident: cell is of a type not covered by finite element
280  // specifications or there is no quadrature rule available for this
281  // reference element type
282  std::stringstream temp;
283  temp << "No local shape function information or no quadrature rule for "
284  "reference element type "
285  << ref_el;
286  throw base::LfException(temp.str());
287  }
288 
289  // Query the shape of the cell
290  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
291  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
292  LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
293  "Only 2D implementation available!");
294  SPDLOG_LOGGER_TRACE(ReactionDiffusionElementMatrixProviderLogger(),
295  "{}, shape = \n{}", ref_el,
296  geo_ptr->Global(ref_el.NodeCoords()));
297 
298  // Physical dimension of the cell
299  const dim_t world_dim = geo_ptr->DimGlobal();
300  // Gram determinant at quadrature points
301  const Eigen::VectorXd determinants(
302  geo_ptr->IntegrationElement(pfe.Qr().Points()));
303  LF_ASSERT_MSG(
304  determinants.size() == pfe.Qr().NumPoints(),
305  "Mismatch " << determinants.size() << " <-> " << pfe.Qr().NumPoints());
306  // Fetch the transformation matrices for the gradients
307  const Eigen::MatrixXd JinvT(
308  geo_ptr->JacobianInverseGramian(pfe.Qr().Points()));
309  LF_ASSERT_MSG(
310  JinvT.cols() == 2 * pfe.Qr().NumPoints(),
311  "Mismatch " << JinvT.cols() << " <-> " << 2 * pfe.Qr().NumPoints());
312  LF_ASSERT_MSG(JinvT.rows() == world_dim,
313  "Mismatch " << JinvT.rows() << " <-> " << world_dim);
314 
315  // compute values of coefficients alpha, gamma at quadrature points:
316  auto alphaval = alpha_(cell, pfe.Qr().Points());
317  auto gammaval = gamma_(cell, pfe.Qr().Points());
318 
319  // Element matrix
320  ElemMat mat(pfe.NumRefShapeFunctions(), pfe.NumRefShapeFunctions());
321  mat.setZero();
322 
323  // Loop over quadrature points
324  for (base::size_type k = 0; k < pfe.Qr().NumPoints(); ++k) {
325  const double w = pfe.Qr().Weights()[k] * determinants[k];
326  // Transformed gradients
327  const auto trf_grad(
328  JinvT.block(0, 2 * static_cast<Eigen::Index>(k), world_dim, 2) *
329  pfe.PrecompGradientsReferenceShapeFunctions()
330  .block(0, 2 * k, mat.rows(), 2)
331  .transpose());
332  // Transformed gradients multiplied with coefficient
333  const auto alpha_trf_grad(alphaval[k] * trf_grad);
334  mat += w * (trf_grad.adjoint() * alpha_trf_grad +
335  (gammaval[k] * pfe.PrecompReferenceShapeFunctions().col(k)) *
336  (pfe.PrecompReferenceShapeFunctions().col(k).adjoint()));
337  }
338  return mat;
339 }

492  // Topological type of the cell
493  const lf::base::RefEl ref_el{edge.RefEl()};
494  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
495  "Edge must be of segment type");
496  // Query the shape of the edge
497  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
498  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
499 
500  // Obtain the metric factors for the quadrature points
501  const Eigen::VectorXd determinants(
502  geo_ptr->IntegrationElement(fe_precomp_.Qr().Points()));
503  LF_ASSERT_MSG(determinants.size() == fe_precomp_.Qr().NumPoints(),
504  "Mismatch " << determinants.size() << " <-> "
505  << fe_precomp_.Qr().NumPoints());
506 
507  // Element matrix
508  ElemMat mat(fe_precomp_.NumRefShapeFunctions(),
509  fe_precomp_.NumRefShapeFunctions());
510  mat.setZero();
511 
512  auto gammaval = gamma_(edge, fe_precomp_.Qr().Points());
513 
514  // Loop over quadrature points
515  for (long k = 0; k < determinants.size(); ++k) {
516  // Build local matrix by summing rank-1 contributions
517  // from quadrature points.
518  const auto w =
519  (fe_precomp_.Qr().Weights()[k] * determinants[k]) * gammaval[k];
520  mat += ((fe_precomp_.PrecompReferenceShapeFunctions().col(k)) *
521  (fe_precomp_.PrecompReferenceShapeFunctions().col(k).adjoint())) *
522  w;
523  }

881 template <class SCALAR, class FUNCTOR, class EDGESELECTOR>
882 typename ScalarLoadEdgeVectorProvider<SCALAR, FUNCTOR, EDGESELECTOR>::ElemVec
883 ScalarLoadEdgeVectorProvider<SCALAR, FUNCTOR, EDGESELECTOR>::Eval(
884  const lf::mesh::Entity &edge) {
885  // Topological type of the cell
886  const lf::base::RefEl ref_el{edge.RefEl()};
887  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
888  "Edge must be of segment type");
889  // Query the shape of the edge
890  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
891  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
892 
893  // Quadrature points on physical edge
894  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(pfe_.Qr().Points()));
895  LF_ASSERT_MSG(
896  mapped_qpts.cols() == pfe_.Qr().NumPoints(),
897  "Mismatch " << mapped_qpts.cols() << " <-> " << pfe_.Qr().NumPoints());
898 
899  // Obtain the metric factors for the quadrature points
900  const Eigen::VectorXd determinants(
901  geo_ptr->IntegrationElement(pfe_.Qr().Points()));
902  LF_ASSERT_MSG(
903  determinants.size() == pfe_.Qr().NumPoints(),
904  "Mismatch " << determinants.size() << " <-> " << pfe_.Qr().NumPoints());
905 
906  // Element vector
907  ElemVec vec(pfe_.NumRefShapeFunctions());
908  vec.setZero();
909 
910  auto g_vals = g_(edge, pfe_.Qr().Points());
911 
912  // Loop over quadrature points
913  for (base::size_type k = 0; k < pfe_.Qr().NumPoints(); ++k) {
914  // Add contribution of quadrature point to local vector
915  const auto w = (pfe_.Qr().Weights()[k] * determinants[k]) * g_vals[k];
916  vec += pfe_.PrecompReferenceShapeFunctions().col(k).conjugate() * w;
917  }
918  return vec;
919 }
