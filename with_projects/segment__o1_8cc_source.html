53  // Return variable

56  // Obtain geometry of children as vector of pairs of lattice coordinates
57  std::vector<Eigen::Matrix<int, Eigen::Dynamic, Eigen::Dynamic>>

61  LF_VERIFY_MSG(

63  "no_children = " << no_children << " <-> " << ref_pat.NumChildren(codim));
64  // For each child create a geometry object and a unique pointer to it.
65  for (int l = 0; l < no_children; l++) {
66  // codim == 0:A single child must be described by an interval, that is
67  // two different lattice coordinates
68  // codim == 1: a point's location is just one number
69  LF_VERIFY_MSG(child_polygons[l].rows() == 1,
70  "child_polygons[l].rows() = " << child_polygons[l].rows());
71  LF_VERIFY_MSG(child_polygons[l].cols() == (2 - codim),
72  "child_polygons[l].cols() = " << child_polygons[l].cols());
73  // Normalize lattice coordinates

75  Global(h_lattice * child_polygons[l].cast<double>()));
76 
77  switch (codim) {

81  }
