520  template <typename LOCALDOFINFO>
521  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
522  LOCALDOFINFO &&locdof)
523  : mesh_p_(std::move(mesh_p)) {
524  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
525 
526  // Index counter for global shape functions = global dof
527  gdof_idx_t dof_idx = 0;
528 
529  // Step I: Set indices for shape functions on nodes
530  // Run through node indices (entities of co-dimension 2)
531  const size_type no_nodes = mesh_p_->NumEntities(2);
532  num_int_dofs_[2].resize(no_nodes, 0);
533  offsets_[2].resize(no_nodes + 1, 0);
534  // Traverse nodes (co-dimension-2 entities) based on indices
535  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
536  // Obtain pointer to node entity
537  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
538  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
539  // Offset for indices of node in index vector
540  glb_idx_t node_dof_offset = dof_idx;
541  offsets_[2][node_idx] = node_dof_offset;
542  // Request number of local shape functions associated with the node
543  size_type no_int_dof_node = locdof(*node_p);
544  num_int_dofs_[2][node_idx] = no_int_dof_node;
545 
546  // Store dof indices in array
547  for (unsigned j = 0; j < no_int_dof_node; j++) {
548  dofs_[2].push_back(dof_idx);
549  dof_entities_.push_back(node_p); // Store entity for current dof
550  dof_idx++; // Move on to next index
551  }
552  }
553  // Set sentinel
554  offsets_[2][no_nodes] = dof_idx;
555 
556  // Step II: Set indices for shape functions on edges (co-dimension = 1)
557  const size_type no_edges = mesh_p_->NumEntities(1);
558  // Set length of edge-related index vectors
559  num_int_dofs_[1].resize(no_edges, 0);
560  offsets_[1].resize(no_edges + 1, 0);
561  // points to the position of the current dof index
562  size_type edge_dof_offset = 0;
563  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
564  // Obtain pointer to edge entity
565  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
566  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
567  // Offset for indices of edge dof in index vector
568  offsets_[1][edge_idx] = edge_dof_offset;
569  size_type no_int_dof_edge = locdof(*edge);
570  num_int_dofs_[1][edge_idx] = no_int_dof_edge;
571 
572  // Obtain indices for basis functions sitting at endpoints
573  // Endpoints are mesh entities with co-dimension = 2
574  for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
575  const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
576  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
577  size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
578  // Copy indices of shape functions from nodes to edge
579  for (unsigned j = 0; j < no_int_dofs_ep; j++) {
580  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
581  edge_dof_offset++;
582  }
583  }
584  // Set indices for interior edge degrees of freedom
585  for (unsigned j = 0; j < no_int_dof_edge; j++) {
586  dofs_[1].push_back(dof_idx);
587  edge_dof_offset++;
588  dof_entities_.push_back(edge);
589  dof_idx++;
590  }
591  }
592  // Set sentinel
593  offsets_[1][no_edges] = edge_dof_offset;
594 
595  // Step III: Set indices for shape functions on cells (co-dimension = 0)
596  const size_type no_cells = mesh_p_->NumEntities(0);
597  // Set length of cell-related index vectors
598  num_int_dofs_[0].resize(no_cells, 0);
599  offsets_[0].resize(no_cells + 1, 0);
600  // points to the position of the current dof index
601  size_type cell_dof_offset = 0;
602  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
603  // Obtain pointer to current ell
604  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
605  // Offset for indices of cell dof in index vector
606  offsets_[0][cell_idx] = cell_dof_offset;
607  size_type no_int_dof_cell = locdof(*cell);
608  num_int_dofs_[0][cell_idx] = no_int_dof_cell;
609 
610  // Obtain indices for basis functions in vertices
611  for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
612  const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
613  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
614  size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
615  // Copy indices of shape functions from nodes to cell
616  for (unsigned j = 0; j < no_int_dofs_vt; j++) {
617  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
618  cell_dof_offset++;
619  }
620  }
621 
622  // Collect indices of interior shape functions of edges
623  // Internal ordering will depend on the orientation of the edge
624  auto edge_orientations = cell->RelativeOrientations();
625  auto edges = cell->SubEntities(1);
626  // Loop over edges
627  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
628  ed_sub_idx++) {
629  const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
630  const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
631  const glb_idx_t edge_int_dof_offset =
632  offsets_[1][edge_idx + 1] - no_int_dof_edge;
633 
634  // Copy indices of shape functions from edges to cell
635  // The order, in which they are copied depends on the relative
636  // orientation of the edge w.r.t. the cell
637  switch (edge_orientations[ed_sub_idx]) {
638  case lf::mesh::Orientation::positive: {
639  for (int j = 0; j < no_int_dof_edge; j++) {
640  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
641  cell_dof_offset++;
642  }
643  break;
644  }
645  case lf::mesh::Orientation::negative: {
646  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
647  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
648  cell_dof_offset++;
649  }
650  break;
651  }
652  } // end switch
653  } // end loop over edges
654 
655  // Set indices for interior shape functions of the cell
656  for (unsigned j = 0; j < no_int_dof_cell; j++) {
657  dofs_[0].push_back(dof_idx);
658  cell_dof_offset++;
659  dof_entities_.push_back(cell);
660  dof_idx++;
661  } // end loop over interior dof of cell
662  } // end loop over cells
663  // Set sentinel
664  offsets_[0][no_cells] = cell_dof_offset;
665 
666  // Finally set number of global shape functions
667  num_dof_ = dof_idx;
668  } // end constructor
