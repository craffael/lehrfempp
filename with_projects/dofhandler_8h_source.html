522  template <typename LOCALDOFINFO>
523  // NOLINTNEXTLINE(readability-function-cognitive-complexity)
524  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
525  LOCALDOFINFO &&locdof)
526  : mesh_p_(std::move(mesh_p)) {
527  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
528 
529  // Index counter for global shape functions = global dof
530  gdof_idx_t dof_idx = 0;
531 
532  // Step I: Set indices for shape functions on nodes
533  // Run through node indices (entities of co-dimension 2)
534  const size_type no_nodes = mesh_p_->NumEntities(2);
535  num_int_dofs_[2].resize(no_nodes, 0);
536  offsets_[2].resize(no_nodes + 1, 0);
537  // Traverse nodes (co-dimension-2 entities) based on indices
538  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
539  // Obtain pointer to node entity
540  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
541  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
542  // Offset for indices of node in index vector
543  glb_idx_t node_dof_offset = dof_idx;
544  offsets_[2][node_idx] = node_dof_offset;
545  // Request number of local shape functions associated with the node
546  size_type no_int_dof_node = locdof(*node_p);
547  num_int_dofs_[2][node_idx] = no_int_dof_node;
548 
549  // Store dof indices in array
550  for (unsigned j = 0; j < no_int_dof_node; j++) {
551  dofs_[2].push_back(dof_idx);
552  dof_entities_.push_back(node_p); // Store entity for current dof
553  dof_idx++; // Move on to next index
554  }
555  }
556  // Set sentinel
557  offsets_[2][no_nodes] = dof_idx;
558 
559  // Step II: Set indices for shape functions on edges (co-dimension = 1)
560  const size_type no_edges = mesh_p_->NumEntities(1);
561  // Set length of edge-related index vectors
562  num_int_dofs_[1].resize(no_edges, 0);
563  offsets_[1].resize(no_edges + 1, 0);
564  // points to the position of the current dof index
565  size_type edge_dof_offset = 0;
566  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
567  // Obtain pointer to edge entity
568  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
569  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
570  // Offset for indices of edge dof in index vector
571  offsets_[1][edge_idx] = edge_dof_offset;
572  size_type no_int_dof_edge = locdof(*edge);
573  num_int_dofs_[1][edge_idx] = no_int_dof_edge;
574 
575  // Obtain indices for basis functions sitting at endpoints
576  // Endpoints are mesh entities with co-dimension = 2
577  for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
578  const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
579  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
580  size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
581  // Copy indices of shape functions from nodes to edge
582  for (unsigned j = 0; j < no_int_dofs_ep; j++) {
583  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
584  edge_dof_offset++;
585  }
586  }
587  // Set indices for interior edge degrees of freedom
588  for (unsigned j = 0; j < no_int_dof_edge; j++) {
589  dofs_[1].push_back(dof_idx);
590  edge_dof_offset++;
591  dof_entities_.push_back(edge);
592  dof_idx++;
593  }
594  }
595  // Set sentinel
596  offsets_[1][no_edges] = edge_dof_offset;
597 
598  // Step III: Set indices for shape functions on cells (co-dimension = 0)
599  const size_type no_cells = mesh_p_->NumEntities(0);
600  // Set length of cell-related index vectors
601  num_int_dofs_[0].resize(no_cells, 0);
602  offsets_[0].resize(no_cells + 1, 0);
603  // points to the position of the current dof index
604  size_type cell_dof_offset = 0;
605  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
606  // Obtain pointer to current ell
607  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
608  // Offset for indices of cell dof in index vector
609  offsets_[0][cell_idx] = cell_dof_offset;
610  size_type no_int_dof_cell = locdof(*cell);
611  num_int_dofs_[0][cell_idx] = no_int_dof_cell;
612 
613  // Obtain indices for basis functions in vertices
614  for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
615  const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
616  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
617  size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
618  // Copy indices of shape functions from nodes to cell
619  for (unsigned j = 0; j < no_int_dofs_vt; j++) {
620  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
621  cell_dof_offset++;
622  }
623  }
624 
625  // Collect indices of interior shape functions of edges
626  // Internal ordering will depend on the orientation of the edge
627  auto edge_orientations = cell->RelativeOrientations();
628  auto edges = cell->SubEntities(1);
629  // Loop over edges
630  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
631  ed_sub_idx++) {
632  const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
633  const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
634  const glb_idx_t edge_int_dof_offset =
635  offsets_[1][edge_idx + 1] - no_int_dof_edge;
636 
637  // Copy indices of shape functions from edges to cell
638  // The order, in which they are copied depends on the relative
639  // orientation of the edge w.r.t. the cell
640  switch (edge_orientations[ed_sub_idx]) {
641  case lf::mesh::Orientation::positive: {
642  for (int j = 0; j < no_int_dof_edge; j++) {
643  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
644  cell_dof_offset++;
645  }
646  break;
647  }
648  case lf::mesh::Orientation::negative: {
649  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
650  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
651  cell_dof_offset++;
652  }
653  break;
654  }
655  } // end switch
656  } // end loop over edges
657 
658  // Set indices for interior shape functions of the cell
659  for (unsigned j = 0; j < no_int_dof_cell; j++) {
660  dofs_[0].push_back(dof_idx);
661  cell_dof_offset++;
662  dof_entities_.push_back(cell);
663  dof_idx++;
664  } // end loop over interior dof of cell
665  } // end loop over cells
666  // Set sentinel
667  offsets_[0][no_cells] = cell_dof_offset;
668 
669  // Finally set number of global shape functions
670  num_dof_ = dof_idx;
671  } // end constructor
