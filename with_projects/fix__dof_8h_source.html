88 void FixFlaggedSolutionComponents(SELECTOR &&selectvals, COOMatrix<SCALAR> &A,
89  RHSVECTOR &b) {
90  const lf::assemble::size_type N(A.cols());
91  LF_ASSERT_MSG(A.rows() == N, "Matrix must be square!");
92  LF_ASSERT_MSG(N == b.size(),
93  "Mismatch N = " << N << " <-> b.size() = " << b.size());
94  {
95  // Multiply sparse matrix with the vector of fixed components and subtract
96  // result from right hand side
97  // To skip irrelevant components of fixed_vec rely on a temporary vector
98  Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> tmp_vec(N);
99  for (lf::assemble::gdof_idx_t k = 0; k < N; ++k) {
100  const auto selval{selectvals(k)};
101  if (selval.first) {
102  tmp_vec[k] = selval.second;
103  } else {
104  tmp_vec[k] = SCALAR();
105  }
106  }
107  A.MatVecMult(-1.0, tmp_vec, b);
108  }
109  // Set vector components of right-hand-side vector for prescribed values
110  for (lf::assemble::gdof_idx_t k = 0; k < N; ++k) {
111  const auto selval{selectvals(k)};
112  if (selval.first) {
113  b[k] = selval.second;
114  }
115  }
116  // Set rows and columns of the sparse matrix corresponding to the fixed
117  // solution components to zero
118  A.setZero([&selectvals](gdof_idx_t i, gdof_idx_t j) {
119  return (selectvals(i).first || selectvals(j).first);
120  });
121  // Old implementation, demonstrating what is going on
122  // lf::assemble::COOMatrix<double>::TripletVec::iterator new_last =
123  // std::remove_if(
124  // A.triplets().begin(), A.triplets().end(),
125  // [&fixed_comp_flags](
126  // typename lf::assemble::COOMatrix<double>::Triplet &triplet) {
127  // return (fixed_comp_flags[triplet.row()] ||
128  // fixed_comp_flags[triplet.col()]);
129  // });
130  // // Adjust size of triplet vector
131  // A.triplets().erase(new_last, A.triplets().end());
132  // Add Unit diagonal entries corrresponding to fixed components
133  for (lf::assemble::gdof_idx_t dofnum = 0; dofnum < N; ++dofnum) {
134  const auto selval{selectvals(dofnum)};
135  if (selval.first) {
136  A.AddToEntry(dofnum, dofnum, 1.0);
137  }
138  }
139 }

256  // Set up a vector containing the negative fixed solution components and a
257  // flag vector, whose entry is `true` if the corresponding

259  Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> fixed_vec(b.size());
260  fixed_vec.setZero();
261  std::vector<bool> fixed_comp_flags(N, false);
262  for (typename fixed_components_t<SCALAR>::const_reference idx_val_pair :
263  fixed_components) {
264  LF_ASSERT_MSG(idx_val_pair.first < N,
265  "Index " << idx_val_pair.first << " >= N = " << N);
266  fixed_vec[idx_val_pair.first] += idx_val_pair.second;
267  fixed_comp_flags[idx_val_pair.first] = true;
268  }
269  // FixFlaggedSolutionComponents<SCALAR>(fixed_comp_flags, fixed_vec, A, b);
270  FixFlaggedSolutionCompAlt<SCALAR>(
271  [&fixed_comp_flags,
272  &fixed_vec](lf::assemble::gdof_idx_t i) -> std::pair<bool, double> {
273  LF_ASSERT_MSG((i < fixed_comp_flags.size()) && (i < fixed_vec.size()),
274  "Illegal index " << i);
275  return std::make_pair(fixed_comp_flags[i], fixed_vec[i]);
