153 template <typename CONVECTION_COEFF>
154 lf::mesh::utils::CodimMeshDataSet<bool> flagEntitiesOnInflowBoundary(
155  const std::shared_ptr<const lf::mesh::Mesh>& mesh_p,
156  CONVECTION_COEFF beta) {
157  // flag edges on the boundary
158  auto bd_flags = lf::mesh::utils::flagEntitiesOnBoundary(mesh_p, 1);
159 
160  // initialize inflow boundary flags to fals
161  lf::mesh::utils::CodimMeshDataSet<bool> inflow_boundary{mesh_p, 1, false};
162 
163  // iterate over all cells:
164  for (const auto* const cell : mesh_p->Entities(0)) {
165  // query geometry
166  auto geo_ptr = cell->Geometry();
167  auto sub_entities = cell->SubEntities(1);
168 
169  // calcualte normals
170  Eigen::MatrixXd normals = OuterNormals(*geo_ptr);
171 
172  // iterate over all edges of the cell
173  for (int i = 0; i < cell->RefEl().NumSubEntities(1); ++i) {
174  const auto* const edge = sub_entities[i];
175 
176  // check the inflow condition at the center of the edge
177  bool inflow = normals.col(i).transpose() *
178  beta(*(sub_entities[i]),
179  (Eigen::MatrixXd(1, 1) << 0.5).finished())[0] <
180  0.0;
181  inflow_boundary(*edge) = bd_flags(*edge) && inflow;
182  }
183  }
184  return inflow_boundary;
185 }

202 template <typename CONVECTION_COEFF>
203 lf::mesh::utils::CodimMeshDataSet<bool> flagEntitiesOnOutflowBoundary(
204  const std::shared_ptr<const lf::mesh::Mesh>& mesh_p,
205  CONVECTION_COEFF beta) {
206  // flag all entities on the boundary:
207  auto bd_flags = lf::mesh::utils::flagEntitiesOnBoundary(mesh_p, 1);
208 
209  // flag all entities on the inflow boundary:
210  auto inflow_boundary = flagEntitiesOnInflowBoundary(mesh_p, beta);
211 
212  // initialize outflow flags:
213  lf::mesh::utils::CodimMeshDataSet<bool> outflow_boundary{mesh_p, 1, false};
214 
215  // outflow boundary dofs are those, that lie on the boundary, but not on the
216  // inflow boundary This guarantues a partition of the boundary edges into
217  // inflow and outflow edges.
218  for (const auto* const edge : mesh_p->Entities(1)) {
219  outflow_boundary(*edge) = bd_flags(*edge) && (!inflow_boundary(*edge));
220  }
221  return outflow_boundary;
222 }

275  EDGESELECTOR_DIRICHLET&& dirichletcondflag,

277  size_type trace_component, size_type flux_component) {

279  LF_ASSERT_MSG(trace_component != flux_component,
280  "Boundary conditions imposed on same component");
281 
282  // retrive the dofhandlers for the corresponding trace and flux components
283  const auto& trace_dofh = dofh.ComponentDofHandler(trace_component);
284  const auto& flux_dofh = dofh.ComponentDofHandler(flux_component);
285 
286  // flag dofs on the component spaces
287  std::vector<std::pair<bool, SCALAR>> componentTraceConditions =
288  lf::uscalfe::InitEssentialConditionFromFunction(
289  trace_dofh, fe_spec_edge_trace, dirichletcondflag, g);
290  std::vector<std::pair<bool, SCALAR>> componentFluxConditions =
291  lf::uscalfe::InitEssentialConditionFromFunction(
292  flux_dofh, fe_spec_edge_flux, neumanncondflag, h);
293 
294  // calculate offsets to transform flags on component spaces
295  // to the full product space
296  size_type trace_offset = dofh.Offset(trace_component);
297  size_type flux_offset = dofh.Offset(flux_component);
298 
299  std::vector<std::pair<bool, SCALAR>> EssentialConditions(dofh.NumDofs(),
300  {false, SCALAR{}});
301 
302  // flag dofs on product space corresponding to fixed trace dofs
303  for (size_type dof = 0; dof < trace_dofh.NumDofs(); dof++) {
304  EssentialConditions[trace_offset + dof] = componentTraceConditions[dof];
305  }
306 
307  // flag dofs on product space corresponding to fixed flux dofs
308  for (size_type dof = 0; dof < flux_dofh.NumDofs(); dof++) {
309  EssentialConditions[flux_offset + dof] = componentFluxConditions[dof];
310  }
311 
312  return EssentialConditions;

329 template <typename SCALAR>
330 lf::mesh::utils::CodimMeshDataSet<SCALAR> ElementErrorEstimators(
331  const ProductUniformFEDofHandler& trial_dofh,
332  // const ProductUniformFEDofHandler& test_dofh,
333  std::shared_ptr<ProductElementMatrixProvider<SCALAR>> stiffness_provider,
334  std::shared_ptr<ProductElementMatrixProvider<SCALAR>> gramian_provider,
335  std::shared_ptr<ProductElementVectorProvider<SCALAR>> load_provider,
336  const Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>& sol_vec) {
337  // retrive the underlying mesh:
338  auto mesh_p = trial_dofh.Mesh();
339 
340  // retrive the number of cells, initialize the return vector
341  lf::mesh::utils::CodimMeshDataSet<SCALAR> element_errors{mesh_p, 0, 0.0};
342 
343  // compute the element wise errors
344  for (const auto* const cell : mesh_p->Entities(0)) {
345  // evaluate the providers on the current cell
346  auto G = gramian_provider->Eval(*cell);
347  auto B = stiffness_provider->Eval(*cell);
348  auto l = load_provider->Eval(*cell);
349 
350  // retrive local solution
351  Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> local_solution(
352  trial_dofh.NumLocalDofs(*cell));
353  auto global_dofs = trial_dofh.GlobalDofIndices(*cell);
354  for (size_type i = 0; i < trial_dofh.NumLocalDofs(*cell); ++i) {
355  local_solution(i) = sol_vec(global_dofs[i]);
356  }
357 
358  // evaluate posterior error on current element:
359  Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> r = l - B * local_solution;
360 
361  SCALAR e = std::sqrt(r.transpose() * G.ldlt().solve(r));
362  element_errors(*cell) = e;
363  }
364  return element_errors;
365 }
