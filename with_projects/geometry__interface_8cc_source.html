11 double Volume(const Geometry& geo) {
12  const lf::base::dim_t refdim = geo.DimLocal();
13 
14  Eigen::Matrix<double, Eigen::Dynamic, 1> refc(refdim, 1);
15  double refvol = 1.0;
16  switch (geo.RefEl()) {
17  case lf::base::RefEl::kPoint(): {
18  return 1.0;
19  }
20  case lf::base::RefEl::kSegment(): {
21  refc << 0.5;
22  break;
23  }
24  case lf::base::RefEl::kTria(): {
25  refvol = 0.5;
26  refc << 1.0 / 3, 1.0 / 3;
27  break;
28  }
29  case lf::base::RefEl::kQuad(): {
30  refc << 0.5, 0.5;
31  break;
32  }
33  default: {
34  LF_VERIFY_MSG(false,
35  "Volume not available for " << geo.RefEl().ToString());
36  }
37  } // end switch RefEl
38  return (refvol * ((geo.IntegrationElement(refc))[0]));
