62
63 const Eigen::VectorXd ones = Eigen::VectorXd::Ones(N_dofs);
64
65 /* Time Steps */
66 // double tau = T_ / m_;
67
68 /* Strang Splitting Method
69 * First half time step [0, tau/2]: Diffusion
70 */
71 bool firstcall = true;
72 sol_cur = StrangSplit::diffusionEvolutionOperator(firstcall, mu);
73 firstcall = false;
74 /* loop through time steps */
75 for (unsigned int i = 1; i < m_ - 1; i++) {
76 /* Reaction for next full time step */
77 sol_next = cap.cwiseQuotient(ones + (cap.cwiseQuotient(sol_cur) - ones) *
78 std::exp(-lambda_ * tau_));
79 sol_cur = sol_next;
80 /* Diffusion for next full time step */
81 sol_next = StrangSplit::diffusionEvolutionOperator(firstcall, sol_cur);
82 sol_cur = sol_next;
83 }
84
85 /* Last half time step: Reaction */
86 sol_next = cap.cwiseQuotient(ones + (cap.cwiseQuotient(sol_cur) - ones) *
87 std::exp(-lambda_ * tau_ / 2.));
88 sol_cur = sol_next;
89
90 sol = sol_cur;
91 return sol;
92}
