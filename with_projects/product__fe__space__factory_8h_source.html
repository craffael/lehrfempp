158 std::shared_ptr<const lf::uscalfe::ScalarReferenceFiniteElement<SCALAR>>
159 ProductUniformFESpaceFactory<SCALAR>::ReferenceFiniteElement(
160  lf::base::RefEl ref_el, size_type degree) {
161  switch (ref_el) {
162  case (lf::base::RefEl::kTria()): {
163  switch (degree) {
164  case (0):
165  return std::make_shared<FeDiscontinuousO0Tria<SCALAR>>();
166  case (1):
167  return std::make_shared<lf::uscalfe::FeLagrangeO1Tria<SCALAR>>();
168  case (2):
169  return std::make_shared<lf::uscalfe::FeLagrangeO2Tria<SCALAR>>();
170  case (3):
171  return std::make_shared<lf::uscalfe::FeLagrangeO3Tria<SCALAR>>();
172  default:
173  LF_ASSERT_MSG(false, "no fe of degree " << degree
174  << " available on ref_el "
175  << ref_el);
176  }
177  }
178  case (lf::base::RefEl::kQuad()): {
179  switch (degree) {
180  case (0):
181  return std::make_shared<FeDiscontinuousO0Quad<SCALAR>>();
182  case (1):
183  return std::make_shared<lf::uscalfe::FeLagrangeO1Quad<SCALAR>>();
184  case (2):
185  return std::make_shared<lf::uscalfe::FeLagrangeO2Quad<SCALAR>>();
186  case (3):
187  return std::make_shared<lf::uscalfe::FeLagrangeO3Quad<SCALAR>>();
188  default:
189  LF_ASSERT_MSG(false, "no fe of degree " << degree
190  << "available on ref_el "
191  << ref_el);
192  }
193  }
194  case (lf::base::RefEl::kSegment()): {
195  switch (degree) {
196  case (0):
197  return std::make_shared<FeDiscontinuousO0Segment<SCALAR>>();
198  case (1):
199  return std::make_shared<lf::uscalfe::FeLagrangeO1Segment<SCALAR>>();
200  case (2):
201  return std::make_shared<lf::uscalfe::FeLagrangeO2Segment<SCALAR>>();
202  case (3):
203  return std::make_shared<lf::uscalfe::FeLagrangeO3Segment<SCALAR>>();
204  default:
205  LF_ASSERT_MSG(false, "no fe of degree " << degree
206  << "available on ref_el "
207  << ref_el);
208  }
209  }
210  case (lf::base::RefEl::kPoint()): {
211  return std::make_shared<lf::uscalfe::FeLagrangePoint<SCALAR>>(degree);
212  }
213  default: {

220 size_type ProductUniformFESpaceFactory<SCALAR>::AddH1Component(
221  size_type degree) {
222  LF_ASSERT_MSG(
223  degree > 0,
224  "Can only represent H1 conforming finite element with degree > 0");
225  // add continuous reference shape functions on triangles, quads and segments.
226  rfs_tria_v_.push_back(
227  ReferenceFiniteElement(lf::base::RefEl::kTria(), degree));
228  rfs_quad_v_.push_back(
229  ReferenceFiniteElement(lf::base::RefEl::kQuad(), degree));
230  rfs_segment_v_.push_back(
231  ReferenceFiniteElement(lf::base::RefEl::kSegment(), degree));
232  rfs_point_v_.push_back(nullptr);
233  return num_components_++;
234 }
