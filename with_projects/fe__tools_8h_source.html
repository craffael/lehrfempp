34 template <class MF, class QR_SELECTOR>
35 auto LocalIntegral(const mesh::Entity &e, const QR_SELECTOR &qr_selector,
36  const MF &mf) -> mesh::utils::MeshFunctionReturnType<MF> {
37  using MfType = mesh::utils::MeshFunctionReturnType<MF>;
38  auto qr = qr_selector(e);
39  auto values = mf(e, qr.Points());
40  auto weights_ie =
41  (qr.Weights().cwiseProduct(e.Geometry()->IntegrationElement(qr.Points())))
42  .eval();
43  LF_ASSERT_MSG(values.size() == qr.NumPoints(),
44  "mf returns vector with wrong size.");
45  if constexpr (std::is_arithmetic_v<MfType>) { // NOLINT
46  auto value_m = Eigen::Map<Eigen::Matrix<MfType, 1, Eigen::Dynamic>>(
47  &values[0], 1, values.size());
48  return (value_m * weights_ie)(0);
49  }
50 
51  if constexpr (base::is_eigen_matrix<MfType>) { // NOLINT
52  constexpr int size = MfType::SizeAtCompileTime;
53  if constexpr (size != Eigen::Dynamic) {
54  auto value_m = Eigen::Map<
55  Eigen::Matrix<typename MfType::Scalar, size, Eigen::Dynamic>>(
56  &values[0](0, 0), size, values.size());
57  MfType result;
58  auto result_m =
59  Eigen::Map<Eigen::Matrix<typename MfType::Scalar, size, 1>>(
60  &result(0, 0));
61  result_m = value_m * weights_ie;
62  return result;
63  }
64  }
65  // fallback: we cannot make any optimizations:
66  MfType temp = weights_ie(0) * values[0];
67  for (Eigen::Index i = 1; i < qr.NumPoints(); ++i) {
68  temp = temp + weights_ie(i) * values[i];
69  }
70  return temp;
71 }
72 }; // namespace internal

106 template <class MF, class QR_SELECTOR,
107  class ENTITY_PREDICATE = base::PredicateTrue>
108 auto IntegrateMeshFunction(const lf::mesh::Mesh &mesh, const MF &mf,
109  const QR_SELECTOR &qr_selector,
110  const ENTITY_PREDICATE &ep = base::PredicateTrue{},
111  int codim = 0)
112  -> mesh::utils::MeshFunctionReturnType<MF> {
113  static_assert(mesh::utils::isMeshFunction<MF>);
114  using MfType = mesh::utils::MeshFunctionReturnType<MF>;
115 
116  auto entities = mesh.Entities(codim);
117  auto result = internal::LocalIntegral(**entities.begin(), qr_selector, mf);
118  for (auto i = entities.begin() + 1; i != entities.end(); ++i) {
119  if (!ep(**i)) {
120  continue;
121  }
122  result = result + internal::LocalIntegral(**i, qr_selector, mf);
123  }
124  return result;
125 }

195 template <typename SCALAR, typename MF, typename SELECTOR = base::PredicateTrue>
196 auto NodalProjection(const UniformScalarFESpace<SCALAR> &fe_space, MF &&u,
197  SELECTOR &&pred = base::PredicateTrue{}) {
198  static_assert(mesh::utils::isMeshFunction<std::remove_reference_t<MF>>);
199  // choose scalar type so it can hold the scalar type of u as well as
200  // SCALAR
201  using scalarMF_t =
202  mesh::utils::MeshFunctionReturnType<std::remove_reference_t<MF>>;
203  using scalar_t = decltype(SCALAR(0) * scalarMF_t(0));
204  // Return type, type for FE coefficient vector
205  using dof_vec_t = Eigen::Matrix<scalar_t, Eigen::Dynamic, 1>;
206 
207  // Underlying mesh instance
208  const lf::mesh::Mesh &mesh{*fe_space.Mesh()};
209  // Fetch local-to-global index mapping for shape functions
210  const lf::assemble::DofHandler &dofh{fe_space.LocGlobMap()};
211 
212  // Vector for returning expansion coefficients
213  dof_vec_t glob_dofvec(dofh.NumDofs());
214  glob_dofvec.setZero();
215 
216  // Loop over all cells
217  for (const lf::mesh::Entity *cell : mesh.Entities(0)) {
218  if (!pred(*cell)) {
219  continue;
220  }
221  // Topological type of the cell
222  const lf::base::RefEl ref_el{cell->RefEl()};
223 
224  // Information about local shape functions on reference element
225  auto ref_shape_fns = fe_space.ShapeFunctionLayout(ref_el);
226  LF_ASSERT_MSG(ref_shape_fns, "reference shape function for "
227  << ref_el << " not available.");
228  // Obtain reference coordinates for evaluation nodes
229  const Eigen::MatrixXd ref_nodes(ref_shape_fns->EvaluationNodes());
230 
231  // Collect values of function to be projected in a row vector
232  auto uvalvec = u(*cell, ref_nodes);
233 
234  // Compute the resulting local degrees of freedom
235  auto dofvec(ref_shape_fns->NodalValuesToDofs(
236  Eigen::Map<Eigen::Matrix<scalar_t, 1, Eigen::Dynamic>>(
237  &uvalvec[0], 1, uvalvec.size())));
238 
239  // Set the corresponing global degrees of freedom
240  // Note: "Setting", not "adding to"
241  // Number of local shape functions
242  const size_type num_loc_dofs = dofh.NumLocalDofs(*cell);
243  LF_ASSERT_MSG(
244  dofvec.size() == num_loc_dofs,
245  "Size mismatch: " << dofvec.size() << " <-> " << num_loc_dofs);
246  // Fetch global numbers of local shape functions
247  nonstd::span<const lf::assemble::gdof_idx_t> ldof_gidx(
248  dofh.GlobalDofIndices(*cell));
249  // Insert dof values into the global coefficient vector
250  for (int j = 0; j < num_loc_dofs; ++j) {
251  glob_dofvec[ldof_gidx[j]] = dofvec[j];
252  }
253  }
254  return glob_dofvec;
255 }

297 template <typename SCALAR, typename EDGESELECTOR, typename FUNCTION>
298 std::vector<std::pair<bool, SCALAR>> InitEssentialConditionFromFunction(
299  const lf::assemble::DofHandler &dofh,
300  const ScalarReferenceFiniteElement<SCALAR> &fe_spec_edge,
301  EDGESELECTOR &&esscondflag, FUNCTION &&g) {
302  static_assert(mesh::utils::isMeshFunction<std::remove_reference_t<FUNCTION>>);
303  // static_assert(isMeshFunction<FUNCTION>, "g must by a MeshFunction object");
304  LF_ASSERT_MSG(fe_spec_edge.RefEl() == lf::base::RefEl::kSegment(),
305  "finite element specification must be for an edge!");
306 
307  // *** I: Preprocessing ***
308  // Fetch numbers of evaluation nodes and local shape functions
309  const size_type num_eval_pts = fe_spec_edge.NumEvaluationNodes();
310  const size_type num_rsf = fe_spec_edge.NumRefShapeFunctions();
311 
312  // Preprocessing: obtain evaluation nodes on reference segment [0,1]
313  const Eigen::MatrixXd ref_eval_pts{fe_spec_edge.EvaluationNodes()};
314  Eigen::MatrixXd eval_pts(2, num_eval_pts);
315  Eigen::Matrix<SCALAR, 1, Eigen::Dynamic> dof_vals(num_eval_pts);
316 
317  // Underlying mesh
318  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
319 
320  // Vector for returning flags and fixed values for dofs
321  const size_type num_coeffs = dofh.NumDofs();
322  std::vector<std::pair<bool, SCALAR>> flag_val_vec(num_coeffs,
323  {false, SCALAR{}});
324 
325  // *** II: Local computations ****
326  // Visit all edges of the mesh (codim-1 entities)
327  for (const lf::mesh::Entity *edge : mesh.Entities(1)) {
328  // Check whether the current edge carries dofs to be imposed by the
329  // function g. The decision relies on the predicate `esscondflag`
330  if (esscondflag(*edge)) {
331  // Evaluate mesh function at several points specified by their
332  // reference coordinates.
333  auto g_vals = g(*edge, ref_eval_pts);
334 
335  // Compute degrees of freedom from function values in evaluation
336  // points
337  dof_vals = fe_spec_edge.NodalValuesToDofs(
338  Eigen::Map<Eigen::Matrix<SCALAR, 1, Eigen::Dynamic>>(&g_vals[0], 1,
339  g_vals.size()));
340  LF_ASSERT_MSG(dof_vals.size() == num_rsf,
341  "Mismatch " << dof_vals.size() << " <-> " << num_rsf);
342  LF_ASSERT_MSG(
343  dofh.NumLocalDofs(*edge) == num_rsf,
344  "Mismatch " << dofh.NumLocalDofs(*edge) << " <-> " << num_rsf);
345  // Fetch indices of global shape functions associated with current
346  // edge
347  auto gdof_indices{dofh.GlobalDofIndices(*edge)};
348  int k = 0;
349  // Set flags and values; setting, no accumulation here!
350  for (const lf::assemble::gdof_idx_t gdof_idx : gdof_indices) {
351  flag_val_vec[gdof_idx] = {true, dof_vals[k++]};
352  }
353  }
354  }
355  return flag_val_vec;
356 }

387 template <typename LOC_COMP, typename COEFFVECTOR, typename SELECTOR>
388 double SumCellFEContrib(const lf::assemble::DofHandler &dofh,
389  LOC_COMP &loc_comp, const COEFFVECTOR &uh,
390  SELECTOR &&pred) {
391  // Type for passing local coefficient vectors
392  using dofvector_t = std::vector<typename COEFFVECTOR::value_type>;
393  // Retrieve the mesh
394  const lf::mesh::Mesh &mesh{*dofh.Mesh()};
395 
396  // Check whether sufficiently large vector uh
397  LF_ASSERT_MSG(uh.size() >= dofh.NumDofs(), "uh vector too short!");
398 
399  double sum = 0.0;
401  for (const lf::mesh::Entity *cell : mesh.Entities(0)) {
402  if (pred(*cell)) {
403  // Number of local shape functions
404  const size_type num_loc_dofs = dofh.NumLocalDofs(*cell);
405  // Allocate a temporary vector of appropriate size
406  dofvector_t loc_coeffs(num_loc_dofs);
407  // Fetch global numbers of local shape functions
408  nonstd::span<const lf::assemble::gdof_idx_t> ldof_gidx(
409  dofh.GlobalDofIndices(*cell));
410  // Copy degrees of freedom into temporory vector
411  for (int j = 0; j < num_loc_dofs; ++j) {
412  loc_coeffs[j] = uh[ldof_gidx[j]];
413  }
414  // Sum local contribution
415  sum += loc_comp(*cell, loc_coeffs);
416  } // end if (isACtive())
417  }
418  return sum;
419 }
