36 Eigen::Vector2d computeU(int n, Eigen::Vector2d x) {
37  const double r = x.norm();
38  const double u0 = 1 - std::cos(2 * lf::base::kPi * n * r);
39  Eigen::Vector2d u;

50 Eigen::Matrix2d computeUGrad(int n, Eigen::Vector2d x) {
51  const double r = x.norm();
52  const double r2 = r * r;
53  const double r3 = r2 * r;
54  const double phi = 2 * lf::base::kPi * n * r;
55  const double cphi = std::cos(phi);
56  const double sphi = std::sin(phi);
57  Eigen::Matrix2d grad;
58  grad << x[0] * x[1] / r3 * (1 - cphi) -
59  2 * lf::base::kPi * n * x[0] * x[1] / r2 * sphi,
60  (1. / r - x[0] * x[0] / r3) * (1 - cphi) +
61  2 * lf::base::kPi * n * x[0] * x[0] / r2 * sphi,
62  (x[0] * x[1] / r3 - 1. / r) * (1 - cphi) +
63  2 * lf::base::kPi * n * x[1] * x[1] / r2 * sphi,
64  -x[0] * x[1] / r3 * (1 - cphi) +
65  2 * lf::base::kPi * n * x[0] * x[1] / r2 * sphi;
66  return grad;
67 }

75 Eigen::Vector2d computeF(int n, Eigen::Vector2d x) {
76  const double r = x.norm();
77  const double f0 =
78  r == 0
79  ? -6 * n * n * lf::base::kPi * lf::base::kPi
80  : (1 -
81  (1 + 4 * n * n * lf::base::kPi * lf::base::kPi * r * r) *
82  std::cos(2 * lf::base::kPi * n * r) -
83  2 * lf::base::kPi * n * r * std::sin(2 * lf::base::kPi * n * r)) /
84  (r * r);
85  Eigen::Vector2d f;
86  f << -x[1], x[0];
87  return f0 * f.normalized();
88 }
