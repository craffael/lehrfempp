36Eigen::Vector2d computeU(int n, Eigen::Vector2d x) {
37 const double r = x.norm();
38 const double u0 = 1 - std::cos(2 * lf::base::kPi * n * r);
39 Eigen::Vector2d u;
40 u << -x[1], x[0];
41 return u0 * u.normalized();
42}

50Eigen::Matrix2d computeUGrad(int n, Eigen::Vector2d x) {
51 const double r = x.norm();
52 const double r2 = r * r;
53 const double r3 = r2 * r;
54 const double phi = 2 * lf::base::kPi * n * r;
55 const double cphi = std::cos(phi);
56 const double sphi = std::sin(phi);
57 Eigen::Matrix2d grad;
58 grad << x[0] * x[1] / r3 * (1 - cphi) -
59 2 * lf::base::kPi * n * x[0] * x[1] / r2 * sphi,
60 (1. / r - x[0] * x[0] / r3) * (1 - cphi) +
61 2 * lf::base::kPi * n * x[0] * x[0] / r2 * sphi,
62 (x[0] * x[1] / r3 - 1. / r) * (1 - cphi) +
63 2 * lf::base::kPi * n * x[1] * x[1] / r2 * sphi,
64 -x[0] * x[1] / r3 * (1 - cphi) +
65 2 * lf::base::kPi * n * x[0] * x[1] / r2 * sphi;
66 return grad;
67}

75Eigen::Vector2d computeF(int n, Eigen::Vector2d x) {
76 const double r = x.norm();
77 const double f0 =
78 r == 0
79 ? -6 * n * n * lf::base::kPi * lf::base::kPi
80 : (1 -
81 (1 + 4 * n * n * lf::base::kPi * lf::base::kPi * r * r) *
82 std::cos(2 * lf::base::kPi * n * r) -
83 2 * lf::base::kPi * n * r * std::sin(2 * lf::base::kPi * n * r)) /
84 (r * r);
85 Eigen::Vector2d f;
86 f << -x[1], x[0];
87 return f0 * f.normalized();
88}

129 // Compute the solution for each mesh
130 std::vector<ProblemSolution> solutions(mesh_count);
131 for (lf::base::size_type lvl = 0; lvl < mesh_count; ++lvl) {
132 std::filesystem::path meshpath = __FILE__;
133 meshpath = meshpath.parent_path() / concat("disk", lvl, ".msh");
134 std::unique_ptr<lf::mesh::MeshFactory> factory =
135 std::make_unique<lf::mesh::hybrid2d::MeshFactory>(2);
136 lf::io::GmshReader reader(std::move(factory), meshpath.string());
137 auto mesh = reader.mesh();
138 auto& sol = solutions[lvl];
139 sol.mesh = mesh;
140 sol.dofh = std::shared_ptr<const lf::assemble::DofHandler>(
141 new lf::assemble::UniformFEDofHandler(
142 mesh, {{lf::base::RefEl::kPoint(), 1},
143 {lf::base::RefEl::kSegment(), 1}}));
144 // Use the volume forces returned by computeF
145 auto f = [n](const Eigen::Vector2d& x) -> Eigen::Vector2d {
146 return computeF(n, x);
147 };
148 // The velocity on the boundary is zero everywhere
149 auto dirichlet_funct = [](const lf::mesh::Entity&
150 /*unused*/) -> Eigen::Vector2d {
151 return Eigen::Vector2d::Zero();
152 };
153 const auto [A, rhs] =
154 projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
155 sol.mesh, *(sol.dofh), f, dirichlet_funct, 1,
156 lf::quad::make_QuadRule(lf::base::RefEl::kTria(), quadrule_degree),
157 false);
158 sol.A = A.makeSparse();
159 sol.rhs = rhs;
160 Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
161 solver.compute(sol.A);
162 sol.solution = solver.solve(rhs);
163 const auto [A_modified, rhs_modified] =
164 projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
165 sol.mesh, *(sol.dofh), f, dirichlet_funct, 1,
166 lf::quad::make_QuadRule(lf::base::RefEl::kTria(), quadrule_degree),
167 true);
168 sol.A_modified = A_modified.makeSparse();
169 const Eigen::SparseLU<Eigen::SparseMatrix<double>> solver_modified(
170 sol.A_modified);
171 sol.solution_modified = solver_modified.solve(rhs_modified);
172 }
