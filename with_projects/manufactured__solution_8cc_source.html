39 Eigen::Vector2d computeU(int n, Eigen::Vector2d x) {
40  const double r = x.norm();
41  const double u0 = 1 - std::cos(2 * M_PI * n * r);
42  Eigen::Vector2d u;
43  u << -x[1], x[0];
44  return u0 * u.normalized();

53 Eigen::Matrix2d computeUGrad(int n, Eigen::Vector2d x) {
54  const double r = x.norm();
55  const double r2 = r * r;
56  const double r3 = r2 * r;
57  const double phi = 2 * M_PI * n * r;
58  const double cphi = std::cos(phi);
59  const double sphi = std::sin(phi);
60  Eigen::Matrix2d grad;
61  grad << x[0] * x[1] / r3 * (1 - cphi) -
62  2 * M_PI * n * x[0] * x[1] / r2 * sphi,
63  (1. / r - x[0] * x[0] / r3) * (1 - cphi) +
64  2 * M_PI * n * x[0] * x[0] / r2 * sphi,
65  (x[0] * x[1] / r3 - 1. / r) * (1 - cphi) +
66  2 * M_PI * n * x[1] * x[1] / r2 * sphi,
67  -x[0] * x[1] / r3 * (1 - cphi) + 2 * M_PI * n * x[0] * x[1] / r2 * sphi;
68  return grad;
69 }

77 Eigen::Vector2d computeF(int n, Eigen::Vector2d x) {
78  const double r = x.norm();
79  const double f0 = r == 0 ? -6 * n * n * M_PI * M_PI
80  : (1 -
81  (1 + 4 * n * n * M_PI * M_PI * r * r) *
82  std::cos(2 * M_PI * n * r) -
83  2 * M_PI * n * r * std::sin(2 * M_PI * n * r)) /
84  (r * r);
85  Eigen::Vector2d f;
86  f << -x[1], x[0];
87  return f0 * f.normalized();
88 }

132  std::filesystem::path meshpath = __FILE__;
133  meshpath = meshpath.parent_path() / concat("disk", lvl, ".msh");
134  std::unique_ptr<lf::mesh::MeshFactory> factory =
135  std::make_unique<lf::mesh::hybrid2d::MeshFactory>(2);
136  lf::io::GmshReader reader(std::move(factory), meshpath.string());
137  auto mesh = reader.mesh();
138  auto& sol = solutions[lvl];
139  sol.mesh = mesh;
140  sol.dofh = std::shared_ptr<const lf::assemble::DofHandler>(
141  new lf::assemble::UniformFEDofHandler(
142  mesh, {{lf::base::RefEl::kPoint(), 1},
143  {lf::base::RefEl::kSegment(), 1}}));
144  // Use the volume forces returned by computeF
145  auto f = [n](const Eigen::Vector2d& x) -> Eigen::Vector2d {
146  return computeF(n, x);
147  };
148  // The velocity on the boundary is zero everywhere
149  auto dirichlet_funct = [](const lf::mesh::Entity &
150  /*unused*/) -> Eigen::Vector2d {
151  return Eigen::Vector2d::Zero();
152  };
153  const auto [A, rhs] =
154  projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
155  sol.mesh, *(sol.dofh), f, dirichlet_funct, 1,
156  lf::quad::make_QuadRule(lf::base::RefEl::kTria(), quadrule_degree),
157  false);
158  sol.A = A.makeSparse();
159  sol.rhs = rhs;
160  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
161  solver.compute(sol.A);
162  sol.solution = solver.solve(rhs);
163  const auto [A_modified, rhs_modified] =
164  projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
165  sol.mesh, *(sol.dofh), f, dirichlet_funct, 1,
166  lf::quad::make_QuadRule(lf::base::RefEl::kTria(), quadrule_degree),
167  true);
168  sol.A_modified = A_modified.makeSparse();
169  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver_modified(
170  sol.A_modified);
171  sol.solution_modified = solver_modified.solve(rhs_modified);
172  }

186  solutions[lvl].mesh);

188  [n](const Eigen::Vector2d& x) { return computeU(n, x); });
189  const auto grad_exact = lf::mesh::utils::MeshFunctionGlobal(
190  [n](const Eigen::Vector2d& x) -> Eigen::Matrix2d {
191  return computeUGrad(n, x);

193  projects::ipdg_stokes::post_processing::MeshFunctionVelocity<double, double>
194  velocity(fe_space, solutions[lvl].solution);

196  velocity_modified(fe_space, solutions[lvl].solution_modified);

200  writer.WriteCellData("analyticF", analyticF);
201  writer.WriteCellData("U", velocity);
202  writer.WriteCellData("U_modified", velocity_modified);
203  // The error in the velocity
204  const auto diff_v = velocity - velocity_exact;
205  const auto diff_v_modified = velocity_modified - velocity_exact;
206  // The error in the gradient of the velocity

222  *(solutions[lvl].mesh),
223  lf::uscalfe::transpose(velocity_modified) * velocity_exact,
224  qr_provider)(0, 0) /
225  lf::uscalfe::IntegrateMeshFunction(
226  *(solutions[lvl].mesh), lf::uscalfe::squaredNorm(velocity_modified),
227  qr_provider);
