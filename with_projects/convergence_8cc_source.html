67int main() {
68 const unsigned refinement_level = 7;
69
70 // Build the mesh
71 std::unique_ptr<lf::mesh::MeshFactory> factory =
72 std::make_unique<lf::mesh::hybrid2d::MeshFactory>(2);
73 lf::mesh::utils::TPTriagMeshBuilder builder(std::move(factory));
74 builder.setBottomLeftCorner(0, 0);
75 builder.setTopRightCorner(1, 1);
76 builder.setNumXCells(2);
77 builder.setNumYCells(2);
78 auto mesh0 = builder.Build();
79
80 // Generate a mesh hierarchy by regular refinement
81 const auto mesh_hierarchy =
82 lf::refinement::GenerateMeshHierarchyByUniformRefinemnt(mesh0,
83 refinement_level);
84
85 // Solve the problem for each mesh in the hierarchy
86 std::vector<ProblemSolution> solutions(refinement_level + 1);
87 for (lf::base::size_type lvl = 0; lvl < refinement_level + 1; ++lvl) {
88 const auto &mesh = mesh_hierarchy->getMesh(lvl);
89 auto &sol = solutions[lvl];
90 sol.mesh = mesh;
91 sol.dofh = std::shared_ptr<const lf::assemble::DofHandler>(
92 new lf::assemble::UniformFEDofHandler(
93 mesh, {{lf::base::RefEl::kPoint(), 1},
94 {lf::base::RefEl::kSegment(), 1}}));
95 // No volume forces are present in this experiment
96 auto f = [](const Eigen::Vector2d & /*unused*/) -> Eigen::Vector2d {
97 return Eigen::Vector2d::Zero();
98 };
99 // The top lid is driven with velocity 1
100 auto dirichlet_funct = [](const lf::mesh::Entity &edge) -> Eigen::Vector2d {
101 static constexpr double eps = 1e-10;
102 const auto *const geom = edge.Geometry();
103 const auto vertices = geom->Global(edge.RefEl().NodeCoords());
104 Eigen::Vector2d v;
105 v << 1, 0;
106 if (vertices(1, 0) <= 1 + eps && vertices(1, 0) >= 1 - eps &&
107 vertices(1, 1) <= 1 + eps && vertices(1, 1) >= 1 - eps) {
108 return -v;
109 }
110 return Eigen::Vector2d::Zero();
111 };
112 const auto [A, rhs] =
113 projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
114 sol.mesh, *(sol.dofh), f, dirichlet_funct, 100,
115 lf::quad::make_TriaQR_MidpointRule(), false);
116 sol.A = A.makeSparse();
117 sol.rhs = rhs;
118 Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
119 solver.compute(sol.A);
120 sol.solution = solver.solve(rhs);
121 const auto [A_modified, rhs_modified] =
122 projects::ipdg_stokes::assemble::buildSystemMatrixNoFlow(
123 sol.mesh, *(sol.dofh), f, dirichlet_funct, 100,
124 lf::quad::make_TriaQR_MidpointRule(), true);
125 sol.A_modified = A_modified.makeSparse();
126 const Eigen::SparseLU<Eigen::SparseMatrix<double>> solver_modified(
127 sol.A_modified);
128 sol.solution_modified = solver_modified.solve(rhs_modified);
129 }
130
131 // Perform post processing on the data
132 const auto fe_space_fine =
133 std::make_shared<lf::uscalfe::FeSpaceLagrangeO1<double>>(
134 solutions.back().mesh);
135 const projects::ipdg_stokes::post_processing::MeshFunctionVelocity<double,
136 double>
137 velocity_exact(fe_space_fine, solutions.back().solution);
138 const projects::ipdg_stokes::post_processing::MeshFunctionVelocity<double,
139 double>
140 velocity_exact_modified(fe_space_fine,
141 solutions.back().solution_modified);
142 lf::io::VtkWriter writer(solutions.back().mesh, "result.vtk");
143 for (lf::base::size_type lvl = 0; lvl < refinement_level; ++lvl) {
144 const auto fe_space_lvl =
145 std::make_shared<lf::uscalfe::FeSpaceLagrangeO1<double>>(
146 solutions[lvl].mesh);
147 const projects::ipdg_stokes::post_processing::MeshFunctionVelocity<double,
148 double>
149 velocity_lvl(fe_space_lvl, solutions[lvl].solution);
150 const projects::ipdg_stokes::post_processing::MeshFunctionVelocity<double,
151 double>
152 velocity_lvl_modified(fe_space_lvl, solutions[lvl].solution_modified);
153 const lf::refinement::MeshFunctionTransfer velocity_fine(
154 *mesh_hierarchy, velocity_lvl, lvl, mesh_hierarchy->NumLevels() - 1);
155 const lf::refinement::MeshFunctionTransfer velocity_fine_modified(
156 *mesh_hierarchy, velocity_lvl_modified, lvl,
157 mesh_hierarchy->NumLevels() - 1);
158 writer.WriteCellData(concat("v_", solutions[lvl].mesh->NumEntities(2)),
159 velocity_fine);
160 writer.WriteCellData(
161 concat("v_modified", solutions[lvl].mesh->NumEntities(2)),
162 velocity_fine_modified);
163 // We need to implement our own binary mesh function for multiplication
164 const auto qr_provider = [](const lf::mesh::Entity &e) {
165 return lf::quad::make_QuadRule(e.RefEl(), 0);
166 };
167 const auto weight =
168 lf::mesh::utils::MeshFunctionGlobal([](const Eigen::Vector2d &x) {
169 return x[1] >= 0.9
170 ? (1 - std::cos(lf::base::kPi / 0.1 * (1 - x[1]))) / 2
171 : 1.;
172 });
173 const auto diff = velocity_fine - velocity_exact;
174 const auto diff_weighted = weight * diff;
175 const auto diff_modified = velocity_fine_modified - velocity_exact_modified;
176 const auto diff_weighted_modified = weight * diff_modified;
177 const double L2 = projects::ipdg_stokes::post_processing::L2norm(
178 solutions.back().mesh, diff, qr_provider);
179 const double L2w = projects::ipdg_stokes::post_processing::L2norm(
180 solutions.back().mesh, diff_weighted, qr_provider);
181 const double DG = projects::ipdg_stokes::post_processing::DGnorm(
182 solutions.back().mesh, diff,
183 lf::mesh::utils::MeshFunctionConstant<Eigen::Matrix2d>(
184 Eigen::Matrix2d::Zero()),
185 qr_provider);
186 const double L2_modified = projects::ipdg_stokes::post_processing::L2norm(
187 solutions.back().mesh, diff_modified, qr_provider);
188 const double L2w_modified = projects::ipdg_stokes::post_processing::L2norm(
189 solutions.back().mesh, diff_weighted_modified, qr_provider);
190 const double DG_modified = projects::ipdg_stokes::post_processing::DGnorm(
191 solutions.back().mesh, diff_modified,
192 lf::mesh::utils::MeshFunctionConstant<Eigen::Matrix2d>(
193 Eigen::Matrix2d::Zero()),
194 qr_provider);
195 std::cout << lvl << ' ' << solutions[lvl].mesh->NumEntities(2) << ' ' << L2
196 << ' ' << L2w << ' ' << DG << ' ' << L2_modified << ' '
197 << L2w_modified << ' ' << DG_modified << '\n';
198 }
199
200 return 0;
