23 Eigen::VectorXd PiecewiseConstElementVectorProvider::Eval(
24  const lf::mesh::Entity &entity) const {
25  const auto geom = entity.Geometry();
26  // Construct the basis functions from the curl of the standard hat functions
27  const lf::uscalfe::FeLagrangeO1Tria<double> hat_func;
28  const Eigen::MatrixXd ref_grads =
29  hat_func.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
30  .transpose();
31  const Eigen::MatrixXd J_inv_trans =
32  geom->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
33  const Eigen::MatrixXd grads = J_inv_trans * ref_grads;
34  Eigen::MatrixXd basis_funct(2, 3);
35  basis_funct << grads.row(1), -grads.row(0);
36  // Compute the elements of the vector via the given quadrature rule
37  Eigen::VectorXd element_vector = Eigen::VectorXd::Zero(6);
38  const Eigen::MatrixXd points = geom->Global(quadrule_.Points());
39  const Eigen::VectorXd weights =
40  (geom->IntegrationElement(quadrule_.Points()).array() *
41  quadrule_.Weights().array())
42  .matrix();
43  for (lf::base::size_type n = 0; n < quadrule_.NumPoints(); ++n) {
44  const auto f_eval = f_(points.col(n));
45  element_vector.head(3) += basis_funct.transpose() * weights[n] * f_eval;
46  }
47  // Weakly impose the dirichlet boundary conditions
48  Eigen::MatrixXd vertices = geom->Global(entity.RefEl().NodeCoords());
49  // Use the vertex coordinates to compute the local normals on the edges
50  const Eigen::Matrix<double, 2, 3> normals =
51  projects::ipdg_stokes::mesh::computeOutwardNormals(entity);
52  // Iterate over all edges of the current entity and set the boundary
53  // conditions where necessary
54  const auto edges = entity.SubEntities(1);
55  for (int edge_idx = 0; edge_idx < 3; ++edge_idx) {
56  const auto &edge = *edges[edge_idx];
57  if (boundary_(edge)) {
58  Eigen::Vector2d t;
59  t << -normals(1, edge_idx), normals(0, edge_idx);
60  element_vector[edge_idx + 3] = dirichlet_data_(edge).transpose() * t;
61  }
62  }
63  return element_vector;
64 }
