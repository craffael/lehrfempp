838 template <class MESH_FUNCTION, class>
839 // NOLINTNEXTLINE(readability-function-cognitive-complexity)
840 void VtkWriter::WritePointData(const std::string& name,
841  const MESH_FUNCTION& mesh_function) {
842  // we have to evaluate the mesh function at all points
843  // and write this into the vtk file:
844  CheckAttributeSetName(vtk_file_.point_data, name);
845  using T = mesh::utils::MeshFunctionReturnType<MESH_FUNCTION>;
846  auto dim_mesh = mesh_->DimMesh();
847 
848  Eigen::Matrix<double, 0, 1> origin;
849 
850  if constexpr (std::is_same_v<T, unsigned char> || std::is_same_v<T, char> ||
851  std::is_same_v<T, unsigned> || std::is_same_v<T, int> ||
852  std::is_same_v<T, float> ||
853  std::is_same_v<T, double>) { // NOLINT
854  // MeshFunction is scalar valued:
855  VtkFile::ScalarData<T> data{};
856  data.data.resize(vtk_file_.unstructured_grid.points.size());
857  data.name = name;
858 
859  // evaluate at nodes:
860  for (const auto* n : mesh_->Entities(dim_mesh)) {
861  data.data[mesh_->Index(*n)] = mesh_function(*n, origin)[0];
862  }
863 
864  for (int codim = static_cast<int>(dim_mesh - 1);
865  codim >= static_cast<char>(codim_); --codim) {
866  for (const auto* e : mesh_->Entities(codim)) {
867  auto ref_el = e->RefEl();
868  if (order_ == 1 || (order_ == 2 && ref_el == base::RefEl::kTria())) {
869  continue;
870  }
871  auto values = mesh_function(*e, aux_nodes_[ref_el.Id()]);
872  auto offset = aux_node_offset_[codim](*e);
873  for (typename std::vector<T>::size_type i = 0; i < values.size(); ++i) {
874  data.data[offset + i] = values[i];
875  }
876  }
877  }
878  vtk_file_.point_data.push_back(std::move(data));
879  } else if constexpr (base::is_eigen_matrix<T>) { // NOLINT
880  static_assert(T::ColsAtCompileTime == 1,
881  "The MeshFunction must return row-vectors");
882  static_assert(
883  T::RowsAtCompileTime == Eigen::Dynamic ||
884  (T::RowsAtCompileTime > 1 && T::RowsAtCompileTime < 4),
885  "The Row vectors returned by the MeshFunction must be either dynamic "
886  "or must have 2 or 3 rows (at compile time).");
887  using Scalar = typename T::Scalar;
888  static_assert(
889  std::is_same_v<double, Scalar> || std::is_same_v<float, Scalar>,
890  "The RowVectors returned by the MeshFunction must be either double "
891  "or float valued.");
892  VtkFile::VectorData<Scalar> data{};
893  data.data.resize(vtk_file_.unstructured_grid.points.size());
894  data.name = name;
895 
896  // evaluate at nodes:
897  for (const auto* n : mesh_->Entities(dim_mesh)) {
898  PadWithZeros<T::RowsAtCompileTime, Scalar>(data.data[mesh_->Index(*n)],
899  mesh_function(*n, origin)[0]);
900  }
901 
902  for (int codim = static_cast<int>(dim_mesh - 1);
903  codim >= static_cast<char>(codim_); --codim) {
904  for (const auto* e : mesh_->Entities(codim)) {
905  auto ref_el = e->RefEl();
906  if (order_ < 2 || (order_ < 3 && ref_el == base::RefEl::kTria())) {
907  continue;
908  }
909  auto values = mesh_function(*e, aux_nodes_[ref_el.Id()]);
910  auto offset = aux_node_offset_[codim](*e);
911  for (unsigned long i = 0; i < values.size(); ++i) {
912  PadWithZeros<T::RowsAtCompileTime, Scalar>(data.data[offset + i],
913  values[i]);
914  }
915  }
916  }
917  vtk_file_.point_data.push_back(std::move(data));
918  } else {
919  LF_VERIFY_MSG(false,
920  "MeshFunction values must be one of: unsigned char, char, "
921  "unsigned, int, float, double, Eigen::Vector<double, ...> "
922  "or Eigen::Vector<float, ...>");
923  }
924 }

926 template <class MESH_FUNCTION, class>
927 void VtkWriter::WriteCellData(const std::string& name,
928  const MESH_FUNCTION& mesh_function) {
929  CheckAttributeSetName(vtk_file_.cell_data, name);
930  using T = mesh::utils::MeshFunctionReturnType<MESH_FUNCTION>;
931 
932  // maps from RefEl::Id() -> barycenter of the reference element
933  std::vector<Eigen::VectorXd> barycenters(5);
934  barycenters[base::RefEl::kPoint().Id()] = Eigen::Matrix<double, 0, 1>();
935  barycenters[base::RefEl::kSegment().Id()] =
936  base::RefEl::kSegment().NodeCoords().rowwise().sum() / 2.;
937  barycenters[base::RefEl::kTria().Id()] =
938  base::RefEl::kTria().NodeCoords().rowwise().sum() / 3.;
939  barycenters[base::RefEl::kQuad().Id()] =
940  base::RefEl::kTria().NodeCoords().rowwise().sum() / 4.;
941 
942  if constexpr (base::is_eigen_array<T> || base::is_eigen_matrix<T>) {
943  static_assert(T::ColsAtCompileTime == 1,
944  "The MeshFunction must return row-vectors");
945  static_assert(
946  T::RowsAtCompileTime == Eigen::Dynamic ||
947  (T::RowsAtCompileTime > 1 && T::RowsAtCompileTime < 4),
948  "The Row vectors returned by the MeshFunction must be either dynamic "
949  "or must have 2 or 3 rows (at compile time).");
950  using Scalar = typename T::Scalar;
951  static_assert(
952  std::is_same_v<double, Scalar> || std::is_same_v<float, Scalar>,
953  "The RowVectors returned by the MeshFunction must be either double "
954  "or float valued.");
955  VtkFile::VectorData<Scalar> data{};
956  data.data.resize(mesh_->NumEntities(codim_));
957  data.name = name;
958  constexpr int rows = T::RowsAtCompileTime;
959  for (const auto* p : mesh_->Entities(codim_)) {
960  this->PadWithZeros<rows, Scalar>(
961  data.data[mesh_->Index(*p)],
962  mesh_function(*p, barycenters[p->RefEl().Id()])[0]);
963  }
964  vtk_file_.cell_data.push_back(std::move(data));
965  } else {
966  VtkFile::ScalarData<T> data{};
967  data.data.resize(mesh_->NumEntities(codim_));
968  data.name = name;
969  for (const auto* e : mesh_->Entities(codim_)) {
970  data.data[mesh_->Index(*e)] =
971  mesh_function(*e, barycenters[e->RefEl().Id()])[0];
972  }
973  vtk_file_.cell_data.push_back(std::move(data));
974  }
975 }

980  if (std::find_if(data.begin(), data.end(), [&](auto& d) {

995 template <int ROWS, class T>
996 void VtkWriter::PadWithZeros(Eigen::Matrix<T, 3, 1>& out,
997  const Eigen::Matrix<T, ROWS, 1>& in) {
998  if constexpr (ROWS == 2) { // NOLINT
999  out.template block<2, 1>(0, 0) = in;
1000  out(2) = T(0); // NOLINT
1001  } else if constexpr (ROWS == 3) { // NOLINT
1002  out = in;
1003  } else if constexpr (ROWS == Eigen::Dynamic) { // NOLINT
1004  LF_ASSERT_MSG(in.rows() == 2 || in.rows() == 3,
1005  "VtkWriter can only write out 2d or 3d vectors.");
1006  if (in.rows() == 2) {
1007  out(2) = T{0};
1008  out.topRows(2) = in;
1009  } else {
1010  out = in;
1011  }
1012  }
