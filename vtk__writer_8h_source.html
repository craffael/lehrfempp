840 template <class MESH_FUNCTION, class>
841 void VtkWriter::WritePointData(const std::string& name,
842  const MESH_FUNCTION& mesh_function) {
843  if (order_ == 1) {
844  Eigen::Matrix<double, 0, 1> origin{};
845  WritePointData(
846  name, *mesh::utils::make_LambdaMeshDataSet(
847  [&](const auto& e) { return mesh_function(e, origin)[0]; }));
848  } else {
849  // for higher orders, we have to evaluate the mesh function at all points
850  // and write this into the vtk file:
851  CheckAttributeSetName(vtk_file_.point_data, name);
852  using T = mesh::utils::MeshFunctionReturnType<MESH_FUNCTION>;
853  auto dim_mesh = mesh_->DimMesh();
854 
855  Eigen::Matrix<double, 0, 1> origin;
856 
857  if constexpr (std::is_same_v<T, unsigned char> || std::is_same_v<T, char> ||
858  std::is_same_v<T, unsigned> || std::is_same_v<T, int> ||
859  std::is_same_v<T, float> || std::is_same_v<T, double>) {
860  // MeshFunction is scalar valued:
861  VtkFile::ScalarData<T> data{};
862  data.data.resize(vtk_file_.unstructured_grid.points.size());
863  data.name = name;
864 
865  // evaluate at nodes:
866  for (auto n : mesh_->Entities(dim_mesh)) {
867  data.data[mesh_->Index(*n)] = mesh_function(*n, origin)[0];
868  }
869 
870  for (int codim = static_cast<int>(dim_mesh - 1);
871  codim >= static_cast<char>(codim_); --codim) {
872  for (auto e : mesh_->Entities(codim)) {
873  auto ref_el = e->RefEl();
874  if (order_ < 3 && ref_el == base::RefEl::kTria()) {
875  continue;
876  }
877  auto values = mesh_function(*e, aux_nodes_[ref_el.Id()]);
878  auto offset = aux_node_offset_[codim](*e);
879  for (int i = 0; i < values.size(); ++i) {
880  data.data[offset + i] = values[i];
881  }
882  }
883  }
884  vtk_file_.point_data.push_back(std::move(data));
885  } else if constexpr (base::is_eigen_matrix<T>) {
886  static_assert(T::ColsAtCompileTime == 1,
887  "The MeshFunction must return row-vectors");
888  static_assert(
889  T::RowsAtCompileTime == Eigen::Dynamic ||
890  (T::RowsAtCompileTime > 1 && T::RowsAtCompileTime < 4),
891  "The Row vectors returned by the MeshFunction must be either dynamic "
892  "or must have 2 or 3 rows (at compile time).");
893  using Scalar = typename T::Scalar;
894  static_assert(
895  std::is_same_v<double, Scalar> || std::is_same_v<float, Scalar>,
896  "The RowVectors returned by the MeshFunction must be either double "
897  "or float valued.");
898  VtkFile::VectorData<Scalar> data{};
899  data.data.resize(vtk_file_.unstructured_grid.points.size());
900  data.name = name;
901 
902  // evaluate at nodes:
903  for (auto n : mesh_->Entities(dim_mesh)) {
904  PadWithZeros<T::RowsAtCompileTime, Scalar>(
905  data.data[mesh_->Index(*n)], mesh_function(*n, origin)[0]);
906  }
907 
908  for (int codim = static_cast<int>(dim_mesh - 1);
909  codim >= static_cast<char>(codim_); --codim) {
910  for (auto e : mesh_->Entities(codim)) {
911  auto ref_el = e->RefEl();
912  if (order_ < 3 && ref_el == base::RefEl::kTria()) {
913  continue;
914  }
915  auto values = mesh_function(*e, aux_nodes_[ref_el.Id()]);
916  auto offset = aux_node_offset_[codim](*e);
917  for (int i = 0; i < values.size(); ++i) {
918  PadWithZeros<T::RowsAtCompileTime, Scalar>(data.data[offset + i],
919  values[i]);
920  }
921  }
922  }
923  vtk_file_.point_data.push_back(std::move(data));
924  } else {
925  LF_VERIFY_MSG(false,
926  "MeshFunction values must be one of: unsigned char, char, "
927  "unsigned, int, float, double, Eigen::Vector<double, ...> "
928  "or Eigen::Vector<float, ...>");
929  }
930  }
931 }

933 template <class MESH_FUNCTION, class>
934 void VtkWriter::WriteCellData(const std::string& name,
935  const MESH_FUNCTION& mesh_function) {
936  // maps from RefEl::Id() -> barycenter of the reference element
937  std::vector<Eigen::VectorXd> barycenters(5);
938  barycenters[base::RefEl::kPoint().Id()] = Eigen::Matrix<double, 0, 1>();
939  barycenters[base::RefEl::kSegment().Id()] =
940  base::RefEl::kSegment().NodeCoords().rowwise().sum() / 2.;
941  barycenters[base::RefEl::kTria().Id()] =
942  base::RefEl::kTria().NodeCoords().rowwise().sum() / 3.;
943  barycenters[base::RefEl::kQuad().Id()] =
944  base::RefEl::kTria().NodeCoords().rowwise().sum() / 4.;
945 
946  WriteCellData(name, *mesh::utils::make_LambdaMeshDataSet([&](const auto& e) {
947  return mesh_function(e, barycenters[e.RefEl().Id()])[0];
948  }));
949 }

969 template <int ROWS, class T>
970 void VtkWriter::PadWithZeros(Eigen::Matrix<T, 3, 1>& out,
971  const Eigen::Matrix<T, ROWS, 1>& in) {
972  if constexpr (ROWS == 2) { // NOLINT
973  out.template block<2, 1>(0, 0) = in;
974  out(2) = T(0);
975  } else if constexpr (ROWS == 3) { // NOLINT
976  out = in;
977  } else if constexpr (ROWS == Eigen::Dynamic) { // NOLINT
978  if (in.rows() == 2) {
979  out(2) = T(0);
980  out.topRows(2) = in;
981  } else {
982  out = in;
983  }
984  }
985 }
