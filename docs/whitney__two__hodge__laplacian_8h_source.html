84 void Compute() {
85 // create element matrix provider
86 projects::hldo_sphere::assemble::WhitneyOneMassMatrixProvider
87 matrix_dot_provider;
88
89 projects::hldo_sphere::assemble::RotWhitneyOneDivMatrixProvider
90 matrix_curl_provider;
91
92 const lf::assemble::DofHandler &dof_handler_div =
93 lf::assemble::UniformFEDofHandler(mesh_p_,
94 {{lf::base::RefEl::kSegment(), 1}});
95
96 const lf::assemble::DofHandler &dof_handler_const =
97 lf::assemble::UniformFEDofHandler(mesh_p_,
98 {{lf::base::RefEl::kTria(), 1}});
99
100 // create COO Matrix
101 const lf::assemble::size_type n_dofs_div(dof_handler_div.NumDofs());
102 const lf::assemble::size_type n_dofs_const(dof_handler_const.NumDofs());
103
104 lf::assemble::COOMatrix<double> coo_A_11(n_dofs_div, n_dofs_div);
105 coo_A_11.setZero();
106 lf::assemble::COOMatrix<double> coo_A_12(n_dofs_div, n_dofs_const);
107 coo_A_12.setZero();
108
109 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
110 0, dof_handler_div, dof_handler_div, matrix_dot_provider, coo_A_11);
111
112 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
113 0, dof_handler_const, dof_handler_div, matrix_curl_provider, coo_A_12);
114
115 // create full matrix
116 lf::assemble::COOMatrix<SCALAR> full_matrix(n_dofs_div + n_dofs_const,
117 n_dofs_div + n_dofs_const);
118 full_matrix.setZero();
119
120 // iterate over all triplets of the previously computed matrices and add up
121 // entries
122 const std::vector<Eigen::Triplet<double>> triplets_A_11 =
123 coo_A_11.triplets();
124
125 const std::vector<Eigen::Triplet<double>> triplets_A_12 =
126 coo_A_12.triplets();
127
128 // Add A_11
129 for (Eigen::Triplet<double> triplet : triplets_A_11) {
130 int col = triplet.col();
131 int row = triplet.row();
132 SCALAR val = triplet.value();
133 full_matrix.AddToEntry(row, col, val);
134 }
135
136 // Add A_12 and A_21
137 for (Eigen::Triplet<double> triplet : triplets_A_12) {
138 int row = triplet.row();
139 int col = triplet.col() + n_dofs_div;
140 SCALAR val = triplet.value();
141
142 // Add A_12
143 full_matrix.AddToEntry(row, col, val);
144
145 // Add A_21
146 full_matrix.AddToEntry(col, row, -val);
147 }
148
149 coo_matrix_ = full_matrix;
150
151 // create element vector provider
152 projects::hldo_sphere::assemble::WhitneyTwoVectorProvider<SCALAR>
153 vector_provider(f_);
154
155 // create load vector
156 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> phi(n_dofs_const);
157 phi.setZero();
158
159 // assemble the global vector over entities with codim=0:
160 lf::assemble::AssembleVectorLocally(0, dof_handler_const, vector_provider,
161 phi);
162
163 // create full vector
164 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> full_vec(n_dofs_div +
165 n_dofs_const);
166 full_vec.setZero();
167 full_vec.tail(n_dofs_const) = phi;
168
169 // set the class attributes
170 phi_ = full_vec;
171 }
