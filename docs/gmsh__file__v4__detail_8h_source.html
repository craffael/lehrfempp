157template <class Enum, class RawValue>
158struct assign_to_attribute_from_value<
159 Enum, RawValue,
160 typename std::enable_if_t<std::is_enum_v<Enum> &&
161 !std::is_same_v<Enum, RawValue>>> {
162 static void call(RawValue const& raw, Enum& cat) {
163 if constexpr (detail::has_value_type<RawValue>::value) { // NOLINT
164 // specialization for endian::endian
165 typename RawValue::value_type value = raw;
166 cat = static_cast<Enum>(value);
167 } else { // NOLINT
168 cat = static_cast<Enum>(raw);
169 }
170 }
171};
