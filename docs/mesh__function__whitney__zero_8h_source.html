95 }
96
97 // for each point evaluate the basis functions and add the

99 for (int i = 0; i < local.cols(); i++) {
100 vals[i] = mu_(global_indices[0]) * lambda(0) +
101 mu_(global_indices[1]) * lambda(1) +
102 mu_(global_indices[2]) * lambda(2);
103 }
