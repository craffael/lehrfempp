87 void Compute() {
88 // create element matrix provider
89 projects::hldo_sphere::assemble::WhitneyOneCurlCurlMatrixProvider
90 matrix_curl_provider;
91 projects::hldo_sphere::assemble::WhitneyOneGradMatrixProvider
92 matrix_grad_provider;
93 projects::hldo_sphere::assemble::MassMatrixProvider matrix_mass_provider;
94
95 const lf::assemble::DofHandler &dof_handler_curl =
96 lf::assemble::UniformFEDofHandler(mesh_p_,
97 {{lf::base::RefEl::kSegment(), 1}});
98 const lf::assemble::DofHandler &dof_handler_bary =
99 lf::assemble::UniformFEDofHandler(mesh_p_,
100 {{lf::base::RefEl::kPoint(), 1}});
101
102 // create COO Matrix
103 const lf::assemble::size_type n_dofs_curl(dof_handler_curl.NumDofs());
104 const lf::assemble::size_type n_dofs_bary(dof_handler_bary.NumDofs());
105
106 lf::assemble::COOMatrix<double> coo_A_11(n_dofs_curl, n_dofs_curl);
107 coo_A_11.setZero();
108 lf::assemble::COOMatrix<double> coo_A_21(n_dofs_curl, n_dofs_bary);
109 coo_A_21.setZero();
110 lf::assemble::COOMatrix<double> coo_A_22(n_dofs_bary, n_dofs_bary);
111 coo_A_22.setZero();
112
113 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
114 0, dof_handler_curl, dof_handler_curl, matrix_curl_provider, coo_A_11);
115
116 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
117 0, dof_handler_bary, dof_handler_curl, matrix_grad_provider, coo_A_21);
118
119 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
120 0, dof_handler_bary, dof_handler_bary, matrix_mass_provider, coo_A_22);
121
122 // create full matrix
123 lf::assemble::COOMatrix<SCALAR> full_matrix(n_dofs_curl + n_dofs_bary,
124 n_dofs_curl + n_dofs_bary);
125 full_matrix.setZero();
126
127 // iterate over all triplets of the previously computed matrice and add up
128 // entries
129 const std::vector<Eigen::Triplet<double>> triplets_A_11 =
130 coo_A_11.triplets();
131 const std::vector<Eigen::Triplet<double>> triplets_A_21 =
132 coo_A_21.triplets();
133 const std::vector<Eigen::Triplet<double>> triplets_A_22 =
134 coo_A_22.triplets();
135
136 // Add A_11
137 for (Eigen::Triplet<double> triplet : triplets_A_11) {
138 int col = triplet.col();
139 int row = triplet.row();
140 SCALAR val = triplet.value();
141 full_matrix.AddToEntry(row, col, val);
142 }
143
144 // Add A_12 and A_21
145 for (Eigen::Triplet<double> triplet : triplets_A_21) {
146 int col = triplet.col() + n_dofs_curl;
147 int row = triplet.row();
148 SCALAR val = triplet.value();
149
150 // Add A_12
151 full_matrix.AddToEntry(row, col, -val);
152
153 // Add A_21
154 full_matrix.AddToEntry(col, row, val);
155 }
156
157 // Add A_22
158 for (Eigen::Triplet<double> triplet : triplets_A_22) {
159 int col = triplet.col() + n_dofs_curl;
160 int row = triplet.row() + n_dofs_curl;
161 SCALAR val = triplet.value();
162 full_matrix.AddToEntry(row, col, val);
163 }
164
165 coo_matrix_ = full_matrix;
166
167 // create element vector provider
168 projects::hldo_sphere::assemble::WhitneyOneVectorProvider<SCALAR>
169 vector_provider(f_);
170
171 // create load vector
172 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> phi(n_dofs_curl);
173 phi.setZero();
174
175 // assemble the global vector over entities with codim=0:
176 AssembleVectorLocally(0, dof_handler_curl, vector_provider, phi);
177
178 // create full vector
179 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> full_vec(n_dofs_curl +
180 n_dofs_bary);
181 full_vec.setZero();
182 full_vec.head(n_dofs_curl) = phi;
183
184 // set the class attributes
185 phi_ = full_vec;
186 }
