16Eigen::VectorXd LoadVectorProvider::Eval(const lf::mesh::Entity &entity) const {
17 const auto *const geom = entity.Geometry();
18
19 // Only triangles are supported
20 LF_VERIFY_MSG(entity.RefEl() == lf::base::RefEl::kTria(),
21 "Unsupported cell type " << entity.RefEl());
22
23 // Compute the global vertex coordinates
24 Eigen::MatrixXd vertices = geom->Global(entity.RefEl().NodeCoords());
25
26 // define quad rule with sufficiantly high degree since the
27 // baricentric coordinate functions have degree 1
28 lf::quad::QuadRule quadrule{lf::quad::make_TriaQR_EdgeMidpointRule()};
29
30 const lf::uscalfe::FeLagrangeO1Tria<double> hat_func;
31 const auto lambda_hat = [&](Eigen::Vector2d x_hat) -> Eigen::MatrixXd {
32 return hat_func.EvalReferenceShapeFunctions(x_hat);
33 };
34
35 // returns evaluation of @f$ \lambda @f$ for at a given point on the reference
36 // triangle f is evaluated at the global coordinates
37 const auto f_tilde_hat = [&](Eigen::Vector2d x_hat) -> Eigen::MatrixXd {
38 Eigen::Vector3d x = geom->Global(x_hat);
39 return lambda_hat(x_hat) * f_(x);
40 };
41
42 // Compute the elements of the vector with given quadrature rule
43 Eigen::Vector3d element_vector = Eigen::Vector3d::Zero();
44 const Eigen::MatrixXd points = quadrule.Points();
45 const Eigen::VectorXd weights =
46 (geom->IntegrationElement(points).array() * quadrule.Weights().array())
47 .matrix();
48 for (lf::base::size_type n = 0; n < quadrule.NumPoints(); ++n) {
49 const Eigen::Vector3d f_eval = f_tilde_hat(points.col(n));
50 element_vector += weights[n] * f_eval;
51 }
52 return element_vector;
53}
