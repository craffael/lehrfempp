90 std::vector<Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>> operator()(
91 const lf::mesh::Entity& e, const Eigen::MatrixXd& local) const {
92 // Only triangles are supported
93 LF_VERIFY_MSG(e.RefEl() == lf::base::RefEl::kTria(),
94 "Unsupported cell type " << e.RefEl());
95
96 // Get the geometry of the entity
97 const auto* geom = e.Geometry();
98
99 // Compute the global vertex coordinates
100 Eigen::MatrixXd vertices = geom->Global(e.RefEl().NodeCoords());
101
102 // get the lambdas
103 const lf::uscalfe::FeLagrangeO1Tria<SCALAR> hat_func;
104
105 // The gradients are constant on the triangle
106 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> ref_grads =
107 hat_func.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
108 .transpose();
109
110 // The JacobianInverseGramian is constant on the triangle
111 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> J_inv_trans =
112 geom->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
113
114 // get the gradients
115 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> grad =
116 J_inv_trans * ref_grads;
117
118 // get lambda values
119 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> lambda =
120 hat_func.EvalReferenceShapeFunctions(local);
121
122 // correct for orientation
123 auto edgeOrientations = e.RelativeOrientations();
124
125 // define return vector
126 std::vector<Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>> vals(local.cols());
127
128 // get global indices of the edges
129 std::vector<lf::base::size_type> global_indices(3);
130 auto edges = e.SubEntities(1);
131 for (lf::base::size_type i = 0; i < 3; i++) {
132 global_indices[i] = mesh_->Index(*edges[i]);
133 }
134
135 // for each point evaluate the basis functions and add the
136 // values together
137 for (int i = 0; i < local.cols(); i++) {
138 Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> bs(grad.rows(), 3);
139 bs.col(0) = lambda(0, i) * grad.col(1) - lambda(1, i) * grad.col(0);
140 bs.col(1) = lambda(1, i) * grad.col(2) - lambda(2, i) * grad.col(1);
141 bs.col(2) = lambda(2, i) * grad.col(0) - lambda(0, i) * grad.col(2);
142
143 bs.col(0) *= lf::mesh::to_sign(edgeOrientations[0]);
144 bs.col(1) *= lf::mesh::to_sign(edgeOrientations[1]);
145 bs.col(2) *= lf::mesh::to_sign(edgeOrientations[2]);
146
147 vals[i] = mu_(global_indices[0]) * bs.col(0) +
148 mu_(global_indices[1]) * bs.col(1) +
149 mu_(global_indices[2]) * bs.col(2);
150 }
151
152 return vals;
153 }

155 private:
156 const Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>& mu_;
157 const std::shared_ptr<const lf::mesh::Mesh> mesh_;
