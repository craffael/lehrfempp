82 auto Power = [](double a, double b) -> double { return std::pow(a, b); };
83 auto Sin = [](double a) -> double { return std::sin(a); };

86 // righthandside for the zero and two form
87 auto f_zero = [&](const Eigen::Vector3d &x_vec) -> double {
88 // first scale to the circle
89 Eigen::Vector3d x_ = x_vec / x_vec.norm() * r;
90 double x = x_(0);
91 double y = x_(1);
92 double z = x_(2);
93
94 // autogenerated by mathematica
95 double ret =
96 Power(k, 2) * (Cos(z) + Sin(x) + Sin(y)) +
97 (2 * x * Cos(x) + 2 * y * Cos(y) + Power(x, 2) * Cos(z) +
98 Power(y, 2) * Cos(z) + Power(y, 2) * Sin(x) + Power(z, 2) * Sin(x) +
99 Power(x, 2) * Sin(y) + Power(z, 2) * Sin(y) - 2 * z * Sin(z)) /
100 (Power(x, 2) + Power(y, 2) + Power(z, 2));
101
102 return ret;
103 };
104
105 // righthandside for the one form
106 auto f_one = [&](const Eigen::Vector3d &x_vec) -> Eigen::VectorXd {
107 // first scale to the circle
108 Eigen::Vector3d x_ = x_vec / x_vec.norm() * r;
109 double x = x_(0);
110 double y = x_(1);
111 double z = x_(2);
112
113 // autogenerated by mathematica
114 Eigen::VectorXd ret(3);
115 ret << -((Power(k, 2) * Power(x, 2) * y) /
116 (Power(x, 2) + Power(y, 2) + Power(z, 2))) -
117 (-Power(x, 5) + 9 * Power(x, 4) * y +
118 Power(x, 3) * (3 * Power(y, 2) + 4 * y * z - Power(z, 2)) +
119 3 * Power(x, 2) * y * (Power(y, 2) + Power(z, 2)) -
120 2 * y * Power(Power(y, 2) + Power(z, 2), 2)) /
121 Power(Power(x, 2) + Power(y, 2) + Power(z, 2), 3),
122 Power(k, 2) * x *
123 (1 - Power(y, 2) / (Power(x, 2) + Power(y, 2) + Power(z, 2))) -
124 (-Power(x, 5) - Power(x, 4) * y + 3 * Power(x, 2) * Power(y, 3) +
125 Power(x, 3) * (3 * Power(y, 2) - 2 * y * z - 2 * Power(z, 2)) +
126 y * Power(z, 2) * (Power(y, 2) + Power(z, 2)) +
127 x * (8 * Power(y, 4) + 2 * Power(y, 3) * z +
128 3 * Power(y, 2) * Power(z, 2) - 2 * y * Power(z, 3) -
129 Power(z, 4))) /
130 Power(Power(x, 2) + Power(y, 2) + Power(z, 2), 3),
131 -((Power(k, 2) * x * y * z) /
132 (Power(x, 2) + Power(y, 2) + Power(z, 2))) +
133 (z * (-3 * Power(x, 2) * y * z + Power(x, 3) * (-8 * y + z) +
134 y * z * (Power(y, 2) + Power(z, 2)) +
135 x * (-8 * Power(y, 3) - 3 * Power(y, 2) * z -
136 12 * y * Power(z, 2) + Power(z, 3)))) /
137 Power(Power(x, 2) + Power(y, 2) + Power(z, 2), 3);
138 return ret;
139 };
