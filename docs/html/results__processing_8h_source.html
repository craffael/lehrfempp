96template <typename U_ZERO, typename U_ONE, typename U_TWO, typename SCALAR>
97void process_results(std::string name, ProblemSolutionWrapper<SCALAR> &results,
98 U_ZERO &u_zero, U_ONE &u_one, U_TWO &u_two, double &k) {
99 // create results direcotry
100 std::string main_dir = "results";
101 std::filesystem::create_directory(main_dir);
102
103 // each containing number of k values
104 int nk = results.k.size();
105 int nl = results.levels.size();
106
107 // prepare output
108 int table_width = 13;
109 int precision = 4;
110
111 // name the csv file accoring to min and max k
112 double min_k = results.k[0];
113 double max_k = results.k[nk - 1];
114
115 // create csv file
116 std::ofstream csv_file;
117 std::string csv_name = concat("result_", name, "_", min_k, "-", max_k);
118 std::replace(csv_name.begin(), csv_name.end(), '.', '_');
119 csv_file.open(concat(main_dir, "/", csv_name, ".csv"));
120 csv_file << "numCells,"
121 << "numEdges,"
122 << "numVerts,"
123 << "hMax";
124
125 // define square functions for norms
126 auto square_scalar = [](SCALAR a) -> double {
127 return std::abs(a) * std::abs(a);
128 };
129 auto square_vector =
130 [](Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> a) -> double {
131 return a.squaredNorm();
132 };
133
134 // define quadrule for norms
135 lf::quad::QuadRule qr = lf::quad::make_QuadRule(lf::base::RefEl::kTria(), 4);
136
137 // create solution matrix with levels on rows and ks in colums
138 Eigen::MatrixXd SupErrorZero = Eigen::MatrixXd::Zero(nl, nk);
139 Eigen::MatrixXd SupErrorOne = Eigen::MatrixXd::Zero(nl, nk);
140 Eigen::MatrixXd SupErrorTwo = Eigen::MatrixXd::Zero(nl, nk);
141 Eigen::MatrixXd L2ErrorZero = Eigen::MatrixXd::Zero(nl, nk);
142 Eigen::MatrixXd L2ErrorOne = Eigen::MatrixXd::Zero(nl, nk);
143 Eigen::MatrixXd L2ErrorTwo = Eigen::MatrixXd::Zero(nl, nk);
144 Eigen::VectorXd hMax = Eigen::VectorXd::Zero(nl);
145
146 // tabulate every k value
147 for (int i = 0; i < nk; i++) {
148 csv_file << ",SupErrorZero_" << results.k[i] << ",RateSupErrorZero_"
149 << results.k[i] << ",SupErrorOne_" << results.k[i]
150 << ",RateSupErrorOne_" << results.k[i] << ",SupErrorTwo_"
151 << results.k[i] << ",RateSupErrorTwo_" << results.k[i]
152 << ",L2ErrorZero_" << results.k[i] << ",RateL2ErrorZero_"
153 << results.k[i] << ",L2ErrorOne_" << results.k[i]
154 << ",RateL2ErrorOne_" << results.k[i] << ",L2ErrorTwo_"
155 << results.k[i] << ",RateL2ErrorTwo_" << results.k[i];
156
157 // create direcotries for each k
158 std::string kstr = concat(results.k[i]);
159 std::replace(kstr.begin(), kstr.end(), '.', '_');
160 std::filesystem::create_directory(concat(main_dir, "/k_", kstr));
161 }
162 csv_file << std::endl;
163
164 // write the k values to the csv file
165 csv_file << ","
166 << ","
167 << ",";
168 for (int i = 0; i < nk; i++) {
169 csv_file << "," << results.k[i] << "," << results.k[i] << ","
170 << results.k[i] << "," << results.k[i] << "," << results.k[i]
171 << "," << results.k[i] << "," << results.k[i] << ","
172 << results.k[i] << "," << results.k[i] << "," << results.k[i]
173 << "," << results.k[i] << "," << results.k[i];
174 }
175 csv_file << std::endl;
176
177 // loop over all levels contained in the solution
178 for (lf::base::size_type lvl = 0; lvl < nl; ++lvl) {
179 // create mesh Functions for solutions the level
180 auto &sol_mesh = results.mesh[lvl];
181 auto &sol_mus = results.solutions[lvl];
182
183 // compute meshwidth
184 for (const lf::mesh::Entity *e : sol_mesh->Entities(1)) {
185 double h = lf::geometry::Volume(*(e->Geometry()));
186 if (h > hMax(lvl)) hMax(lvl) = h;
187 }
188
189 // print level and mesh informations
190 csv_file << sol_mesh->NumEntities(0) << "," << sol_mesh->NumEntities(1)
191 << "," << sol_mesh->NumEntities(2) << "," << hMax(lvl);
192
193 // compute the errors for each k
194 for (int ik = 0; ik < nk; ik++) {
195 std::string kstr = concat(results.k[ik]);
196 std::replace(kstr.begin(), kstr.end(), '.', '_');
197 std::string folder = concat(main_dir, "/k_", kstr);
198
199 projects::hldo_sphere::post_processing::MeshFunctionWhitneyZero mf_zero(
200 sol_mus.mu_zero[ik], sol_mesh);
201 projects::hldo_sphere::post_processing::MeshFunctionWhitneyOne mf_one(
202 sol_mus.mu_one[ik], sol_mesh);
203 projects::hldo_sphere::post_processing::MeshFunctionWhitneyTwo mf_two(
204 sol_mus.mu_two[ik], sol_mesh);
205
206 // set k and therefore change function since the k is used in the
207 // funcitons
208 k = results.k[ik];
209 lf::mesh::utils::MeshFunctionGlobal<U_ZERO> mf_zero_ana(u_zero);
210 lf::mesh::utils::MeshFunctionGlobal<U_ONE> mf_one_ana(u_one);
211 lf::mesh::utils::MeshFunctionGlobal<U_TWO> mf_two_ana(u_two);
212
213 // Compute the error of the solutions
214 auto mf_diff_zero = mf_zero - mf_zero_ana;
215 auto mf_diff_one = mf_one - mf_one_ana;
216 auto mf_diff_two = mf_two - mf_two_ana;
217
218 // Perform post processing on the data
219 lf::io::VtkWriter writer(sol_mesh, concat(folder, "/result_", name, "_",
220 lvl, "_k_", kstr, ".vtk"));
221
222 // get error for zero form
223 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
224 L2_zero = projects::hldo_sphere::post_processing::L2norm(
225 sol_mesh, mf_diff_zero, square_scalar, qr);
226 L2ErrorZero(lvl, ik) = std::get<0>(L2_zero);
227 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_zero =
228 std::get<1>(L2_zero);
229 writer.WriteCellData(concat("mf_zero_diff_", lvl), data_set_error_zero);
230
231 // get error for one form
232 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
233 L2_one = projects::hldo_sphere::post_processing::L2norm(
234 sol_mesh, mf_diff_one, square_vector, qr);
235 L2ErrorOne(lvl, ik) = std::get<0>(L2_one);
236 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_one =
237 std::get<1>(L2_one);
238 writer.WriteCellData(concat("mf_one_diff_", lvl), data_set_error_one);
239
240 // get error for two form
241 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
242 L2_two = projects::hldo_sphere::post_processing::L2norm(
243 sol_mesh, mf_diff_two, square_scalar, qr);
244 L2ErrorTwo(lvl, ik) = std::get<0>(L2_two);
245 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_two =
246 std::get<1>(L2_two);
247 writer.WriteCellData(concat("mf_two_diff_", lvl), data_set_error_two);
248
249 // get sup error for zero form
250 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
251 sup_zero = projects::hldo_sphere::post_processing::SupNorm(
252 sol_mesh, mf_diff_zero, square_scalar, qr);
253 SupErrorZero(lvl, ik) = std::get<0>(sup_zero);
254 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_sup_zero =
255 std::get<1>(sup_zero);
256 writer.WriteCellData(concat("mf_zero_diff_sup_", lvl),
257 data_set_error_sup_zero);
258
259 // get error for one form
260 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
261 sup_one = projects::hldo_sphere::post_processing::SupNorm(
262 sol_mesh, mf_diff_one, square_vector, qr);
263 SupErrorOne(lvl, ik) = std::get<0>(sup_one);
264 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_sup_one =
265 std::get<1>(sup_one);
266 writer.WriteCellData(concat("mf_one_diff_sup_", lvl),
267 data_set_error_sup_one);
268
269 // get error for two form
270 const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
271 sup_two = projects::hldo_sphere::post_processing::SupNorm(
272 sol_mesh, mf_diff_two, square_scalar, qr);
273 SupErrorTwo(lvl, ik) = std::get<0>(sup_two);
274 const lf::mesh::utils::CodimMeshDataSet<double> data_set_error_sup_two =
275 std::get<1>(sup_two);
276 writer.WriteCellData(concat("mf_two_diff_sup_", lvl),
277 data_set_error_sup_two);
278
279 double l2RateZero = 0;
280 double l2RateOne = 0;
281 double l2RateTwo = 0;
282 double supRateZero = 0;
283 double supRateOne = 0;
284 double supRateTwo = 0;
285
286 // we can only compute order form the second level
287 if (lvl > 0) {
288 l2RateZero =
289 (log(L2ErrorZero(lvl, ik)) - log(L2ErrorZero(lvl - 1, ik))) /
290 (hMax(lvl) - hMax(lvl - 1));
291 l2RateOne = (log(L2ErrorOne(lvl, ik)) - log(L2ErrorOne(lvl - 1, ik))) /
292 (hMax(lvl) - hMax(lvl - 1));
293 l2RateTwo = (log(L2ErrorTwo(lvl, ik)) - log(L2ErrorTwo(lvl - 1, ik))) /
294 (hMax(lvl) - hMax(lvl - 1));
295 supRateZero =
296 (log(SupErrorZero(lvl, ik)) - log(SupErrorZero(lvl - 1, ik))) /
297 (hMax(lvl) - hMax(lvl - 1));
298 supRateOne =
299 (log(SupErrorOne(lvl, ik)) - log(SupErrorOne(lvl - 1, ik))) /
300 (hMax(lvl) - hMax(lvl - 1));
301 supRateTwo =
302 (log(SupErrorTwo(lvl, ik)) - log(SupErrorTwo(lvl - 1, ik))) /
303 (hMax(lvl) - hMax(lvl - 1));
304 }
305
306 /******************************
307 * append solution of the current k in the outputs
308 ******************************/
309
310 csv_file << "," << SupErrorZero(lvl, ik) << "," << supRateZero << ","
311 << SupErrorOne(lvl, ik) << "," << supRateOne << ","
312 << SupErrorTwo(lvl, ik) << "," << supRateTwo << ","
313 << L2ErrorZero(lvl, ik) << "," << l2RateZero << ","
314 << L2ErrorOne(lvl, ik) << "," << l2RateOne << ","
315 << L2ErrorTwo(lvl, ik) << "," << l2RateTwo;
316
317 } // end loop over k
318
319 csv_file << "\n";
320 } // end loop over levels
321
322 // close csv file
323 csv_file.close();
324}
