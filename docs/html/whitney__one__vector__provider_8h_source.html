72 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> Eval(
73 const lf::mesh::Entity &entity) const {
74 LF_ASSERT_MSG(entity.RefEl() == lf::base::RefEl::kTria(),
75 "Only Triangles are supported no " << entity.RefEl());
76
77 const auto *const geom = entity.Geometry();
78
79 // Compute the global vertex coordinates
80 Eigen::MatrixXd vertices = geom->Global(entity.RefEl().NodeCoords());
81
82 // define quad rule with sufficiantly high degree
83 lf::quad::QuadRule quadrule{lf::quad::make_TriaQR_EdgeMidpointRule()};
84
85 // Construct the whitney one form basis functions
86 const lf::uscalfe::FeLagrangeO1Tria<SCALAR> hat_func;
87 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> ref_grads =
88 hat_func.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
89 .transpose();
90 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> J_inv_trans =
91 geom->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
92 const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> grads =
93 J_inv_trans * ref_grads;
94
95 // get edge orientations
96 auto edgeOrientations = entity.RelativeOrientations();
97 Eigen::Vector3d s;
98 s(0) = lf::mesh::to_sign(edgeOrientations[0]);
99 s(1) = lf::mesh::to_sign(edgeOrientations[1]);
100 s(2) = lf::mesh::to_sign(edgeOrientations[2]);
101
102 const auto b_hat = [&](Eigen::Vector2d x_hat)
103 -> Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> {
104 Eigen::Matrix<SCALAR, 3, 3> b_hat;
105 Eigen::Matrix<SCALAR, 3, 1> lambda_hat =
106 hat_func.EvalReferenceShapeFunctions(x_hat);
107 for (int i = 0; i < 3; i++) {
108 int ip1 = (i + 1) % 3;
109
110 b_hat.col(i) = s(i) * (lambda_hat(i) * grads.col(ip1) -
111 lambda_hat(ip1) * grads.col(i));
112 }
113
114 return b_hat;
115 };
116
117 // returns evaluation of @f$\textbf{f} * \textbf{b}@f$ for at a given point
118 // on the reference triangle f is evaluated at the global coordinates of the
119 // triangle radially projected on the sphere
120 const auto f_tilde_hat =
121 [&](Eigen::Vector2d x_hat) -> Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> {
122 Eigen::Vector3d x = geom->Global(x_hat);
123 Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> b_hat_out =
124 b_hat(x_hat);
125 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> result =
126 b_hat(x_hat).transpose() * f_(x);
127 return result;
128 };
129
130 // Compute the elements of the vector with given quadrature rule
131 Eigen::Matrix<SCALAR, 3, 1> element_vector;
132 element_vector.setZero();
133 const Eigen::MatrixXd points = quadrule.Points();
134 const Eigen::VectorXd weights =
135 (geom->IntegrationElement(points).array() * quadrule.Weights().array())
136 .matrix();
137 for (lf::base::size_type n = 0; n < quadrule.NumPoints(); ++n) {
138 const Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> f_eval =
139 f_tilde_hat(points.col(n));
140 element_vector += weights[n] * f_eval;
141 }
142 return element_vector;
143 }
