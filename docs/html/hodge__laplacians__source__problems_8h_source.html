90 void Compute() {
91 //*****************
92 // Whitney 0 form
93 //*****************
94
95 // get the Hodge laplacian for the zero form and take the negative
96 projects::hldo_sphere::discretization::WhitneyZeroHodgeLaplace zero_builder;
97 zero_builder.SetMesh(mesh_p_);
98 zero_builder.SetLoadFunction(f0_);
99 zero_builder.Compute();
100 lf::assemble::COOMatrix<double> coo_mat_zero_pos =
101 zero_builder.GetGalerkinMatrix();
102 lf::assemble::COOMatrix<double> coo_mat_zero(coo_mat_zero_pos.rows(),
103 coo_mat_zero_pos.cols());
104 coo_mat_zero.setZero();
105
106 for (Eigen::Triplet<double> triplet : coo_mat_zero_pos.triplets()) {
107 int col = triplet.col();
108 int row = triplet.row();
109 // we need the negative laplacian
110 double val = triplet.value();
111 coo_mat_zero.AddToEntry(row, col, val);
112 }
113
114 // create mass Matrix
115
116 projects::hldo_sphere::assemble::MassMatrixProvider
117 mass_matrix_provider_zero;
118
119 const lf::assemble::DofHandler &dof_handler_zero =
120 lf::assemble::UniformFEDofHandler(mesh_p_,
121 {{lf::base::RefEl::kPoint(), 1}});
122 const lf::assemble::size_type n_dofs_zero(dof_handler_zero.NumDofs());
123
124 // prepare matrix
125 lf::assemble::COOMatrix<double> coo_mass_mat_zero(n_dofs_zero, n_dofs_zero);
126 coo_mass_mat_zero.setZero();
127
128 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
129 0, dof_handler_zero, dof_handler_zero, mass_matrix_provider_zero,
130 coo_mass_mat_zero);
131
132 // merge whitney zero for with k^2 * mass Matrix
133 for (Eigen::Triplet<double> triplet : coo_mass_mat_zero.triplets()) {
134 int row = triplet.row();
135 int col = triplet.col();
136 double val = k_ * k_ * triplet.value();
137 coo_mat_zero.AddToEntry(row, col, val);
138 };
139
140 coo_matrix_[0] = coo_mat_zero;
141
142 // get righthandside
143 Eigen::Matrix<double, Eigen::Dynamic, 1> phi_zero =
144 zero_builder.GetLoadVector();
145 phi_[0] = phi_zero;
146
147 //*****************
148 // Whitney 1 form
149 //*****************
150
151 // get the Hodge laplacian for the one form and take the negative
152 projects::hldo_sphere::discretization::WhitneyOneHodgeLaplace one_builder;
153 one_builder.SetMesh(mesh_p_);
154 one_builder.SetLoadFunction(f1_);
155 one_builder.Compute();
156 lf::assemble::COOMatrix<double> coo_mat_one_pos =
157 one_builder.GetGalerkinMatrix();
158 lf::assemble::COOMatrix<double> coo_mat_one(coo_mat_one_pos.rows(),
159 coo_mat_one_pos.cols());
160 coo_mat_one.setZero();
161
162 for (Eigen::Triplet<double> triplet : coo_mat_one_pos.triplets()) {
163 int col = triplet.col();
164 int row = triplet.row();
165 double val = triplet.value();
166 coo_mat_one.AddToEntry(row, col, val);
167 }
168
169 // create mass Matrix
170 projects::hldo_sphere::assemble::WhitneyOneMassMatrixProvider
171 mass_matrix_provider_one;
172 const lf::assemble::DofHandler &dof_handler_one =
173 lf::assemble::UniformFEDofHandler(mesh_p_,
174 {{lf::base::RefEl::kSegment(), 1}});
175 const lf::assemble::size_type n_dofs_one(dof_handler_one.NumDofs());
176 lf::assemble::COOMatrix<double> coo_mass_mat_one(n_dofs_one, n_dofs_one);
177 coo_mass_mat_one.setZero();
178
179 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
180 0, dof_handler_one, dof_handler_one, mass_matrix_provider_one,
181 coo_mass_mat_one);
182
183 // merge whitney one for with k^2 * mass Matrix
184 for (Eigen::Triplet<double> triplet : coo_mass_mat_one.triplets()) {
185 int col = triplet.col();
186 int row = triplet.row();
187 double val = k_ * k_ * triplet.value();
188 coo_mat_one.AddToEntry(row, col, val);
189 }
190 coo_matrix_[1] = coo_mat_one;
191
192 // get righthandside
193 Eigen::Matrix<double, Eigen::Dynamic, 1> phi_one =
194 one_builder.GetLoadVector();
195 phi_[1] = phi_one;
196
197 //*****************
198 // Whitney 2 form
199 //*****************
200
201 // get the Hodge laplacian for the one form and take the negative
202 projects::hldo_sphere::discretization::WhitneyTwoHodgeLaplace two_builder;
203 two_builder.SetMesh(mesh_p_);
204 two_builder.SetLoadFunction(f2_);
205 two_builder.Compute();
206
207 lf::assemble::COOMatrix<double> coo_mat_two_pos =
208 two_builder.GetGalerkinMatrix();
209 lf::assemble::COOMatrix<double> coo_mat_two(coo_mat_two_pos.rows(),
210 coo_mat_two_pos.cols());
211
212 coo_mat_two.setZero();
213 for (Eigen::Triplet<double> triplet : coo_mat_two_pos.triplets()) {
214 int col = triplet.col();
215 int row = triplet.row();
216 double val = triplet.value();
217 coo_mat_two.AddToEntry(row, col, val);
218 }
219
220 // create mass Matrix
221 projects::hldo_sphere::assemble::WhitneyTwoMassMatrixProvider
222 mass_matrix_provider_two;
223 const lf::assemble::DofHandler &dof_handler_two =
224 lf::assemble::UniformFEDofHandler(mesh_p_,
225 {{lf::base::RefEl::kTria(), 1}});
226 const lf::assemble::size_type n_dofs_two(dof_handler_two.NumDofs());
227 lf::assemble::COOMatrix<double> coo_mass_mat_two(n_dofs_two, n_dofs_two);
228 coo_mass_mat_two.setZero();
229 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<double>>(
230 0, dof_handler_two, dof_handler_two, mass_matrix_provider_two,
231 coo_mass_mat_two);
232
233 // merge whitney two for with k^2 * mass Matrix
234 for (Eigen::Triplet<double> triplet : coo_mass_mat_two.triplets()) {
235 // n_dofs_one contains the number of edges and hence the
236 // dimension of A_{11}
237 int col = triplet.col() + n_dofs_one;
238 int row = triplet.row() + n_dofs_one;
239 double val = k_ * k_ * triplet.value();
240 coo_mat_two.AddToEntry(row, col, val);
241 }
242
243 coo_matrix_[2] = coo_mat_two;
244
245 // get righthandside
246 Eigen::Matrix<double, Eigen::Dynamic, 1> phi_two =
247 two_builder.GetLoadVector();
248 phi_[2] = phi_two;
249 }

268 void Solve() {
269 Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
270 for (int l = 0; l < 3; l++) {
271 Eigen::SparseMatrix<double> sparse_mat = coo_matrix_[l].makeSparse();
272 solver.compute(sparse_mat);
273 if (solver.info() != Eigen::Success) {
274 throw std::runtime_error("Could not decompose the matrix");
275 }
276
277 mu_[l] = solver.solve(phi_[l]);
278 }
279 }
