17void PrintInfo(std::ostream &o, const lf::mesh::Mesh &mesh, int ctrl) {
18 using dim_t = lf::base::dim_t;
19 using size_type = lf::base::size_type;
20
21 const dim_t dim_mesh = mesh.DimMesh();
22 const dim_t dim_world = mesh.DimWorld();
23 o << "Mesh of dimension " << static_cast<int>(dim_mesh)
24 << ", ambient dimension " << static_cast<int>(dim_world) << std::endl;
25
26 if (ctrl > 10) {
27 // Loop over codimensions
28
29 for (int co_dim = dim_mesh; co_dim >= 0; co_dim--) {
30 const size_type no_ent = mesh.NumEntities(co_dim);
31 o << "Co-dimension " << co_dim << ": " << no_ent << " entities"
32 << std::endl;
33
34 // Loop over entities
35 for (const Entity *e : mesh.Entities(co_dim)) {
36 size_type e_idx = mesh.Index(*e);
37 dim_t e_codim = e->Codim();
38 const geometry::Geometry *e_geo_ptr = e->Geometry();
39 lf::base::RefEl e_refel = e->RefEl();
40
41 LF_VERIFY_MSG(e_geo_ptr,
42 co_dim << "-entity " << e_idx << ": missing geometry");
43 LF_VERIFY_MSG(e_geo_ptr->DimLocal() == dim_mesh - co_dim,
44 co_dim << "-entity " << e_idx << ": wrong dimension");
45 LF_VERIFY_MSG(e_geo_ptr->RefEl() == e_refel,
46 co_dim << "-entity " << e_idx << ": refEl mismatch");
47 LF_VERIFY_MSG(e_codim == co_dim, co_dim << "-entity " << e_idx
48 << " co-dimension mismatch");
49 const Eigen::MatrixXd &ref_el_corners(e_refel.NodeCoords());
50 o << "entity " << e_idx << " (" << e_refel << "): ";
51
52 // Loop over local co-dimensions
53 if (ctrl > 90) {
54 for (int l = 1; l <= dim_mesh - co_dim; l++) {
55 o << "rel codim-" << l << " subent: [";
56 // Fetch subentities of co-dimension l
57 auto sub_ent_range = e->SubEntities(l);
58 for (const Entity *sub_ent : sub_ent_range) {
59 o << mesh.Index(*sub_ent) << ' ';
60 }
61 o << "]";
62 }
63 }
64 if (ctrl > 50) {
65 o << std::endl << e_geo_ptr->Global(ref_el_corners);
66 }
67 o << std::endl;
68 } // end loop over entities
69 } // end loop over co-dimensions
70 } // end printinfo_ctrl > 10
71} // end function PrintInfo

73// Print function for Entity object
74void PrintInfo(std::ostream &stream, const lf::mesh::Entity &e,
75 int output_ctrl) {
76 // Topological type of entity
77 lf::base::RefEl e_ref_el = e.RefEl();
78 int dim_ref_el = e_ref_el.Dimension();
79 // Geometry of entity and coordinates
80 const geometry::Geometry *e_geo_ptr = e.Geometry();
81 LF_ASSERT_MSG(e_geo_ptr != nullptr, "Missing geometry information!");
82
83 stream << "Entity " << e_ref_el << "/" << typeid(*e_geo_ptr).name()
84 << std::endl;
85
86 if (output_ctrl > 10) {
87 stream << "Dimension: " << dim_ref_el << std::endl;
88
89 const Eigen::MatrixXd &ref_el_corners(e_ref_el.NodeCoords());
90
91 // Loop over codimensions
92 for (int co_dim = dim_ref_el; co_dim > 0; co_dim--) {
93 int num_sub_ent = e_ref_el.NumSubEntities(co_dim);
94 stream << std::endl
95 << "Codimension " << co_dim << ": " << num_sub_ent
96 << " sub-entities" << std::endl;
97
98 if (output_ctrl > 50) {
99 int sub_ent_num = 0;
100 // Loop over subentities
101 for (const Entity *sub_ent : e.SubEntities(co_dim)) {
102 lf::base::RefEl sub_ent_refel = sub_ent->RefEl();
103 stream << "* Subentity " << sub_ent_num << " (" << sub_ent_refel
104 << ")" << std::endl;
105
106 if (output_ctrl > 90) {
107 // Print coordinates
108 stream << e_geo_ptr->Global(ref_el_corners).col(sub_ent_num)
109 << std::endl;
110 }
111
112 sub_ent_num += 1;
113
114 } // loop sub-ent
115 } // if output ctrl
116 } // loop codim
117
118 if (e_ref_el == lf::base::RefEl::kPoint() && output_ctrl > 90) {
119 stream << e_geo_ptr->Global(ref_el_corners) << std::endl;
120 }
121
122 stream << "-----------------------" << std::endl;
123
124 } // if
125} // PrintInfo
