67 // Compute the global vertex coordinates
68 Eigen::MatrixXd vertices = geom->Global(entity.RefEl().NodeCoords());

70 // define quad rule with sufficiantly high degree since the
71 // baricentric coordinate functions have degree 1
72 lf::quad::QuadRule quadrule{lf::quad::make_TriaQR_EdgeMidpointRule()};
73
74 // Compute the elements of the vector with given quadrature rule
75 const Eigen::MatrixXd points = geom->Global(quadrule.Points());
76 const Eigen::VectorXd weights =
77 (geom->IntegrationElement(quadrule.Points()).array() *
78 quadrule.Weights().array())
79 .matrix();
80
81 SCALAR sum = 0;
82 for (lf::base::size_type n = 0; n < quadrule.NumPoints(); ++n) {
83 Eigen::VectorXd x = points.col(n);
84 sum += weights[n] * f_(x);
85 }
86
87 Eigen::Matrix<SCALAR, 1, 1> element_vector;
88 element_vector(0) = sum;
89
