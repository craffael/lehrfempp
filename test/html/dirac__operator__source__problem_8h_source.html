52 public:
63 DiracOperatorSourceProblem()
64 : coo_matrix_(lf::assemble::COOMatrix<SCALAR>(1, 1)) {
65 // create mesh factory for basic mesh
66 std::unique_ptr<lf::mesh::MeshFactory> factory =
67 std::make_unique<lf::mesh::hybrid2d::MeshFactory>(3);
68
69 std::shared_ptr<projects::hldo_sphere::mesh::SphereTriagMeshBuilder>
70 sphere = std::make_shared<
71 projects::hldo_sphere::mesh::SphereTriagMeshBuilder>(
72 std::move(factory));
73 sphere->setRefinementLevel(0);
74 sphere->setRadius(1);
75
76 mesh_p_ = sphere->Build();
77
78 k_ = 1.;
79
80 phi_ = Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>(1);
81
82 mu_ = Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>(1);
83
84 // create basic function
85 auto f_0 = [](Eigen::Matrix<double, 3, 1> x) -> SCALAR { return 0; };
86 auto f_1 =
87 [](Eigen::Matrix<double, 3, 1> x) -> Eigen::Matrix<SCALAR, 3, 1> {
88 return Eigen::Matrix<SCALAR, 3, 1>::Zero();
89 };
90 auto f_2 = [](Eigen::Matrix<double, 3, 1> x) -> SCALAR { return 0; };
91 f0_ = f_0;
92 f1_ = f_1;
93 f2_ = f_2;
94 }

107 void Compute() {
108 // get Dirac Operator Matrix
109 projects::hldo_sphere::discretization::DiracOperator dirac_operator;
110 dirac_operator.SetLoadFunctions(f0_, f1_, f2_);
111 dirac_operator.SetMesh(mesh_p_);
112 dirac_operator.Compute();
113
114 lf::assemble::COOMatrix<SCALAR> coo_mat =
115 dirac_operator.GetGalerkinMatrix();
116
117 // get righthandside vector
118 Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> phi =
119 dirac_operator.GetLoadVector();
120 phi_ = phi;
121
122 //**********************
123 // create mass matrices
124 //**********************
125
126 // Zero form mass matrix
127 projects::hldo_sphere::assemble::MassMatrixProvider
128 mass_matrix_provider_zero;
129
130 const lf::assemble::DofHandler &dof_handler_zero =
131 lf::assemble::UniformFEDofHandler(mesh_p_,
132 {{lf::base::RefEl::kPoint(), 1}});
133 const lf::assemble::size_type n_dofs_zero(dof_handler_zero.NumDofs());
134
135 lf::assemble::COOMatrix<SCALAR> coo_mass_mat_zero(n_dofs_zero, n_dofs_zero);
136 coo_mass_mat_zero.setZero();
137
138 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<SCALAR>>(
139 0, dof_handler_zero, dof_handler_zero, mass_matrix_provider_zero,
140 coo_mass_mat_zero);
141
142 for (Eigen::Triplet<SCALAR> triplet : coo_mass_mat_zero.triplets()) {
143 int col = triplet.col();
144 int row = triplet.row();
145 SCALAR val = std::complex<double>(0., 1.) * k_ * triplet.value();
146 coo_mat.AddToEntry(row, col, val);
147 };
148
149 // one form mass matrix
150 projects::hldo_sphere::assemble::WhitneyOneMassMatrixProvider
151 mass_matrix_provider_one;
152 const lf::assemble::DofHandler &dof_handler_one =
153 lf::assemble::UniformFEDofHandler(mesh_p_,
154 {{lf::base::RefEl::kSegment(), 1}});
155 const lf::assemble::size_type n_dofs_one(dof_handler_one.NumDofs());
156 lf::assemble::COOMatrix<SCALAR> coo_mass_mat_one(n_dofs_one, n_dofs_one);
157 coo_mass_mat_one.setZero();
158 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<SCALAR>>(
159 0, dof_handler_one, dof_handler_one, mass_matrix_provider_one,
160 coo_mass_mat_one);
161
162 for (Eigen::Triplet<SCALAR> triplet : coo_mass_mat_one.triplets()) {
163 int col = triplet.col() + n_dofs_zero;
164 int row = triplet.row() + n_dofs_zero;
165 SCALAR val = std::complex<double>(0., 1.) * k_ * triplet.value();
166 coo_mat.AddToEntry(row, col, val);
167 }
168
169 // two form mass matrix
170 projects::hldo_sphere::assemble::WhitneyTwoMassMatrixProvider
171 mass_matrix_provider_two;
172 const lf::assemble::DofHandler &dof_handler_two =
173 lf::assemble::UniformFEDofHandler(mesh_p_,
174 {{lf::base::RefEl::kTria(), 1}});
175 const lf::assemble::size_type n_dofs_two(dof_handler_two.NumDofs());
176 lf::assemble::COOMatrix<SCALAR> coo_mass_mat_two(n_dofs_two, n_dofs_two);
177 coo_mass_mat_two.setZero();
178 lf::assemble::AssembleMatrixLocally<lf::assemble::COOMatrix<SCALAR>>(
179 0, dof_handler_two, dof_handler_two, mass_matrix_provider_two,
180 coo_mass_mat_two);
181
182 for (Eigen::Triplet<SCALAR> triplet : coo_mass_mat_two.triplets()) {
183 // n_dofs_one contains the number of edges and hence the
184 // dimension of A_{11}
185 int col = triplet.col() + n_dofs_zero + n_dofs_one;
186 int row = triplet.row() + n_dofs_zero + n_dofs_one;
187 SCALAR val = std::complex<double>(0., 1.) * k_ * triplet.value();
188 coo_mat.AddToEntry(row, col, val);
189 }
190
191 coo_matrix_ = coo_mat;
192 }

211 void Solve() {
212 Eigen::SparseLU<Eigen::SparseMatrix<SCALAR>> solver;
213 Eigen::SparseMatrix<SCALAR> sparse_mat = coo_matrix_.makeSparse();
214 sparse_mat.makeCompressed();
215 solver.analyzePattern(sparse_mat);
216 solver.factorize(sparse_mat);
217 if (solver.info() != Eigen::Success) {
218 throw std::runtime_error("Could not decompose the matrix");
219 }
220
221 mu_ = solver.solve(phi_);
222 }

320 Eigen::Matrix<double, Eigen::Dynamic, 1> GetMu(int index) {
321 LF_ASSERT_MSG(index < 3 && index >= 0,
322 "Index must be in {0,1,2}, given " << index);
323
324 Eigen::Vector3d n;
325 Eigen::Vector3d s;
326 n(0) = mesh_p_->NumEntities(2);
327 s(0) = 0;
328 n(1) = mesh_p_->NumEntities(1);
329 s(1) = n(0);
330 n(2) = mesh_p_->NumEntities(0);
331 s(2) = s(1) + n(1);
332 return mu_.segment(s(index), n(index)).real();
333 }
