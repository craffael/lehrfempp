22  // Each square is split into two triangles
23  const unsigned no_of_cells = 2 * nx * ny;
24  const unsigned no_of_edges = nx * ny + (nx + 1) * ny + nx * (ny + 1);
25  const unsigned no_of_vertices = (nx + 1) * (ny + 1);

31  if (no_of_cells == 0) {
32  return nullptr;
33  }
34  // define rectangle; return if none
35  const double x_size = top_right_corner_[0] - bottom_left_corner_[0];
36  const double y_size = top_right_corner_[1] - bottom_left_corner_[1];
37  if ((x_size <= 0.0) || (y_size <= 0.0)) {
38  return nullptr;
39  }
40  // meshwidths
41  const double hx = x_size / nx;
42  const double hy = y_size / ny;
43 
44  // Initialize vertices
45  std::vector<size_type> v_idx(no_of_vertices);
46  int node_cnt = 0; // index of current vertex: lexicographic numbering
47  for (size_type j = 0; j <= ny; ++j) {
48  for (size_type i = 0; i <= nx; ++i, ++node_cnt) {
49  // Tensor-product node locations
50  coord_t node_coord(2);
51  node_coord << bottom_left_corner_[0] + i * hx,
52  bottom_left_corner_[1] + j * hy;
53  // Diagnostics
54  SPDLOG_LOGGER_TRACE(Logger(), "Adding vertex {}: {}", node_cnt,
55  node_coord.transpose());
56 
57  // Enlist vertex
58  v_idx[node_cnt] = mesh_factory_->AddPoint(node_coord);
59  }
60  }

63  // Just in case of index permutations
64  std::vector<size_type> e_idx(no_of_edges);
65  int edge_cnt = 0; // index of current edge
66  // First horizontal edges
67  for (size_type i = 0; i < nx; ++i) {
68  for (size_type j = 0; j <= ny; ++j, ++edge_cnt) {
69  // Indices of the two endpoints of the edge
70  auto first_endpoint_idx = v_idx[VertexIndex(i, j)];
71  auto second_endpoint_idx = v_idx[VertexIndex(i + 1, j)];
72  // Diagnostics
73  SPDLOG_LOGGER_TRACE(Logger(), "horizontal edge {}: {} <-> {}", edge_cnt,
74  first_endpoint_idx, second_endpoint_idx);
75 
76  std::vector<size_type> nodes_index_list{first_endpoint_idx,
77  second_endpoint_idx};
78  // Coordinates of endpoints a columns of a 2x2 matrix
79  Eigen::Matrix<double, Eigen::Dynamic, 2> edge_geo(2, 2);
80  edge_geo << bottom_left_corner_[0] + i * hx,
81  bottom_left_corner_[0] + (i + 1) * hx,
82  bottom_left_corner_[1] + j * hy, bottom_left_corner_[1] + j * hy;
83  e_idx[edge_cnt] = mesh_factory_->AddEntity(
84  lf::base::RefEl::kSegment(), nodes_index_list,
85  std::make_unique<geometry::SegmentO1>(edge_geo));
86  }
87  }

90  for (size_type j = 0; j < ny; ++j, ++edge_cnt) {
91  // Indices of the two endpoints of the edge
92  const size_type first_endpoint_idx = v_idx[VertexIndex(i, j)];
93  const size_type second_endpoint_idx = v_idx[VertexIndex(i, j + 1)];
94  // Diagnostics
95  SPDLOG_LOGGER_TRACE(Logger(), "vertical edge {}: {} <-> {}", edge_cnt,
96  first_endpoint_idx, second_endpoint_idx);

103  bottom_left_corner_[0] + i * hx, bottom_left_corner_[1] + j * hy,
104  bottom_left_corner_[1] + (j + 1) * hy;
105  e_idx[edge_cnt] = mesh_factory_->AddEntity(
106  lf::base::RefEl::kSegment(), nodes_index_list,
107  std::make_unique<geometry::SegmentO1>(edge_geo));
108  }
109  }
110  // Then the skew edges (diagonals of squares)
111  for (size_type i = 0; i < nx; ++i) {
112  for (size_type j = 0; j < ny; ++j, ++edge_cnt) {
113  // Indices of the two endpoints of the edge
114  const size_type first_endpoint_idx = v_idx[VertexIndex(i, j)];
115  const size_type second_endpoint_idx = v_idx[VertexIndex(i + 1, j + 1)];
116  // Diagnostics
117  SPDLOG_LOGGER_TRACE(Logger(), "diagonal edge {}: {} <-> {}", edge_cnt,
118  first_endpoint_idx, second_endpoint_idx);
119 
120  std::vector<size_type> nodes_index_list{first_endpoint_idx,
121  second_endpoint_idx};
122  // Coordinates of endpoints a columns of a 2x2 matrix
123  Eigen::Matrix<double, Eigen::Dynamic, 2> edge_geo(2, 2);
124  edge_geo << bottom_left_corner_[0] + i * hx,
125  bottom_left_corner_[0] + (i + 1) * hx,
126  bottom_left_corner_[1] + j * hy,
127  bottom_left_corner_[1] + (j + 1) * hy;
128  e_idx[edge_cnt] = mesh_factory_->AddEntity(
129  lf::base::RefEl::kSegment(), nodes_index_list,
130  std::make_unique<geometry::SegmentO1>(edge_geo));
131  }
132  }
133

135  // Index remapping for triangles
136  std::vector<size_type> t_idx(no_of_cells);
137 
138  size_type tria_cnt = 0; // index of current triangle
139  for (size_type i = 0; i < nx; ++i) {
140  for (size_type j = 0; j < ny; ++j, tria_cnt += 2) {
141  // Triangle above the diagonal
142  // Indices of the vertices
143  std::vector<size_type> vertex_index_list_up{
144  v_idx[VertexIndex(i, j)], v_idx[VertexIndex(i + 1, j + 1)],
145  v_idx[VertexIndex(i, j + 1)]};
146  // Construct geometry

148  tria_geo_up << bottom_left_corner_[0] + i * hx,

150  bottom_left_corner_[0] + i * hx, bottom_left_corner_[1] + j * hy,
151  bottom_left_corner_[1] + (j + 1) * hy,
152  bottom_left_corner_[1] + (j + 1) * hy;
153  // Enroll the triangle entity
154  t_idx[tria_cnt] = mesh_factory_->AddEntity(
155  lf::base::RefEl::kTria(), vertex_index_list_up,
156  std::make_unique<geometry::TriaO1>(tria_geo_up));
157  // Triangle below the diagonal
158  // Indices of the vertices
159  std::vector<size_type> vertex_index_list_low{
160  v_idx[VertexIndex(i, j)], v_idx[VertexIndex(i + 1, j)],
161  v_idx[VertexIndex(i + 1, j + 1)]};
162  // Construct geometry
163  Eigen::Matrix<double, Eigen::Dynamic, 3> tria_geo_low(2, 3);
164  tria_geo_low << bottom_left_corner_[0] + i * hx,
165  bottom_left_corner_[0] + (i + 1) * hx,
166  bottom_left_corner_[0] + (i + 1) * hx,
167  bottom_left_corner_[1] + j * hy, bottom_left_corner_[1] + j * hy,
168  bottom_left_corner_[1] + (j + 1) * hy;
169  auto tria_geo_low_ptr = std::make_unique<geometry::TriaO1>(tria_geo_low);
170  // Generate the triangle entity
171  t_idx[tria_cnt + 1] = mesh_factory_->AddEntity(
172  lf::base::RefEl::kTria(), vertex_index_list_low,
173  std::make_unique<geometry::TriaO1>(tria_geo_low));
174  }
175  }
