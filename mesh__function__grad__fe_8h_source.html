97  std::vector<Eigen::Matrix<Scalar, Eigen::Dynamic, 1>> operator()(
98  const lf::mesh::Entity& e, const Eigen::MatrixXd& local) const {
99  // Access information on local shape functions for the entity
100  const unsigned int type_id = e.RefEl().Id();
101  LF_ASSERT_MSG(fe_[type_id] != nullptr, "Missing LSF information for " << e);
102  auto grad_sf_eval = fe_[type_id]->GradientsReferenceShapeFunctions(local);
103  // Fetch the coefficients of the global shape functions associated with the
104  // current mesh entity
105  Eigen::Matrix<SCALAR_COEFF, 1, Eigen::Dynamic> local_dofs(
106  1, grad_sf_eval.rows());
107  auto global_dofs = fe_space_->LocGlobMap().GlobalDofIndices(e);
108  for (Eigen::Index i = 0; i < grad_sf_eval.rows(); ++i) {
109  local_dofs(i) = dof_vector_(global_dofs[i]);
110  }
111  // gradients w.r.t. reference element coordinates \hat{x}
112  auto local_grads = (local_dofs * grad_sf_eval).eval();
113  // Transform to Cartesian coordinates
114  auto jac_t = e.Geometry()->JacobianInverseGramian(local);
115  auto dim_local = e.RefEl().Dimension();
116  std::vector<Eigen::Matrix<Scalar, Eigen::Dynamic, 1>> result(local.cols());
117  // Transform all the local gradients in the evaluation points
118  for (std::size_t i = 0; i < result.size(); ++i) {
119  result[i] = jac_t.block(0, dim_local * i, jac_t.rows(), dim_local) *
120  local_grads.block(0, i * dim_local, 1, dim_local).transpose();
121  }
122  return result;
123  }
