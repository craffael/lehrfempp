9 bool assertNonDegenerateQuad(
10  const Eigen::Matrix<double, Eigen::Dynamic, 4>& coords, double tol) {
11  // World dimension
12  const Geometry::dim_t wd = coords.rows();
13  // Length tests
14  double e0lensq = (coords.col(1) - coords.col(0)).squaredNorm();
15  double e1lensq = (coords.col(2) - coords.col(1)).squaredNorm();
16  double e2lensq = (coords.col(3) - coords.col(2)).squaredNorm();
17  double e3lensq = (coords.col(0) - coords.col(3)).squaredNorm();
18  // Test lengths of edges versus circumference.
19  double circum = e0lensq + e1lensq + e2lensq + e3lensq;
20  LF_VERIFY_MSG(e0lensq > tol * circum, "Collapsed edge 0");
21  LF_VERIFY_MSG(e1lensq > tol * circum, "Collapsed edge 1");
22  LF_VERIFY_MSG(e2lensq > tol * circum, "Collapsed edge 2");
23  LF_VERIFY_MSG(e3lensq > tol * circum, "Collapsed edge 3");
24  // Area test
25  switch (wd) {
26  case 2: {
27  double ar1 =
28  ((coords(0, 1) - coords(0, 0)) * (coords(1, 2) - coords(1, 0)) -
29  (coords(1, 1) - coords(1, 0)) * (coords(0, 2) - coords(0, 0)));
30  double ar2 =
31  ((coords(0, 3) - coords(0, 0)) * (coords(1, 2) - coords(1, 0)) -
32  (coords(1, 3) - coords(1, 0)) * (coords(0, 2) - coords(0, 0)));
33  double area = std::fabs(ar1) + std::fabs(ar2);
34  LF_VERIFY_MSG(area > tol * circum, "Degenerate 2D quad");
35  return true;
36  break;
37  }
38  case 3: {
39  const Eigen::Matrix<double, 3, 4> c3d(coords.block<3, 4>(0, 0));
40  double ar1 =
41  ((c3d.col(1) - c3d.col(0)).cross(c3d.col(2) - c3d.col(0))).norm();
42  double ar2 =
43  ((c3d.col(3) - c3d.col(0)).cross(c3d.col(2) - c3d.col(0))).norm();
44  double area = ar1 + ar2;
45  LF_VERIFY_MSG(area > tol * circum, "Degenerate 3D quad");
46  return true;
47  break;
48  }
49  default: {
50  LF_ASSERT_MSG(false, "Illegal world dimension" << wd);
51  break;
52  }
53  }
54  return false;
55 }

64 Eigen::MatrixXd QuadO1::Global(const Eigen::MatrixXd& local) const {
65  LF_ASSERT_MSG(local.rows() == 2, "reference coords must be 2-vectors");
66  // Componentwise bilinear transformation from unit square in a
67  // vectorized fashion.
68  // Note the use of array and matrix views offered by Eigen.
69  return coords_.col(0) *
70  ((1 - local.array().row(0)) * (1 - local.array().row(1)))
71  .matrix() +
72  coords_.col(1) *
73  (local.array().row(0) * (1 - local.array().row(1))).matrix() +
74  coords_.col(2) *
75  (local.array().row(0) * local.array().row(1)).matrix() +
76  coords_.col(3) *
77  ((1 - local.array().row(0)) * local.array().row(1)).matrix();
78 }

82 Eigen::MatrixXd QuadO1::Jacobian(const Eigen::MatrixXd& local) const {
83  Eigen::MatrixXd result(DimGlobal(), local.cols() * 2);
84 
85  // Note that coords\_ stores the coordinates of the vertices of
86  // the quadrilateral in its columns.
87  for (Eigen::Index i = 0; i < local.cols(); ++i) {
88  // Partial derivative of componentwise bilinear mapping
89  // w.r.t. to first reference coordinate
90  result.col(2 * i) = (coords_.col(1) - coords_.col(0)) * (1 - local(1, i)) +
91  (coords_.col(2) - coords_.col(3)) * local(1, i);
92  // Partial derivative of componentwise bilinear mapping
93  // w.r.t. to second reference coordinate
94  result.col(2 * i + 1) =
95  (coords_.col(3) - coords_.col(0)) * (1 - local(0, i)) +
96  (coords_.col(2) - coords_.col(1)) * local(0, i);
97  }
98  return result;
99 }

103 Eigen::MatrixXd QuadO1::JacobianInverseGramian(
104  const Eigen::MatrixXd& local) const {
105  Eigen::MatrixXd result(DimGlobal(), local.cols() * 2);
106  Eigen::MatrixXd jacobian(DimGlobal(), 2);
107 
108  // Loop over all evaluatin points
109  for (Eigen::Index i = 0; i < local.cols(); ++i) {
110  // Compute Jacobian matrix in one evaluation point.
111  jacobian.col(0) = (coords_.col(1) - coords_.col(0)) * (1 - local(1, i)) +
112  (coords_.col(2) - coords_.col(3)) * local(1, i);
113  jacobian.col(1) = (coords_.col(3) - coords_.col(0)) * (1 - local(0, i)) +
114  (coords_.col(2) - coords_.col(1)) * local(0, i);
115  // Distinguish whether the cell lies in a planar mesh or a surface mesh
116  if (DimGlobal() == 2) {
117  // Standard case: 2D planar mesh
118  // Eigen has a built-in function for computing the inverse of a small
119  // matrix
120  result.block(0, 2 * i, DimGlobal(), 2) = jacobian.transpose().inverse();
121  } else {
122  result.block(0, 2 * i, DimGlobal(), 2) = (jacobian.transpose() * jacobian)
123  .colPivHouseholderQr()
124  .solve(jacobian.transpose())
125  .transpose();
126  }
127  }
128  return result;
129 }

133 Eigen::VectorXd QuadO1::IntegrationElement(const Eigen::MatrixXd& local) const {
134  Eigen::VectorXd result(local.cols());
135  Eigen::MatrixXd jacobian(DimGlobal(), 2);
136 
137  // Loop over all evaluatin points
138  for (Eigen::Index i = 0; i < local.cols(); ++i) {
139  // Compute Jacobian matrix in one evaluation point.
140  jacobian.col(0) = (coords_.col(1) - coords_.col(0)) * (1 - local(1, i)) +
141  (coords_.col(2) - coords_.col(3)) * local(1, i);
142  jacobian.col(1) = (coords_.col(3) - coords_.col(0)) * (1 - local(0, i)) +
143  (coords_.col(2) - coords_.col(1)) * local(0, i);
144  // For planar cell, simply the determinant, for a cell in 3D space, the
145  // volume form
146  if (DimGlobal() == 2) {
147  result(i) = std::abs(jacobian.determinant());
148  } else {
149  result(i) = std::sqrt((jacobian.transpose() * jacobian).determinant());
150  }
151  }
152  return result;
153 }

177 std::vector<std::unique_ptr<Geometry>> QuadO1::ChildGeometry(
178  const RefinementPattern& ref_pat, base::dim_t codim) const {
179  // The refinement pattern must be for a quadrilateral
180  LF_VERIFY_MSG(ref_pat.RefEl() == lf::base::RefEl::kQuad(),
181  "Refinement pattern for " << ref_pat.RefEl().ToString());
182  // Allowed condimensions:
183  // 0 -> child cells, 1-> child edges, 2 -> child points
184  LF_VERIFY_MSG(codim < 3, "Illegal codim " << codim);
185 
186  // Lattice meshwidth
187  const double h_lattice = 1.0 / static_cast<double>(ref_pat.LatticeConst());
188  // Obtain geometry of children as lattice polygons
189  std::vector<Eigen::Matrix<int, Eigen::Dynamic, Eigen::Dynamic>>
190  child_polygons(ref_pat.ChildPolygons(codim));
191  // Number of child segments
192  const int no_children = child_polygons.size();
193  // Check consistency of data
194  LF_ASSERT_MSG(
195  no_children == ref_pat.NumChildren(codim),
196  "no_children = " << no_children << " <-> " << ref_pat.NumChildren(codim));
197 
198  // Variable for returning (unique pointers to) child geometries
199  std::vector<std::unique_ptr<Geometry>> child_geo_uptrs{};
200  // For each child entity create a geometry object and a unique pointer to it.
201  for (int l = 0; l < no_children; l++) {
202  // A single child entity is described by a lattice polygon with
203  // a certain number of corners
204  LF_VERIFY_MSG(
205  child_polygons[l].rows() == 2,
206  "child_polygons[" << l << "].rows() = " << child_polygons[l].rows());
207  // Obtain physical (world) coordinates of the vertices of
208  // of the child entities after normalizing lattice coordinates
209  const Eigen::MatrixXd child_geo(
210  Global(h_lattice * child_polygons[l].cast<double>()));
211  // Treat child entities of different dimension
212  switch (codim) {
213  case 0: { // Child cells
214  if (child_polygons[l].cols() == 3) {
215  // Child cell is a triangle
216  child_geo_uptrs.push_back(std::make_unique<TriaO1>(child_geo));
217  } else if (child_polygons[l].cols() == 4) {
218  // Child cell is a quadrilateral
219  child_geo_uptrs.push_back(std::make_unique<QuadO1>(child_geo));
220  } else {
221  LF_VERIFY_MSG(false, "child_polygons[" << l << "].cols() = "
222  << child_polygons[l].cols());
223  }
224  break;
225  }
226  case 1: {
227  // Child is an edge
228  LF_VERIFY_MSG(
229  child_polygons[l].cols() == 2,
230  "child_polygons[l].cols() = " << child_polygons[l].cols());
231  child_geo_uptrs.push_back(std::make_unique<SegmentO1>(child_geo));
232  break;
233  }
234  case 2: {
235  LF_VERIFY_MSG(
236  child_polygons[l].cols() == 1,
237  "child_polygons[l].cols() = " << child_polygons[l].cols());
238  child_geo_uptrs.push_back(std::make_unique<Point>(child_geo));
239  break;
240  }
241  default: {
242  LF_VERIFY_MSG(false, "Illegal co-dimension");
243  break;
244  }
245  } // end switch codim
246  } // end loop over children
247  return (child_geo_uptrs);
248 } // end ChildGeometry()

286 void Parallelogram::init() {
287  jacobian_ << coords_.col(1) - coords_.col(0), coords_.col(3) - coords_.col(0);
288  // Distinguish between different world dimensions
289  if (coords_.rows() == 2) {
290  // 2D case: Simpler formula!
291  jacobian_inverse_gramian_ = jacobian_.transpose().inverse();
292  integrationElement_ = std::abs(jacobian_.determinant());
293  } else {
294  // 3D case: complicated formula
295  jacobian_inverse_gramian_ = Eigen::MatrixXd(
296  jacobian_ * (jacobian_.transpose() * jacobian_).inverse());
297  integrationElement_ =
298  std::sqrt((jacobian_.transpose() * jacobian_).determinant());
299  }
300 } // end init()
